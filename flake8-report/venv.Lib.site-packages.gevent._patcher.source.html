162 def _import_all(self, queue):
163 # Called while monitoring for patch changes.
164 while queue:
165 module = queue.pop(0)
166 name = module.__name__
167 mod_all = tuple(getattr(module, '__all__', ())) + self.extra_all(name)
168 for attr_name in mod_all:
169 try:
170 getattr(module, attr_name)

172 module_name = module.__name__ + '.' + attr_name
173 new_module = g_import(module_name, {}, {}, attr_name)

179 extra_all=lambda mod_name: (),
180 after_import_hook=lambda module: None):
181 """
182 Import *module_name* with gevent monkey-patches active,
183 and return an object holding the greened module as *module*.
184  
185 Any sub-modules that were imported by the package are also
186 saved.
187  
188 .. versionchanged:: 1.5a4
189 If the module defines ``__all__``, then each of those
190 attributes/modules is also imported as part of the same transaction,
191 recursively. The order of ``__all__`` is respected. Anything passed in
192 *extra_all* (which must be in the same namespace tree) is also imported.
193  
194 .. versionchanged:: 1.5a4
195 You must now do all patching for a given module tree
196 with one call to this method, or at least by using the returned
197 object.
198 """

200 with cached_platform_architecture():
201 # Save the current module state, and restore on exit,
202 # capturing desirable changes in the modules package.
203 patcher = _SysModulesPatcher(module_name, extra_all)

208 class cached_platform_architecture(object):
209 """
210 Context manager that caches ``platform.architecture``.
211  
212 Some things that load shared libraries (like Cryptodome, via
213 dnspython) invoke ``platform.architecture()`` for each one. That
214 in turn wants to fork and run commands , which in turn wants to
215 call ``threading._after_fork`` if the GIL has been initialized.
216 All of that means that certain imports done early may wind up
217 wanting to have the hub initialized potentially much earlier than
218 before.
219  
220 Part of the fix is to observe when that happens and delay
221 initializing parts of gevent until as late as possible (e.g., we
222 delay importing and creating the resolver until the hub needs it,
223 unless explicitly configured).
224  
225 The rest of the fix is to avoid the ``_after_fork`` issues by
226 first caching the results of platform.architecture before doing
227 patched imports.
228  
229 (See events.py for similar issues with platform, and
230 test__threading_2.py for notes about threading._after_fork if the
231 GIL has been initialized)
232 """
