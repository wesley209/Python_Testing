321 def _create_wheel_file(self, bdist_wheel):
322 from wheel.wheelfile import WheelFile
323  
324 dist_info = self.get_finalized_command("dist_info")
325 dist_name = dist_info.name
326 tag = "-".join(bdist_wheel.get_tag())
327 build_tag = "0.editable" # According to PEP 427 needs to start with digit
328 archive_name = f"{dist_name}-{build_tag}-{tag}.whl"
329 wheel_path = Path(self.dist_dir, archive_name)
330 if wheel_path.exists():
331 wheel_path.unlink()
332  
333 unpacked_wheel = TemporaryDirectory(suffix=archive_name)
334 build_lib = TemporaryDirectory(suffix=".build-lib")
335 build_tmp = TemporaryDirectory(suffix=".build-temp")
336  
337 with unpacked_wheel as unpacked, build_lib as lib, build_tmp as tmp:
338 unpacked_dist_info = Path(unpacked, Path(self.dist_info_dir).name)
339 shutil.copytree(self.dist_info_dir, unpacked_dist_info)
340 self._install_namespaces(unpacked, dist_name)
341 files, mapping = self._run_build_commands(dist_name, unpacked, lib, tmp)
342 strategy = self._select_strategy(dist_name, tag, lib)
343 with strategy, WheelFile(wheel_path, "w") as wheel_obj:
344 strategy(wheel_obj, files, mapping)
345 wheel_obj.write_files(unpacked)
346  
347 return wheel_path

355 def _select_strategy(
356 self,
357 name: str,
358 tag: str,
359 build_lib: StrPath,
360 ) -> EditableStrategy:
361 """Decides which strategy to use to implement an editable installation."""
362 build_name = f"__editable__.{name}-{tag}"
363 project_dir = Path(self.project_dir)
364 mode = _EditableMode.convert(self.mode)
365  
366 if mode is _EditableMode.STRICT:
367 auxiliary_dir = _empty_dir(Path(self.project_dir, "build", build_name))
368 return _LinkTree(self.distribution, name, auxiliary_dir, build_lib)
369  
370 packages = _find_packages(self.distribution)
371 has_simple_layout = _simple_layout(packages, self.package_dir, project_dir)
372 is_compat_mode = mode is _EditableMode.COMPAT
373 if set(self.package_dir) == {""} and has_simple_layout or is_compat_mode:
374 # src-layout(ish) is relatively safe for a simple pth file
375 src_dir = self.package_dir.get("", ".")
376 return _StaticPth(self.distribution, name, [Path(project_dir, src_dir)])
377  
378 # Use a MetaPathFinder to avoid adding accidental top-level packages/modules
379 return _TopLevelFinder(self.distribution, name)

424 class _LinkTree(_StaticPth):
425 """
426 Creates a ``.pth`` file that points to a link tree in the ``auxiliary_dir``.
427  
428 This strategy will only link files (not dirs), so it can be implemented in
429 any OS, even if that means using hardlinks instead of symlinks.
430  
431 By collocating ``auxiliary_dir`` and the original source code, limitations
432 with hardlinks should be avoided.
433 """
434  
435 def __init__(
436 self,
437 dist: Distribution,
438 name: str,
439 auxiliary_dir: StrPath,
440 build_lib: StrPath,
441 ):
442 self.auxiliary_dir = Path(auxiliary_dir)
443 self.build_lib = Path(build_lib).resolve()
444 self._file = dist.get_command_obj("build_py").copy_file
445 super().__init__(dist, name, [self.auxiliary_dir])
446  
447 def __call__(self, wheel: WheelFile, files: list[str], mapping: dict[str, str]):
448 self._create_links(files, mapping)
449 super().__call__(wheel, files, mapping)

451 def _normalize_output(self, file: str) -> str | None:
452 # Files relative to build_lib will be normalized to None
453 with suppress(ValueError):
454 path = Path(file).resolve().relative_to(self.build_lib)
455 return str(path).replace(os.sep, '/')
456 return None

458 def _create_file(self, relative_output: str, src_file: str, link=None):
459 dest = self.auxiliary_dir / relative_output
460 if not dest.parent.is_dir():
461 dest.parent.mkdir(parents=True)
462 self._file(src_file, dest, link=link)
463  
464 def _create_links(self, outputs, output_mapping):
465 self.auxiliary_dir.mkdir(parents=True, exist_ok=True)
466 link_type = "sym" if _can_symlink_files(self.auxiliary_dir) else "hard"
467 normalised = ((self._normalize_output(k), v) for k, v in output_mapping.items())
468 # remove files that are not relative to build_lib
469 mappings = {k: v for k, v in normalised if k is not None}
470  
471 for output in outputs:
472 relative = self._normalize_output(output)
473 if relative and relative not in mappings:
474 self._create_file(relative, output)
475  
476 for relative, src in mappings.items():
477 self._create_file(relative, src, link=link_type)

563 def _encode_pth(content: str) -> bytes:
564 """
565 Prior to Python 3.13 (see https://github.com/python/cpython/issues/77102),
566 .pth files are always read with 'locale' encoding, the recommendation
567 from the cpython core developers is to write them as ``open(path, "w")``
568 and ignore warnings (see python/cpython#77102, pypa/setuptools#3937).
569 This function tries to simulate this behaviour without having to create an
570 actual file, in a way that supports a range of active Python versions.
571 (There seems to be some variety in the way different version of Python handle
572 ``encoding=None``, not all of them use ``locale.getpreferredencoding(False)``
573 or ``locale.getencoding()``).
574 """
575 with io.BytesIO() as buffer:
576 wrapper = io.TextIOWrapper(buffer, encoding=py312.PTH_ENCODING)
577 # TODO: Python 3.13 replace the whole function with `bytes(content, "utf-8")`
578 wrapper.write(content)
579 wrapper.flush()
580 buffer.seek(0)
581 return buffer.read()

584 def _can_symlink_files(base_dir: Path) -> bool:
585 with TemporaryDirectory(dir=str(base_dir.resolve())) as tmp:
586 path1, path2 = Path(tmp, "file1.txt"), Path(tmp, "file2.txt")
587 path1.write_text("file1", encoding="utf-8")
588 with suppress(AttributeError, NotImplementedError, OSError):
589 os.symlink(path1, path2)
590 if path2.is_symlink() and path2.read_text(encoding="utf-8") == "file1":
591 return True
592  
593 try:
594 os.link(path1, path2) # Ensure hard links can be created
595 except Exception as ex:
596 msg = (
597 "File system does not seem to support either symlinks or hard links. "
598 "Strict editable installs require one of them to be supported."
599 )
600 raise LinksNotSupported(msg) from ex
601 return False

604 def _simple_layout(
605 packages: Iterable[str], package_dir: dict[str, str], project_dir: StrPath
606 ) -> bool:
607 """Return ``True`` if:
608 - all packages are contained by the same parent directory, **and**
609 - all packages become importable if the parent directory is added to ``sys.path``.
610  
611 >>> _simple_layout(['a'], {"": "src"}, "/tmp/myproj")
612 True
613 >>> _simple_layout(['a', 'a.b'], {"": "src"}, "/tmp/myproj")
614 True
615 >>> _simple_layout(['a', 'a.b'], {}, "/tmp/myproj")
616 True
617 >>> _simple_layout(['a', 'a.a1', 'a.a1.a2', 'b'], {"": "src"}, "/tmp/myproj")
618 True
619 >>> _simple_layout(['a', 'a.a1', 'a.a1.a2', 'b'], {"a": "a", "b": "b"}, ".")
620 True
621 >>> _simple_layout(['a', 'a.a1', 'a.a1.a2', 'b'], {"a": "_a", "b": "_b"}, ".")
622 False
623 >>> _simple_layout(['a', 'a.a1', 'a.a1.a2', 'b'], {"a": "_a"}, "/tmp/myproj")
624 False
625 >>> _simple_layout(['a', 'a.a1', 'a.a1.a2', 'b'], {"a.a1.a2": "_a2"}, ".")
626 False
627 >>> _simple_layout(['a', 'a.b'], {"": "src", "a.b": "_ab"}, "/tmp/myproj")
628 False
629 >>> # Special cases, no packages yet:
630 >>> _simple_layout([], {"": "src"}, "/tmp/myproj")
631 True
632 >>> _simple_layout([], {"a": "_a", "": "src"}, "/tmp/myproj")
633 False
634 """
635 layout = {pkg: find_package_path(pkg, package_dir, project_dir) for pkg in packages}
636 if not layout:
637 return set(package_dir) in ({}, {""})
638 parent = os.path.commonpath(starmap(_parent_path, layout.items()))
639 return all(
640 _path.same_path(Path(parent, *key.split('.')), value)
641 for key, value in layout.items()
642 )

645 def _parent_path(pkg, pkg_path):
646 """Infer the parent path containing a package, that if added to ``sys.path`` would
647 allow importing that package.
648 When ``pkg`` is directly mapped into a directory with a different name, return its
649 own path.
650 >>> _parent_path("a", "src/a")
651 'src'
652 >>> _parent_path("b", "src/c")
653 'src/c'
654 """
655 parent = pkg_path[: -len(pkg)] if pkg_path.endswith(pkg) else pkg_path
656 return parent.rstrip("/" + os.sep)

659 def _find_packages(dist: Distribution) -> Iterator[str]:
660 yield from iter(dist.packages or [])
661  
662 py_modules = dist.py_modules or []
663 nested_modules = [mod for mod in py_modules if "." in mod]
664 if dist.ext_package:
665 yield dist.ext_package
666 else:
667 ext_modules = dist.ext_modules or []
668 nested_modules += [x.name for x in ext_modules if "." in x.name]
669  
670 for module in nested_modules:
671 package, _, _ = module.rpartition(".")
672 yield package

675 def _find_top_level_modules(dist: Distribution) -> Iterator[str]:
676 py_modules = dist.py_modules or []
677 yield from (mod for mod in py_modules if "." not in mod)
678  
679 if not dist.ext_package:
680 ext_modules = dist.ext_modules or []
681 yield from (x.name for x in ext_modules if "." not in x.name)

687 src_root: StrPath,

697 def _absolute_root(path: StrPath) -> str:
698 """Works for packages and top-level modules"""
699 path_ = Path(path)
700 parent = path_.parent
701  
702 if path_.exists():
703 return str(path_.resolve())
704 else:
705 return str(parent.resolve() / path_.name)

708 def _find_virtual_namespaces(pkg_roots: dict[str, str]) -> Iterator[str]:
709 """By carefully designing ``package_dir``, it is possible to implement the logical
710 structure of PEP 420 in a package without the corresponding directories.
711  
712 Moreover a parent package can be purposefully/accidentally skipped in the discovery
713 phase (e.g. ``find_packages(include=["mypkg.*"])``, when ``mypkg.foo`` is included
714 by ``mypkg`` itself is not).
715 We consider this case to also be a virtual namespace (ignoring the original
716 directory) to emulate a non-editable installation.
717  
718 This function will try to find these kinds of namespaces.
719 """
720 for pkg in pkg_roots:
721 if "." not in pkg:
722 continue
723 parts = pkg.split(".")
724 for i in range(len(parts) - 1, 0, -1):
725 partial_name = ".".join(parts[:i])
726 path = Path(find_package_path(partial_name, pkg_roots, ""))
727 if not path.exists() or partial_name not in pkg_roots:
728 # partial_name not in pkg_roots ==> purposefully/accidentally skipped
729 yield partial_name

732 def _find_namespaces(
733 packages: list[str], pkg_roots: dict[str, str]
734 ) -> Iterator[tuple[str, list[str]]]:
735 for pkg in packages:
736 path = find_package_path(pkg, pkg_roots, "")
737 if Path(path).exists() and not Path(path, "__init__.py").exists():
738 yield (pkg, [path])

741 def _remove_nested(pkg_roots: dict[str, str]) -> dict[str, str]:
742 output = dict(pkg_roots.copy())
743  
744 for pkg, path in reversed(list(pkg_roots.items())):
745 if any(
746 pkg != other and _is_nested(pkg, path, other, other_path)
747 for other, other_path in pkg_roots.items()
748 ):
749 output.pop(pkg)
750  
751 return output

754 def _is_nested(pkg: str, pkg_path: str, parent: str, parent_path: str) -> bool:
755 """
756 Return ``True`` if ``pkg`` is nested inside ``parent`` both logically and in the
757 file system.
758 >>> _is_nested("a.b", "path/a/b", "a", "path/a")
759 True
760 >>> _is_nested("a.b", "path/a/b", "a", "otherpath/a")
761 False
762 >>> _is_nested("a.b", "path/a/b", "c", "path/c")
763 False
764 >>> _is_nested("a.a", "path/a/a", "a", "path/a")
765 True
766 >>> _is_nested("b.a", "path/b/a", "a", "path/a")
767 False
768 """
769 norm_pkg_path = _path.normpath(pkg_path)
770 rest = pkg.replace(parent, "", 1).strip(".").split(".")
771 return pkg.startswith(parent) and norm_pkg_path == _path.normpath(
772 Path(parent_path, *rest)
773 )

801 _FINDER_TEMPLATE = """\
802 from __future__ import annotations
803 import sys
804 from importlib.machinery import ModuleSpec, PathFinder
805 from importlib.machinery import all_suffixes as module_suffixes
806 from importlib.util import spec_from_file_location
807 from itertools import chain
808 from pathlib import Path
809  
810 MAPPING: dict[str, str] = {mapping!r}
811 NAMESPACES: dict[str, list[str]] = {namespaces!r}
812 PATH_PLACEHOLDER = {name!r} + ".__path_hook__"
813  
814  
815 class _EditableFinder: # MetaPathFinder
816 @classmethod
817 def find_spec(cls, fullname: str, path=None, target=None) -> ModuleSpec | None: # type: ignore
818 # Top-level packages and modules (we know these exist in the FS)
819 if fullname in MAPPING:
820 pkg_path = MAPPING[fullname]
821 return cls._find_spec(fullname, Path(pkg_path))
822  
823 # Handle immediate children modules (required for namespaces to work)
824 # To avoid problems with case sensitivity in the file system we delegate
825 # to the importlib.machinery implementation.
826 parent, _, child = fullname.rpartition(".")
827 if parent and parent in MAPPING:
828 return PathFinder.find_spec(fullname, path=[MAPPING[parent]])
829  
830 # Other levels of nesting should be handled automatically by importlib
831 # using the parent path.
832 return None
833  
834 @classmethod
835 def _find_spec(cls, fullname: str, candidate_path: Path) -> ModuleSpec | None:
836 init = candidate_path / "__init__.py"
837 candidates = (candidate_path.with_suffix(x) for x in module_suffixes())
838 for candidate in chain([init], candidates):
839 if candidate.exists():
840 return spec_from_file_location(fullname, candidate)
841 return None
842  
843  
844 class _EditableNamespaceFinder: # PathEntryFinder
845 @classmethod
846 def _path_hook(cls, path) -> type[_EditableNamespaceFinder]:
847 if path == PATH_PLACEHOLDER:
848 return cls
849 raise ImportError
850  
851 @classmethod
852 def _paths(cls, fullname: str) -> list[str]:
853 paths = NAMESPACES[fullname]
854 if not paths and fullname in MAPPING:
855 paths = [MAPPING[fullname]]
856 # Always add placeholder, for 2 reasons:
857 # 1. __path__ cannot be empty for the spec to be considered namespace.
858 # 2. In the case of nested namespaces, we need to force
859 # import machinery to query _EditableNamespaceFinder again.
860 return [*paths, PATH_PLACEHOLDER]
861  
862 @classmethod
863 def find_spec(cls, fullname: str, target=None) -> ModuleSpec | None: # type: ignore
864 if fullname in NAMESPACES:
865 spec = ModuleSpec(fullname, None, is_package=True)
866 spec.submodule_search_locations = cls._paths(fullname)
867 return spec
868 return None
869  
870 @classmethod
871 def find_module(cls, _fullname) -> None:
872 return None
873  
874  
875 def install():
876 if not any(finder == _EditableFinder for finder in sys.meta_path):
877 sys.meta_path.append(_EditableFinder)
878  
879 if not NAMESPACES:
880 return
881  
882 if not any(hook == _EditableNamespaceFinder._path_hook for hook in sys.path_hooks):
883 # PathEntryFinder is needed to create NamespaceSpec without private APIS
884 sys.path_hooks.append(_EditableNamespaceFinder._path_hook)
885 if PATH_PLACEHOLDER not in sys.path:
886 sys.path.append(PATH_PLACEHOLDER) # Used just to trigger the path hook
887 """

890 def _finder_template(
891 name: str, mapping: Mapping[str, str], namespaces: dict[str, list[str]]
892 ) -> str:
893 """Create a string containing the code for the``MetaPathFinder`` and
894 ``PathEntryFinder``.
895 """
896 mapping = dict(sorted(mapping.items(), key=lambda p: p[0]))
897 return _FINDER_TEMPLATE.format(name=name, mapping=mapping, namespaces=namespaces)
