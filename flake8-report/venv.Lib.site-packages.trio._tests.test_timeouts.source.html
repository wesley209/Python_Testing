26 async def check_takes_about(f: Callable[[], Awaitable[T]], expected_dur: float) -> T:
27 start = time.perf_counter()
28 result = await outcome.acapture(f)
29 dur = time.perf_counter() - start
30 print(dur / expected_dur)
31 # 1.5 is an arbitrary fudge factor because there's always some delay
32 # between when we become eligible to wake up and when we actually do. We
33 # used to sleep for 0.05, and regularly observed overruns of 1.6x on
34 # Appveyor, and then started seeing overruns of 2.3x on Travis's macOS, so
35 # now we bumped up the sleep to 1 second, marked the tests as slow, and
36 # hopefully now the proportional error will be less huge.
37 #
38 # We also also for durations that are a hair shorter than expected. For
39 # example, here's a run on Windows where a 1.0 second sleep was measured
40 # to take 0.9999999999999858 seconds:
41 # https://ci.appveyor.com/project/njsmith/trio/build/1.0.768/job/3lbdyxl63q3h9s21
42 # I believe that what happened here is that Windows's low clock resolution
43 # meant that our calls to time.monotonic() returned exactly the same
44 # values as the calls inside the actual run loop, but the two subtractions
45 # returned slightly different values because the run loop's clock adds a
46 # random floating point offset to both times, which should cancel out, but
47 # lol floating point we got slightly different rounding errors. (That
48 # value above is exactly 128 ULPs below 1.0, which would make sense if it
49 # started as a 1 ULP error at a different dynamic range.)
50 assert (1 - 1e-8) <= (dur / expected_dur) < 1.5
51  
52 return result.unwrap()
