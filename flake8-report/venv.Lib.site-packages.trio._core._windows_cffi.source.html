482 def _handle(obj: int | CData) -> Handle:
483 # For now, represent handles as either cffi HANDLEs or as ints. If you
484 # try to pass in a file descriptor instead, it's not going to work
485 # out. (For that msvcrt.get_osfhandle does the trick, but I don't know if
486 # we'll actually need that for anything...) For sockets this doesn't
487 # matter, Python never allocates an fd. So let's wait until we actually
488 # encounter the problem before worrying about it.
489 if isinstance(obj, int):
490 return Handle(ffi.cast("HANDLE", obj))
491 return Handle(obj)

499 def raise_winerror(
500 winerror: int | None = None,
501 *,
502 filename: str | None = None,
503 filename2: str | None = None,
504 ) -> NoReturn:
505 # assert sys.platform == "win32" # TODO: make this work in MyPy
506 # ... in the meanwhile, ffi.getwinerror() is undefined on non-Windows, necessitating the type
507 # ignores.
508  
509 if winerror is None:
510 err = ffi.getwinerror() # type: ignore[attr-defined,unused-ignore]
511 if err is None:
512 raise RuntimeError("No error set?")
513 winerror, msg = err
514 else:
515 err = ffi.getwinerror(winerror) # type: ignore[attr-defined,unused-ignore]
516 if err is None:
517 raise RuntimeError("No error set?")
518 _, msg = err
519 # https://docs.python.org/3/library/exceptions.html#OSError
520 raise OSError(0, msg, filename, winerror, filename2)
