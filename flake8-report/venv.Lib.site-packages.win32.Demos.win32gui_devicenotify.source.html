19 # WM_DEVICECHANGE message handler.
20 def OnDeviceChange(hwnd, msg, wp, lp):
21 # Unpack the 'lp' into the appropriate DEV_BROADCAST_* structure,
22 # using the self-identifying data inside the DEV_BROADCAST_HDR.
23 info = win32gui_struct.UnpackDEV_BROADCAST(lp)
24 print("Device change notification:", wp, str(info))
25 if (
26 wp == win32con.DBT_DEVICEQUERYREMOVE
27 and info.devicetype == win32con.DBT_DEVTYP_HANDLE
28 ):
29 # Our handle is stored away in the structure - just close it
30 print("Device being removed - closing handle")
31 win32file.CloseHandle(info.handle)
32 # and cancel our notifications - if it gets plugged back in we get
33 # the same notification and try and close the same handle...
34 win32gui.UnregisterDeviceNotification(info.hdevnotify)
35 return True
