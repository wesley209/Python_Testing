10 def testPyTimeFormat(self):
11 struct_current = time.localtime()
12 pytime_current = pywintypes.Time(struct_current)
13 # try and test all the standard parts of the format
14 # Note we used to include '%Z' testing, but that was pretty useless as
15 # it always returned the local timezone.
16 format_strings = "%a %A %b %B %c %d %H %I %j %m %M %p %S %U %w %W %x %X %y %Y"
17 for fmt in format_strings.split():
18 v1 = pytime_current.Format(fmt)
19 v2 = time.strftime(fmt, struct_current)
20 self.assertEqual(v1, v2, f"format {fmt} failed - {v1!r} != {v2!r}")
