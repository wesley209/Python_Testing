45 def select_random_ports(n):
46 """Select and return n random ports that are available."""
47 ports = []
48 sockets = []
49 for i in range(n):
50 sock = socket.socket()
51 sock.bind(('', 0))
52 ports.append(sock.getsockname()[1])
53 sockets.append(sock)
54 for sock in sockets:
55 sock.close()
56 return ports

65 def try_passwordless_ssh(server, keyfile, paramiko=None):
66 """Attempt to make an ssh connection without a password.
67 This is mainly used for requiring password input only once
68 when many tunnels may be connected to the same server.
69  
70 If paramiko is None, the default for the platform is chosen.
71 """
72 if paramiko is None:
73 paramiko = sys.platform == 'win32'
74 if not paramiko:
75 f = _try_passwordless_openssh
76 else:
77 f = _try_passwordless_paramiko
78 return f(server, keyfile)

81 def _try_passwordless_openssh(server, keyfile):
82 """Try passwordless login with shell ssh command."""
83 if pexpect is None:
84 raise ImportError("pexpect unavailable, use paramiko")
85 cmd = 'ssh -f ' + server
86 if keyfile:
87 cmd += ' -i ' + keyfile
88 cmd += ' exit'
89  
90 # pop SSH_ASKPASS from env
91 env = os.environ.copy()
92 env.pop('SSH_ASKPASS', None)
93  
94 ssh_newkey = 'Are you sure you want to continue connecting'
95 p = pexpect.spawn(cmd, env=env)
96  
97 MAX_RETRY = 10
98  
99 for _ in range(MAX_RETRY):
100 try:
101 i = p.expect([ssh_newkey, _password_pat], timeout=0.1)
102 if i == 0:
103 raise SSHException(
104 'The authenticity of the host can\'t be established.'
105 )
106 except pexpect.TIMEOUT:
107 continue
108 except pexpect.EOF:
109 return True
110 else:
111 return False
112  
113 raise MaxRetryExceeded(f"Failed after {MAX_RETRY} attempts")

116 def _try_passwordless_paramiko(server, keyfile):
117 """Try passwordless login with paramiko."""
118 if paramiko is None:
119 msg = "Paramiko unavailable, "
120 if sys.platform == 'win32':
121 msg += "Paramiko is required for ssh tunneled connections on Windows."
122 else:
123 msg += "use OpenSSH."
124 raise ImportError(msg)
125 username, server, port = _split_server(server)
126 client = paramiko.SSHClient()
127 known_hosts = os.path.expanduser("~/.ssh/known_hosts")
128 try:
129 client.load_host_keys(known_hosts)
130 except FileNotFoundError:
131 pass
132  
133 policy_name = os.environ.get("PYZMQ_PARAMIKO_HOST_KEY_POLICY", None)
134 if policy_name:
135 policy = getattr(paramiko, f"{policy_name}Policy")
136 client.set_missing_host_key_policy(policy())
137 try:
138 client.connect(
139 server, port, username=username, key_filename=keyfile, look_for_keys=True
140 )
141 except paramiko.AuthenticationException:
142 return False
143 else:
144 client.close()
145 return True

149 socket, addr, server, keyfile=None, password=None, paramiko=None, timeout=60
150 ):
151 """Connect a socket to an address via an ssh tunnel.
152  
153 This is a wrapper for socket.connect(addr), when addr is not accessible
154 from the local machine. It simply creates an ssh tunnel using the remaining args,
155 and calls socket.connect('tcp://localhost:lport') where lport is the randomly
156 selected local port of the tunnel.
157  
158 """
159 new_url, tunnel = open_tunnel(
160 addr,
161 server,
162 keyfile=keyfile,
163 password=password,
164 paramiko=paramiko,
165 timeout=timeout,
166 )
167 socket.connect(new_url)
168 return tunnel

171 def open_tunnel(addr, server, keyfile=None, password=None, paramiko=None, timeout=60):
172 """Open a tunneled connection from a 0MQ url.
173  
174 For use inside tunnel_connection.
175  
176 Returns
177 -------
178  
179 (url, tunnel) : (str, object)
180 The 0MQ url that has been forwarded, and the tunnel object
181 """
182  
183 lport = select_random_ports(1)[0]
184 transport, addr = addr.split('://')
185 ip, rport = addr.split(':')
186 rport = int(rport)
187 if paramiko is None:
188 paramiko = sys.platform == 'win32'
189 if paramiko:
190 tunnelf = paramiko_tunnel

192 tunnelf = openssh_tunnel

195 lport,
196 rport,
197 server,
198 remoteip=ip,
199 keyfile=keyfile,
200 password=password,
201 timeout=timeout,
202 )
203 return f'tcp://127.0.0.1:{lport}', tunnel

206 def openssh_tunnel(
207 lport, rport, server, remoteip='127.0.0.1', keyfile=None, password=None, timeout=60
208 ):
209 """Create an ssh tunnel using command-line ssh that connects port lport
210 on this machine to localhost:rport on server. The tunnel
211 will automatically close when not in use, remaining open
212 for a minimum of timeout seconds for an initial connection.
213  
214 This creates a tunnel redirecting `localhost:lport` to `remoteip:rport`,
215 as seen from `server`.
216  
217 keyfile and password may be specified, but ssh config is checked for defaults.
218  
219 Parameters
220 ----------
221  
222 lport : int
223 local port for connecting to the tunnel from this machine.
224 rport : int
225 port on the remote machine to connect to.
226 server : str
227 The ssh server to connect to. The full ssh server string will be parsed.
228 user@server:port
229 remoteip : str [Default: 127.0.0.1]
230 The remote ip, specifying the destination of the tunnel.
231 Default is localhost, which means that the tunnel would redirect
232 localhost:lport on this machine to localhost:rport on the *server*.
233  
234 keyfile : str; path to private key file
235 This specifies a key to be used in ssh login, default None.
236 Regular default ssh keys will be used without specifying this argument.
237 password : str;
238 Your ssh password to the ssh server. Note that if this is left None,
239 you will be prompted for it if passwordless key based login is unavailable.
240 timeout : int [default: 60]
241 The time (in seconds) after which no activity will result in the tunnel
242 closing. This prevents orphaned tunnels from running forever.
243 """
244 if pexpect is None:
245 raise ImportError("pexpect unavailable, use paramiko_tunnel")
246 ssh = "ssh "
247 if keyfile:
248 ssh += "-i " + keyfile
249  
250 if ':' in server:
251 server, port = server.split(':')
252 ssh += f" -p {port}"
253  
254 cmd = f"{ssh} -O check {server}"
255 (output, exitstatus) = pexpect.run(cmd, withexitstatus=True)
256 if not exitstatus:
  • E203 Whitespace before ':'
257 pid = int(output[output.find(b"(pid=") + 5 : output.find(b")")])
258 cmd = f"{ssh} -O forward -L 127.0.0.1:{lport}:{remoteip}:{rport} {server}"
259 (output, exitstatus) = pexpect.run(cmd, withexitstatus=True)
260 if not exitstatus:
261 atexit.register(_stop_tunnel, cmd.replace("-O forward", "-O cancel", 1))
262 return pid
263 cmd = f"{ssh} -f -S none -L 127.0.0.1:{lport}:{remoteip}:{rport} {server} sleep {timeout}"
264  
265 # pop SSH_ASKPASS from env
266 env = os.environ.copy()
267 env.pop('SSH_ASKPASS', None)
268  
269 ssh_newkey = 'Are you sure you want to continue connecting'
270 tunnel = pexpect.spawn(cmd, env=env)
271 failed = False
272 MAX_RETRY = 10
273 for _ in range(MAX_RETRY):
274 try:
275 i = tunnel.expect([ssh_newkey, _password_pat], timeout=0.1)
276 if i == 0:
277 raise SSHException(
278 'The authenticity of the host can\'t be established.'
279 )
280 except pexpect.TIMEOUT:
281 continue
282 except pexpect.EOF:
283 if tunnel.exitstatus:
284 print(tunnel.exitstatus)
285 print(tunnel.before)
286 print(tunnel.after)
287 raise RuntimeError(f"tunnel '{cmd}' failed to start")
288 else:
289 return tunnel.pid
290 else:
291 if failed:
292 print("Password rejected, try again")
293 password = None
294 if password is None:
295 password = getpass(f"{server}'s password: ")
296 tunnel.sendline(password)
297 failed = True
298 raise MaxRetryExceeded(f"Failed after {MAX_RETRY} attempts")

305 def _split_server(server):
306 if '@' in server:
307 username, server = server.split('@', 1)
308 else:
309 username = getuser()
310 if ':' in server:
311 server, port = server.split(':')
312 port = int(port)
313 else:
314 port = 22
315 return username, server, port

318 def paramiko_tunnel(
319 lport, rport, server, remoteip='127.0.0.1', keyfile=None, password=None, timeout=60
320 ):
321 """launch a tunner with paramiko in a subprocess. This should only be used
322 when shell ssh is unavailable (e.g. Windows).
323  
324 This creates a tunnel redirecting `localhost:lport` to `remoteip:rport`,
325 as seen from `server`.
326  
327 If you are familiar with ssh tunnels, this creates the tunnel:
328  
329 ssh server -L localhost:lport:remoteip:rport
330  
331 keyfile and password may be specified, but ssh config is checked for defaults.
332  
333  
334 Parameters
335 ----------
336  
337 lport : int
338 local port for connecting to the tunnel from this machine.
339 rport : int
340 port on the remote machine to connect to.
341 server : str
342 The ssh server to connect to. The full ssh server string will be parsed.
343 user@server:port
344 remoteip : str [Default: 127.0.0.1]
345 The remote ip, specifying the destination of the tunnel.
346 Default is localhost, which means that the tunnel would redirect
347 localhost:lport on this machine to localhost:rport on the *server*.
348  
349 keyfile : str; path to private key file
350 This specifies a key to be used in ssh login, default None.
351 Regular default ssh keys will be used without specifying this argument.
352 password : str;
353 Your ssh password to the ssh server. Note that if this is left None,
354 you will be prompted for it if passwordless key based login is unavailable.
355 timeout : int [default: 60]
356 The time (in seconds) after which no activity will result in the tunnel
357 closing. This prevents orphaned tunnels from running forever.
358  
359 """
360 if paramiko is None:
361 raise ImportError("Paramiko not available")
362  
363 if password is None:
364 if not _try_passwordless_paramiko(server, keyfile):
365 password = getpass(f"{server}'s password: ")
366  
367 p = Process(
368 target=_paramiko_tunnel,
369 args=(lport, rport, server, remoteip),
370 kwargs=dict(keyfile=keyfile, password=password),
371 )
372 p.daemon = True
373 p.start()
374 return p

400 # raise
