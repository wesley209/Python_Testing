1 """login -- PythonWin user ID and password dialog box
2  
3 (Adapted from originally distributed with Mark Hammond's PythonWin -
4 this now replaces it!)
5  
6 login.GetLogin() displays a modal "OK/Cancel" dialog box with input
7 fields for a user ID and password. The password field input is masked
8 with *'s. GetLogin takes two optional parameters, a window title, and a
9 default user ID. If these parameters are omitted, the title defaults to
10 "Login", and the user ID is left blank. GetLogin returns a (userid, password)
11 tuple. GetLogin can be called from scripts running on the console - i.e. you
12 don't need to write a full-blown GUI app to use it.
13  
14 login.GetPassword() is similar, except there is no username field.
15  
16 Example:
17 import pywin.dialogs.login
18 title = "FTP Login"
19 def_user = "fred"
20 userid, password = pywin.dialogs.login.GetLogin(title, def_user)
21  
22 Jim Eggleston, 28 August 1996
23 Merged with dlgpass and moved to pywin.dialogs by Mark Hammond Jan 1998.
24 """

104 def __init__(self, title):
105 dialog.Dialog.__init__(self, MakeLoginDlgTemplate(title))
106 self.AddDDX(win32ui.IDC_EDIT1, "userid")
107 self.AddDDX(win32ui.IDC_EDIT2, "password")

110 def GetLogin(title="Login", userid="", password=""):
111 d = LoginDlg(title)
112 d["userid"] = userid
113 d["password"] = password
114 if d.DoModal() != win32con.IDOK:
115 return (None, None)
116 else:
117 return (d["userid"], d["password"])
