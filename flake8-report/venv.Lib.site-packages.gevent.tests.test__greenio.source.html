47 def accept_close_early(listener):
48 # verify that the makefile and the socket are truly independent
49 # by closing the socket prior to using the made file
50 try:
51 conn, _ = listener.accept()
52 fd = conn.makefile(mode='wb')
53 conn.close()
54 fd.write(b'hello\n')
55 fd.close()
56 _write_to_closed(fd, b'a')
57 self.assertRaises(socket.error, conn.send, b'b')
58 finally:
59 listener.close()
60  
61 def accept_close_late(listener):
62 # verify that the makefile and the socket are truly independent
63 # by closing the made file and then sending a character
64 try:
65 conn, _ = listener.accept()
66 fd = conn.makefile(mode='wb')
67 fd.write(b'hello')
68 fd.close()
69 conn.send(b'\n')
70 conn.close()
71 _write_to_closed(fd, b'a')
72 self.assertRaises(socket.error, conn.send, b'b')
73 finally:
74 listener.close()
