173 def get_cors_headers(options, request_headers, request_method):
174 origins_to_set = get_cors_origins(options, request_headers.get('Origin'))
175 headers = MultiDict()
176  
177 if not origins_to_set: # CORS is not enabled for this route
178 return headers
179  
180 for origin in origins_to_set:
181 headers.add(ACL_ORIGIN, origin)
182  
183 headers[ACL_EXPOSE_HEADERS] = options.get('expose_headers')
184  
185 if options.get('supports_credentials'):
186 headers[ACL_CREDENTIALS] = 'true' # case sensitive
187  
188 if ACL_REQUEST_HEADER_PRIVATE_NETWORK in request_headers \
189 and request_headers.get(ACL_REQUEST_HEADER_PRIVATE_NETWORK) == 'true':
190 allow_private_network = 'true' if options.get('allow_private_network') else 'false'
191 headers[ACL_RESPONSE_PRIVATE_NETWORK] = allow_private_network
192  
193 # This is a preflight request
194 # http://www.w3.org/TR/cors/#resource-preflight-requests
195 if request_method == 'OPTIONS':
196 acl_request_method = request_headers.get(ACL_REQUEST_METHOD, '').upper()
197  
198 # If there is no Access-Control-Request-Method header or if parsing
199 # failed, do not set any additional headers
200 if acl_request_method and acl_request_method in options.get('methods'):
201  
202 # If method is not a case-sensitive match for any of the values in
203 # list of methods do not set any additional headers and terminate
204 # this set of steps.
205 headers[ACL_ALLOW_HEADERS] = get_allow_headers(options, request_headers.get(ACL_REQUEST_HEADERS))
206 headers[ACL_MAX_AGE] = options.get('max_age')
207 headers[ACL_METHODS] = options.get('methods')
208 else:
209 LOG.info("The request's Access-Control-Request-Method header does not match allowed methods. CORS headers will not be applied.")

211 # http://www.w3.org/TR/cors/#resource-implementation
212 if options.get('vary_header'):
213 # Only set header if the origin returned will vary dynamically,
214 # i.e. if we are not returning an asterisk, and there are multiple
215 # origins that can be matched.
216 if headers[ACL_ORIGIN] == '*':
217 pass

222
