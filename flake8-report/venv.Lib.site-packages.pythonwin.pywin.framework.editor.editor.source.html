247 def Indent(self):
248 """Insert an indent to move the cursor to the next tab position.
249  
250 Honors the tab size and 'use tabs' settings. Assumes the cursor is already at the
251 position to be indented, and the selection is a single character (ie, not a block)
252 """
253 start, end = self._obj_.GetSel()
254 startLine = self._obj_.LineFromChar(start)
255 line = self._obj_.GetLine(startLine)
256 realCol = start - self._obj_.LineIndex(startLine)
257 # Calulate the next tab stop.
258 # Expand existing tabs.
259 curCol = 0
260 for ch in line[:realCol]:
261 if ch == "\t":
262 curCol = ((curCol / self.tabSize) + 1) * self.tabSize
263 else:
264 curCol += 1
265 nextColumn = ((curCol / self.indentSize) + 1) * self.indentSize
266 # print("curCol is", curCol, "nextColumn is", nextColumn)
267 ins = None
268 if self.bSmartTabs:
269 # Look for some context.
270 if realCol == 0: # Start of the line - see if the line above can tell us
271 lookLine = startLine - 1
272 while lookLine >= 0:
273 check = self._obj_.GetLine(lookLine)[0:1]
274 if check in ("\t", " "):
275 ins = check
276 break
277 lookLine -= 1
278 else: # See if the previous char can tell us
279 check = line[realCol - 1]
280 if check in ("\t", " "):
281 ins = check
282  
283 # Either smart tabs off, or not smart enough!
284 # Use the "old style" settings.
285 if ins is None:
286 if self.bUseTabs and nextColumn % self.tabSize == 0:
287 ins = "\t"
288 else:
289 ins = " "
290  
291 if ins == " ":
292 # Calc the number of spaces to take us to the next stop
293 ins *= nextColumn - curCol
294  
295 self._obj_.ReplaceSel(ins)

297 def BlockDent(self, isIndent, startLine, endLine):
298 "Indent/Undent all lines specified"
299 if not self.GetDocument().CheckMakeDocumentWritable():
300 return 0
301 tabSize = self.tabSize # hard-code for now!
302 info = self._PrepareUserStateChange()
303 try:
304 for lineNo in range(startLine, endLine):
305 pos = self._obj_.LineIndex(lineNo)
306 self._obj_.SetSel(pos, pos)
307 if isIndent:
308 self.Indent()
309 else:
310 line = self._obj_.GetLine(lineNo)
311 try:
312 noToDel = 0
313 if line[0] == "\t":
314 noToDel = 1
315 elif line[0] == " ":
316 for noToDel in range(0, tabSize):
317 if line[noToDel] != " ":
318 break
319 else:
320 noToDel = tabSize
321 if noToDel:
322 self._obj_.SetSel(pos, pos + noToDel)
323 self._obj_.Clear()
324 except IndexError:
325 pass
326 finally:
327 self._EndUserStateChange(info)
328 self.GetDocument().SetModifiedFlag(1) # Now dirty
329 self._obj_.SetSel(self.LineIndex(startLine), self.LineIndex(endLine))
