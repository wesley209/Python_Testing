238 def virtual_memory():
239 """System virtual memory as a namedtuple."""
240 mem = cext.virtual_mem()
241 totphys, availphys, _totsys, _availsys = mem
242 total = totphys
243 avail = availphys
244 free = availphys
245 used = total - avail
246 percent = usage_percent((total - avail), total, round_=1)
247 return svmem(total, avail, percent, used, free)

250 def swap_memory():
251 """Swap system memory as a (total, used, free, sin, sout) tuple."""
252 mem = cext.virtual_mem()
253  
254 total_phys = mem[0]
255 total_system = mem[2]
256  
257 # system memory (commit total/limit) is the sum of physical and swap
258 # thus physical memory values need to be subtracted to get swap values
259 total = total_system - total_phys
260 # commit total is incremented immediately (decrementing free_system)
261 # while the corresponding free physical value is not decremented until
262 # pages are accessed, so we can't use free system memory for swap.
263 # instead, we calculate page file usage based on performance counter
264 if total > 0:
265 percentswap = cext.swap_percent()
266 used = int(0.01 * percentswap * total)
267 else:
268 percentswap = 0.0
269 used = 0
270  
271 free = total - used
272 percent = round(percentswap, 1)
273 return _common.sswap(total, used, free, percent, 0, 0)

379 def net_connections(kind, _pid=-1):
380 """Return socket connections. If pid == -1 return system-wide
381 connections (as opposed to connections opened by one process only).
382 """
383 if kind not in conn_tmap:
384 raise ValueError(
385 "invalid %r kind argument; choose between %s"
386 % (kind, ', '.join([repr(x) for x in conn_tmap]))
387 )
388 families, types = conn_tmap[kind]
389 rawlist = cext.net_connections(_pid, families, types)
390 ret = set()
391 for item in rawlist:
392 fd, fam, type, laddr, raddr, status, pid = item
393 nt = conn_to_ntuple(
394 fd,
395 fam,
396 type,
397 laddr,
398 raddr,
399 status,
400 TCP_STATUSES,
401 pid=pid if _pid == -1 else None,
402 )
403 ret.add(nt)
404 return list(ret)

445 def sensors_battery():
446 """Return battery information."""
447 # For constants meaning see:
448 # https://msdn.microsoft.com/en-us/library/windows/desktop/
449 # aa373232(v=vs.85).aspx
450 acline_status, flags, percent, secsleft = cext.sensors_battery()
451 power_plugged = acline_status == 1
452 no_battery = bool(flags & 128)
453 charging = bool(flags & 8)
454  
455 if no_battery:
456 return None
457 if power_plugged or charging:
458 secsleft = _common.POWER_TIME_UNLIMITED
459 elif secsleft == -1:
460 secsleft = _common.POWER_TIME_UNKNOWN
461  
462 return _common.sbattery(percent, secsleft, power_plugged)

564 @contextlib.contextmanager
565 def _wrap_exceptions(self):
566 """Ctx manager which translates bare OSError and WindowsError
567 exceptions into NoSuchProcess and AccessDenied.
568 """
569 try:
570 yield
571 except OSError as err:
572 if is_permission_err(err):
573 msg = (
574 "service %r is not querable (not enough privileges)"
575 % self._name
576 )
577 raise AccessDenied(pid=None, name=self._name, msg=msg)
578 elif err.winerror in (
579 cext.ERROR_INVALID_NAME,
580 cext.ERROR_SERVICE_DOES_NOT_EXIST,
581 ):
582 msg = "service %r does not exist" % self._name
583 raise NoSuchProcess(pid=None, name=self._name, msg=msg)
584 else:
585 raise

697 def is_permission_err(exc):
698 """Return True if this is a permission error."""
699 assert isinstance(exc, OSError), exc
700 if exc.errno in (errno.EPERM, errno.EACCES):
701 return True
702 # On Python 2 OSError doesn't always have 'winerror'. Sometimes
703 # it does, in which case the original exception was WindowsError
704 # (which is a subclass of OSError).
705 return getattr(exc, "winerror", -1) in (
706 cext.ERROR_ACCESS_DENIED,
707 cext.ERROR_PRIVILEGE_NOT_HELD,
708 )

711 def convert_oserror(exc, pid=None, name=None):
712 """Convert OSError into NoSuchProcess or AccessDenied."""
713 assert isinstance(exc, OSError), exc
714 if is_permission_err(exc):
715 return AccessDenied(pid=pid, name=name)
716 if exc.errno == errno.ESRCH:
717 return NoSuchProcess(pid=pid, name=name)
718 raise exc

721 def wrap_exceptions(fun):
722 """Decorator which converts OSError into NoSuchProcess or AccessDenied."""
723  
724 @functools.wraps(fun)
725 def wrapper(self, *args, **kwargs):
726 try:
727 return fun(self, *args, **kwargs)
728 except OSError as err:
729 raise convert_oserror(err, pid=self.pid, name=self._name)
730  
731 return wrapper

734 def retry_error_partial_copy(fun):
735 """Workaround for https://github.com/giampaolo/psutil/issues/875.
736 See: https://stackoverflow.com/questions/4457745#4457745.
737 """
738  
739 @functools.wraps(fun)
740 def wrapper(self, *args, **kwargs):
741 delay = 0.0001
742 times = 33
743 for _ in range(times): # retries for roughly 1 second
744 try:
745 return fun(self, *args, **kwargs)
746 except WindowsError as _:
747 err = _
748 if err.winerror == ERROR_PARTIAL_COPY:
749 time.sleep(delay)
750 delay = min(delay * 2, 0.04)
751 continue
752 raise
753 msg = (
754 "{} retried {} times, converted to AccessDenied as it's still"
755 "returning {}".format(fun, times, err)
756 )
757 raise AccessDenied(pid=self.pid, name=self._name, msg=msg)
758  
759 return wrapper

934 @wrap_exceptions
935 def wait(self, timeout=None):
936 if timeout is None:
937 cext_timeout = cext.INFINITE
938 else:
939 # WaitForSingleObject() expects time in milliseconds.
940 cext_timeout = int(timeout * 1000)
941  
942 timer = getattr(time, 'monotonic', time.time)
943 stop_at = timer() + timeout if timeout is not None else None
944  
945 try:
946 # Exit code is supposed to come from GetExitCodeProcess().
947 # May also be None if OpenProcess() failed with
948 # ERROR_INVALID_PARAMETER, meaning PID is already gone.
949 exit_code = cext.proc_wait(self.pid, cext_timeout)
950 except cext.TimeoutExpired:
951 # WaitForSingleObject() returned WAIT_TIMEOUT. Just raise.
952 raise TimeoutExpired(timeout, self.pid, self._name)
953 except cext.TimeoutAbandoned:
954 # WaitForSingleObject() returned WAIT_ABANDONED, see:
955 # https://github.com/giampaolo/psutil/issues/1224
956 # We'll just rely on the internal polling and return None
957 # when the PID disappears. Subprocess module does the same
958 # (return None):
959 # https://github.com/python/cpython/blob/
960 # be50a7b627d0aa37e08fa8e2d5568891f19903ce/
961 # Lib/subprocess.py#L1193-L1194
962 exit_code = None
963  
964 # At this point WaitForSingleObject() returned WAIT_OBJECT_0,
965 # meaning the process is gone. Stupidly there are cases where
966 # its PID may still stick around so we do a further internal
967 # polling.
968 delay = 0.0001
969 while True:
970 if not pid_exists(self.pid):
971 return exit_code
972 if stop_at and timer() >= stop_at:
973 raise TimeoutExpired(timeout, pid=self.pid, name=self._name)
974 time.sleep(delay)
975 delay = min(delay * 2, 0.04) # incremental delay
