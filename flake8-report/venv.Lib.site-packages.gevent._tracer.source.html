27 class GreenletTracer(object):
28 def __init__(self):
29 # A counter, incremented by the greenlet trace function
30 # we install on every greenlet switch. This is reset when the
31 # periodic monitoring thread runs.
32  
33 self.greenlet_switch_counter = 0
34  
35 # The greenlet last switched to.
36 self.active_greenlet = None
37  
38 # The trace function that was previously installed,
39 # if any.
40 # NOTE: Calling a class instance is cheaper than
41 # calling a bound method (at least when compiled with cython)
42 # even when it redirects to another function.
43 prev_trace = settrace(self)
44  
45 self.previous_trace_function = prev_trace
46  
47 self._killed = False
48  
49 def kill(self):
50 # Must be called in the monitored thread.
51 if not self._killed:
52 self._killed = True
53 settrace(self.previous_trace_function)
54 self.previous_trace_function = None
55  
56 def _trace(self, event, args):
57 # This function runs in the thread we are monitoring.
58 self.greenlet_switch_counter += 1
59 if event in ('switch', 'throw'):
60 # args is (origin, target). This is the only defined
61 # case
62 self.active_greenlet = args[1]
63 else:
64 self.active_greenlet = None
65 if self.previous_trace_function is not None:
66 self.previous_trace_function(event, args)
67  
68 def __call__(self, event, args):
69 return self._trace(event, args)

71 def did_block_hub(self, hub):
72 # Check to see if we have blocked since the last call to this
73 # method. Returns a true value if we blocked (not in the hub),
74 # a false value if everything is fine.
75  
76 # This may be called in the same thread being traced or a
77 # different thread; if a different thread, there is a race
78 # condition with this being incremented in the thread we're
79 # monitoring, but probably not often enough to lead to
80 # annoying false positives.
81  
82 active_greenlet = self.active_greenlet
83 did_switch = self.greenlet_switch_counter != 0
84 self.greenlet_switch_counter = 0
85  
86 if did_switch or active_greenlet is None or active_greenlet is hub:
87 # Either we switched, or nothing is running (we got a
88 # trace event we don't know about or were requested to
89 # ignore), or we spent the whole time in the hub, blocked
90 # for IO. Nothing to report.
91 return False
92 return True, active_greenlet

94 def ignore_current_greenlet_blocking(self):
95 # Don't pay attention to the current greenlet.
96 self.active_greenlet = None
97  
98 def monitor_current_greenlet_blocking(self):
99 self.active_greenlet = getcurrent()

101 def did_block_hub_report(self, hub, active_greenlet, format_kwargs):
102 # XXX: On Python 2 with greenlet 1.0a1, '%s' formatting a greenlet
103 # results in a unicode object. This is a bug in greenlet, I think.
104 # https://github.com/python-greenlet/greenlet/issues/218
105 report = ['=' * 80,
106 '\n%s : Greenlet %s appears to be blocked' %
107 (gmctime(), str(active_greenlet))]
108 report.append(" Reported by %s" % (self,))
109 try:
110 frame = sys._current_frames()[hub.thread_ident]
111 except KeyError:
112 # The thread holding the hub has died. Perhaps we shouldn't
113 # even report this?
114 stack = ["Unknown: No thread found for hub %r\n" % (hub,)]
115 else:
116 stack = traceback.format_stack(frame)
117 report.append('Blocked Stack (for thread id %s):' % (hex(hub.thread_ident),))
118 report.append(''.join(stack))
119 report.append("Info:")
120 report.extend(format_run_info(**format_kwargs))
121  
122 return report

136 class HubSwitchTracer(_HubTracer):
137 # A greenlet tracer that records the last time we switched *into* the hub.
138  
139 def __init__(self, hub, max_blocking_time):
140 _HubTracer.__init__(self, hub, max_blocking_time)
141 self.last_entered_hub = 0
142  
143 def _trace(self, event, args):
144 GreenletTracer._trace(self, event, args)
145 if self.active_greenlet is self.hub:
146 self.last_entered_hub = perf_counter()
147  
148 def did_block_hub(self, hub):
149 if perf_counter() - self.last_entered_hub > self.max_blocking_time:
150 return True, self.active_greenlet

153 class MaxSwitchTracer(_HubTracer):
154 # A greenlet tracer that records the maximum time between switches,
155 # not including time spent in the hub.
156  
157 def __init__(self, hub, max_blocking_time):
158 _HubTracer.__init__(self, hub, max_blocking_time)
159 self.last_switch = perf_counter()
160 self.max_blocking = 0
161  
162 def _trace(self, event, args):
163 old_active = self.active_greenlet
164 GreenletTracer._trace(self, event, args)
165 if old_active is not self.hub and old_active is not None:
166 # If we're switching out of the hub, the blocking
167 # time doesn't count.
168 switched_at = perf_counter()
169 self.max_blocking = max(self.max_blocking,
170 switched_at - self.last_switch)
171  
172 def did_block_hub(self, hub):
173 if self.max_blocking == 0:
174 # We never switched. Check the time now
175 self.max_blocking = perf_counter() - self.last_switch
176  
177 if self.max_blocking > self.max_blocking_time:
178 return True, self.active_greenlet
