235 def SetSite(self, unknown):
236 if unknown:
237 # retrieve the parent window interface for this site
238 olewindow = unknown.QueryInterface(pythoncom.IID_IOleWindow)
239 # ask the window for its handle
240 hwndparent = olewindow.GetWindow()
241  
242 # first get a command target
243 cmdtarget = unknown.QueryInterface(axcontrol.IID_IOleCommandTarget)
244 # then travel over to a service provider
245 serviceprovider = cmdtarget.QueryInterface(pythoncom.IID_IServiceProvider)
246 # finally ask for the internet explorer application, returned as a dispatch object
247 self.webbrowser = Dispatch(
248 serviceprovider.QueryService(
249 "{0002DF05-0000-0000-C000-000000000046}", pythoncom.IID_IDispatch
250 )
251 )
252  
253 # now create and set up the toolbar
254 self.toolbar = IEToolbarCtrl(hwndparent)

256 buttons = [
257 ("Visit PyWin32 Homepage", self.on_first_button),
258 ("Another Button", self.on_second_button),
259 ("Yet Another Button", self.on_third_button),
260 ]

263 # wrap our parent window so we can hook message handlers
264 window = win32ui.CreateWindowFromHandle(hwndparent)

273 button.fsState = commctrl.TBSTATE_ENABLED
274 button.fsStyle = commctrl.TBSTYLE_BUTTON
275 button.iString = name
