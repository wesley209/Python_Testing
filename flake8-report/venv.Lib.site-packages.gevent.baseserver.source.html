23 # We define a helper function to handle closing the socket in
24 # do_handle; We'd like to bind it to a kwarg to avoid *any* lookups at
25 # all, but that's incompatible with the calling convention of
26 # do_handle. On CPython, this is ~20% faster than creating and calling
27 # a closure and ~10% faster than using a @staticmethod. (In theory, we
28 # could create a closure only once in set_handle, to wrap self._handle,
29 # but this is safer from a backwards compat standpoint.)
30 # we also avoid unpacking the *args tuple when calling/spawning this object
31 # for a tiny improvement (benchmark shows a wash)
32 def _handle_and_close_when_done(handle, close, args_tuple):
33 try:
34 return handle(*args_tuple)
35 finally:
36 close(*args_tuple)

39 class BaseServer(object):
40 """
41 An abstract base class that implements some common functionality for the servers in gevent.
42  
43 :param listener: Either be an address that the server should bind
44 on or a :class:`gevent.socket.socket` instance that is already
45 bound (and put into listening mode in case of TCP socket).
46  
47 :keyword handle: If given, the request handler. The request
48 handler can be defined in a few ways. Most commonly,
49 subclasses will implement a ``handle`` method as an
50 instance method. Alternatively, a function can be passed
51 as the ``handle`` argument to the constructor. In either
52 case, the handler can later be changed by calling
53 :meth:`set_handle`.
54  
55 When the request handler returns, the socket used for the
56 request will be closed. Therefore, the handler must not return if
57 the socket is still in use (for example, by manually spawned greenlets).
58  
59 :keyword spawn: If provided, is called to create a new
60 greenlet to run the handler. By default,
61 :func:`gevent.spawn` is used (meaning there is no
62 artificial limit on the number of concurrent requests). Possible values for *spawn*:
63  
64 - a :class:`gevent.pool.Pool` instance -- ``handle`` will be executed
65 using :meth:`gevent.pool.Pool.spawn` only if the pool is not full.
66 While it is full, no new connections are accepted;
67 - :func:`gevent.spawn_raw` -- ``handle`` will be executed in a raw
68 greenlet which has a little less overhead then :class:`gevent.Greenlet` instances spawned by default;
69 - ``None`` -- ``handle`` will be executed right away, in the :class:`Hub` greenlet.
70 ``handle`` cannot use any blocking functions as it would mean switching to the :class:`Hub`.
71 - an integer -- a shortcut for ``gevent.pool.Pool(integer)``
72  
73 .. versionchanged:: 1.1a1
74 When the *handle* function returns from processing a connection,
75 the client socket will be closed. This resolves the non-deterministic
76 closing of the socket, fixing ResourceWarnings under Python 3 and PyPy.
77 .. versionchanged:: 1.5
78 Now a context manager that returns itself and calls :meth:`stop` on exit.
79

372 def stop(self, timeout=None):
373 """
374 Stop accepting the connections and close the listening socket.
375  
376 If the server uses a pool to spawn the requests, then
377 :meth:`stop` also waits for all the handlers to exit. If there
378 are still handlers executing after *timeout* has expired
379 (default 1 second, :attr:`stop_timeout`), then the currently
380 running handlers in the pool are killed.
381

383 any spawned greenlets that are handling requests continue running until
384 they naturally complete.
385 """

388 timeout = self.stop_timeout
