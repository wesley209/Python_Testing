237 def _wrap_sni_socket(sock: socket.socket, sslopt: dict, hostname, check_hostname):
238 context = sslopt.get("context", None)
239 if not context:
240 context = ssl.SSLContext(sslopt.get("ssl_version", ssl.PROTOCOL_TLS_CLIENT))
241 # Non default context need to manually enable SSLKEYLOGFILE support by setting the keylog_filename attribute.
242 # For more details see also:
243 # * https://docs.python.org/3.8/library/ssl.html?highlight=sslkeylogfile#context-creation
244 # * https://docs.python.org/3.8/library/ssl.html?highlight=sslkeylogfile#ssl.SSLContext.keylog_filename
245 context.keylog_filename = os.environ.get("SSLKEYLOGFILE", None)
246  
247 if sslopt.get("cert_reqs", ssl.CERT_NONE) != ssl.CERT_NONE:
248 cafile = sslopt.get("ca_certs", None)
249 capath = sslopt.get("ca_cert_path", None)
250 if cafile or capath:
251 context.load_verify_locations(cafile=cafile, capath=capath)
252 elif hasattr(context, "load_default_certs"):
253 context.load_default_certs(ssl.Purpose.SERVER_AUTH)
254 if sslopt.get("certfile", None):
255 context.load_cert_chain(
256 sslopt["certfile"],
257 sslopt.get("keyfile", None),
258 sslopt.get("password", None),
259 )
260  
261 # Python 3.10 switch to PROTOCOL_TLS_CLIENT defaults to "cert_reqs = ssl.CERT_REQUIRED" and "check_hostname = True"
262 # If both disabled, set check_hostname before verify_mode
263 # see https://github.com/liris/websocket-client/commit/b96a2e8fa765753e82eea531adb19716b52ca3ca#commitcomment-10803153
264 if sslopt.get("cert_reqs", ssl.CERT_NONE) == ssl.CERT_NONE and not sslopt.get(
265 "check_hostname", False
266 ):
267 context.check_hostname = False
268 context.verify_mode = ssl.CERT_NONE

270 context.check_hostname = sslopt.get("check_hostname", True)

316 def _tunnel(sock: socket.socket, host, port: int, auth) -> socket.socket:
317 debug("Connecting proxy...")
318 connect_header = f"CONNECT {host}:{port} HTTP/1.1\r\n"
319 connect_header += f"Host: {host}:{port}\r\n"

321 # TODO: support digest auth.
322 if auth and auth[0]:
323 auth_str = auth[0]
324 if auth[1]:
325 auth_str += f":{auth[1]}"
326 encoded_str = base64encode(auth_str.encode()).strip().decode().replace("\n", "")
327 connect_header += f"Proxy-Authorization: Basic {encoded_str}\r\n"
328 connect_header += "\r\n"
329 dump("request header", connect_header)

331 send(sock, connect_header)

335 except Exception as e:
336 raise WebSocketProxyException(str(e))
337

345 status = None
346 status_message = None
347 headers: dict = {}
348 trace("--- response header ---")
349  
350 while True:
351 line = recv_line(sock)
352 line = line.decode("utf-8").strip()
353 if not line:
354 break
355 trace(line)
356 if not status:
357 status_info = line.split(" ", 2)
358 status = int(status_info[1])
359 if len(status_info) > 2:
360 status_message = status_info[2]
361 else:
362 kv = line.split(":", 1)
363 if len(kv) != 2:
364 raise WebSocketException("Invalid header")
365 key, value = kv
366 if key.lower() == "set-cookie" and headers.get("set-cookie"):
367 headers["set-cookie"] = headers.get("set-cookie") + "; " + value.strip()
368 else:
369 headers[key.lower()] = value.strip()

371 trace("-----------------------")
372  
373 return status, headers, status_message
