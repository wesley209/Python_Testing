35 def find_pdh_counter_localized_name(english_name, machine_name=None):
36 if not counter_english_map:
37 import win32api
38 import win32con
39  
40 counter_reg_value = win32api.RegQueryValueEx(
41 win32con.HKEY_PERFORMANCE_DATA, "Counter 009"
42 )
43 counter_list = counter_reg_value[0]
44 for i in range(0, len(counter_list) - 1, 2):
45 try:
46 counter_id = int(counter_list[i])
47 except ValueError:
48 continue

55 def GetPerformanceAttributes(
56 object, counter, instance=None, inum=-1, format=win32pdh.PDH_FMT_LONG, machine=None
57 ):
58 # NOTE: Many counters require 2 samples to give accurate results,
59 # including "% Processor Time" (as by definition, at any instant, a
60 # thread's CPU usage is either 0 or 100). To read counters like this,
61 # you should copy this function, but keep the counter open, and call
62 # CollectQueryData() each time you need to know.
63 # See http://support.microsoft.com/default.aspx?scid=kb;EN-US;q262938
64 # and http://msdn.microsoft.com/library/en-us/dnperfmo/html/perfmonpt2.asp
65 # My older explanation for this was that the "AddCounter" process forced
66 # the CPU to 100%, but the above makes more sense :)
67 path = win32pdh.MakeCounterPath((machine, object, instance, None, inum, counter))
68 hq = win32pdh.OpenQuery()
69 try:
70 hc = win32pdh.AddCounter(hq, path)
71 try:
72 win32pdh.CollectQueryData(hq)
73 type, val = win32pdh.GetFormattedCounterValue(hc, format)
74 return val
75 finally:
76 win32pdh.RemoveCounter(hc)
77 finally:
78 win32pdh.CloseQuery(hq)

88 ):
89 """Find performance attributes by (case insensitive) instance name.
90  
91 Given a process name, return a list with the requested attributes.
92 Most useful for returning a tuple of PIDs given a process name.
93 """
94 if object is None:
95 object = find_pdh_counter_localized_name("Process", machine)
96 if counter is None:
97 counter = find_pdh_counter_localized_name("ID Process", machine)
98 if bRefresh: # PDH docs say this is how you do a refresh.
99 win32pdh.EnumObjects(None, machine, 0, 1)
100 instanceName = instanceName.lower()
101 items, instances = win32pdh.EnumObjectItems(None, None, object, -1)
102 # Track multiple instances.
103 instance_dict = {}
104 for instance in instances:
105 try:
106 instance_dict[instance] += 1
107 except KeyError:
108 instance_dict[instance] = 0
109  
110 ret = []
111 for instance, max_instances in instance_dict.items():
112 for inum in range(max_instances + 1):
113 if instance.lower() == instanceName:
114 ret.append(
115 GetPerformanceAttributes(
116 object, counter, instance, inum, format, machine
117 )
118 )
119 return ret
