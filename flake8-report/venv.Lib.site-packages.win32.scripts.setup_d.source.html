70 try:
71 key = winreg.OpenKey(
72 winreg.HKEY_LOCAL_MACHINE,
73 f"Software\\Python\\PythonCore\\{sys.winver}\\Modules\\{mod_name}",
74 )
75 except winreg.error:

79 f"Software\\Python\\PythonCore\\{sys.winver}\\Modules\\{mod_name}",
