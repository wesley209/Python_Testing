47 """Start searching for relative objects your search criteria with By.

51 - using: search term to find the element with.

61 class RelativeBy:
62 """Gives the opportunity to find elements based on their relative location
63 on the page from a root elelemt. It is recommended that you use the helper
64 function to create it.
65  
66 Example:
67 lowest = driver.find_element(By.ID, "below")
68  
69 elements = driver.find_elements(locate_with(By.CSS_SELECTOR, "p").above(lowest))
70  
71 ids = [el.get_attribute('id') for el in elements]
72 assert "above" in ids
73 assert "mid" in ids
74 """
75  
76 LocatorType = Dict[ByType, str]

78 def __init__(self, root: Optional[Dict[ByType, str]] = None, filters: Optional[List] = None):
79 """Creates a new RelativeBy object. It is preferred if you use the
80 `locate_with` method as this signature could change.
81  
82 :Args:
83 root - A dict with `By` enum as the key and the search query as the value
84 filters - A list of the filters that will be searched. If none are passed
85 in please use the fluent API on the object to create the filters
86 """
87 self.root = root
88 self.filters = filters or []

168 def near(self, element_or_locator: Union[WebElement, LocatorType, None] = None, distance: int = 50) -> "RelativeBy":
169 """Add a filter to look for elements near.
170  
171 :Args:
172 - element_or_locator: Element to look near by the element or within a distance
173 - distance: distance in pixel
174 """
175 if not element_or_locator:
176 raise WebDriverException("Element or locator must be given when calling near method")
177 if distance <= 0:
178 raise WebDriverException("Distance must be positive")
179  
180 self.filters.append({"kind": "near", "args": [element_or_locator, distance]})
181 return self
