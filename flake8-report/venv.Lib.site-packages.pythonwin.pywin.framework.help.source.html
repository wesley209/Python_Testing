79 def _ListAllHelpFilesInRoot(root):
80 """Returns a list of (helpDesc, helpFname) for all registered help files"""
81  
82 retList = []
83 try:
84 key = win32api.RegOpenKey(
85 root, regutil.BuildDefaultPythonKey() + "\\Help", 0, win32con.KEY_READ
86 )
87 except win32api.error as exc:
88 import winerror
89  
90 if exc.winerror != winerror.ERROR_FILE_NOT_FOUND:
91 raise
92 return retList
93 try:
94 keyNo = 0
95 while 1:
96 try:
97 helpDesc = win32api.RegEnumKey(key, keyNo)
98 helpFile = win32api.RegQueryValue(key, helpDesc)
99 retList.append((helpDesc, helpFile))
100 keyNo += 1
101 except win32api.error as exc:
102 import winerror
103  
104 if exc.winerror != winerror.ERROR_NO_MORE_ITEMS:
105 raise
106 break
107 finally:
108 win32api.RegCloseKey(key)
