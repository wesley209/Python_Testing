59 def TestObjectFromWindow():
60 # Check we can use ObjectFromLresult to get the COM object from the
61 hwnd = win32gui.FindWindow("IEFrame", None)
62 # Thanks https://stackoverflow.com/a/10154498/18450412 for the child stack on IE8+
63 for child_class in (
64 "Frame Tab",
65 "TabWindowClass",
66 "Shell DocObject View",
67 "Internet Explorer_Server",
68 ):
69 hwnd = win32gui.FindWindowEx(hwnd, 0, child_class, None)
70 # Once you have an 'Internet Explorer_Server',
71 # you can send a message and use ObjectFromLresult to get it back.
72 msg = win32gui.RegisterWindowMessage("WM_HTML_GETOBJECT")
73 rc, result = win32gui.SendMessageTimeout(
74 hwnd, msg, 0, 0, win32con.SMTO_ABORTIFHUNG, 1000
75 )
76 ob = pythoncom.ObjectFromLresult(result, pythoncom.IID_IDispatch, 0)
77 doc = Dispatch(ob)
