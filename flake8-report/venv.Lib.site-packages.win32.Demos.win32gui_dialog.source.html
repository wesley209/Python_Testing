50 def toparam(self):
51 self._buffs = []
52 full_fmt = ""
53 vals = []
54 mask = 0
55 # calc the mask
56 for name, fmt, default, this_mask in self._struct_items_:
57 if this_mask is not None and self.__dict__.get(name) is not None:
58 mask |= this_mask
59 self.mask = mask
60 for name, fmt, default, this_mask in self._struct_items_:
61 val = self.__dict__[name]
62 if fmt == "z":
63 fmt = "Pi"
64 if val is None:
65 vals.append(0)
66 vals.append(0)
67 else:
68 # Note this demo still works with byte strings. An
69 # alternate strategy would be to use unicode natively
70 # and use the 'W' version of the messages - eg,
71 # LVM_SETITEMW etc.
72 val += "\x00"
73 if isinstance(val, str):
74 val = val.encode("mbcs")
75 str_buf = array.array("b", val)
76 vals.append(str_buf.buffer_info()[0])
77 vals.append(len(val))
78 self._buffs.append(str_buf) # keep alive during the call.
79 else:
80 if val is None:
81 val = default
82 vals.append(val)
83 full_fmt += fmt
84 return struct.pack(*(full_fmt,) + tuple(vals))
