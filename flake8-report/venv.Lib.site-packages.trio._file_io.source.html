85 # This is a little complicated. IO objects have a lot of methods, and which are available on
86 # different types varies wildly. We want to match the interface of whatever file we're wrapping.
87 # This pile of protocols each has one sync method/property, meaning they're going to be compatible
88 # with a file class that supports that method/property. The ones parameterized with AnyStr take
89 # either str or bytes depending.
90  
91 # The wrapper is then a generic class, where the typevar is set to the type of the sync file we're
92 # wrapping. For generics, adding a type to self has a special meaning - properties/methods can be
93 # conditional - it's only valid to call them if the object you're accessing them on is compatible
94 # with that type hint. By using the protocols, the type checker will be checking to see if the
95 # wrapped type has that method, and only allow the methods that do to be called. We can then alter
96 # the signature however it needs to match runtime behaviour.
97 # More info: https://mypy.readthedocs.io/en/stable/more_types.html#advanced-uses-of-self-types

218 class AsyncIOWrapper(AsyncResource, Generic[FileT_co]):
219 """A generic :class:`~io.IOBase` wrapper that implements the :term:`asynchronous
220 file object` interface. Wrapped methods that could block are executed in
221 :meth:`trio.to_thread.run_sync`.
222  
223 All properties and methods defined in :mod:`~io` are exposed by this
224 wrapper, if they exist in the wrapped file object.
225 """
226  
227 def __init__(self, file: FileT_co) -> None:
228 self._wrapped = file
229  
230 @property
231 def wrapped(self) -> FileT_co:
232 """object: A reference to the wrapped file object"""
233  
234 return self._wrapped

238 def __getattr__(self, name: str) -> object:
239 if name in _FILE_SYNC_ATTRS:
240 return getattr(self._wrapped, name)
241 if name in _FILE_ASYNC_METHODS:
242 meth = getattr(self._wrapped, name)
243  
244 @async_wraps(self.__class__, self._wrapped.__class__, name)
245 async def wrapper(*args, **kwargs):
246 func = partial(meth, *args, **kwargs)
247 return await trio.to_thread.run_sync(func)
248  
249 # cache the generated method
250 setattr(self, name, wrapper)
251 return wrapper
252  
253 raise AttributeError(name)

449 """Asynchronous version of :func:`open`.
450  
451 Returns:
452 An :term:`asynchronous file object`
453  
454 Example::
455  
456 async with await trio.open_file(filename) as f:
457 async for line in f:
458 pass
459  
460 assert f.closed
461  
462 See also:
463 :func:`trio.Path.open`
464  
465 """

467 await trio.to_thread.run_sync(
468 io.open,
469 file,
470 mode,
471 buffering,
472 encoding,
473 errors,
474 newline,
475 closefd,
476 opener,
477 ),
478 )
479 return _file

482 def wrap_file(file: FileT) -> AsyncIOWrapper[FileT]:
483 """This wraps any file object in a wrapper that provides an asynchronous
484 file object interface.
485  
486 Args:
487 file: a :term:`file object`
488  
489 Returns:
490 An :term:`asynchronous file object` that wraps ``file``
491  
492 Example::
493  
494 async_file = trio.wrap_file(StringIO('asdf'))
495  
496 assert await async_file.read() == 'asdf'
497  
498 """
499  
500 def has(attr: str) -> bool:
501 return hasattr(file, attr) and callable(getattr(file, attr))
502  
503 if not (has("close") and (has("read") or has("write"))):
504 raise TypeError(
505 f"{file} does not implement required duck-file methods: "
506 "close and (read or write)",
507 )
508  
509 return AsyncIOWrapper(file)
