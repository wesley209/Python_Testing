47 # Send a simple "message" over a socket - send the number of bytes first,
48 # then the string. Ditto for receive.
49 def _send_msg(s, m):
50 s.send(struct.pack("i", len(m)))
51 s.send(m)

54 def _get_msg(s):
55 size_data = s.recv(struct.calcsize("i"))
56 if not size_data:
57 return None
58 cb = struct.unpack("i", size_data)[0]
59 return s.recv(cb)
