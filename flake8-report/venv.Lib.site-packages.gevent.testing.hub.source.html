35 def ignoring_expected_test_error(self):
36 """
37 Code in the body of this context manager will ignore
38 ``EXPECTED_TEST_ERROR`` objects reported to ``handle_error``;
39 they will not get a chance to go to the hub's parent.
40  
41 This completely changes the semantics of normal error handling
42 by avoiding some switches (to the main greenlet, and eventually
43 once a callback is processed, back to the hub). This should be used
44 in narrow ways for test compatibility for tests that assume
45 ``ExpectedException`` objects behave this way.
46 """
47 old = self.IGNORE_EXPECTED_TEST_ERROR
48 self.IGNORE_EXPECTED_TEST_ERROR = True
49 try:
50 yield
51 finally:
52 self.IGNORE_EXPECTED_TEST_ERROR = old
