34 def get_vk(chardesc):
35 if len(chardesc) == 1:
36 # it is a character.
37 info = win32api.VkKeyScan(chardesc)
38 if info == -1:
39 # Note: returning None, None causes an error when keyboard layout is non-English, see the report below
40 # https://stackoverflow.com/questions/45138084/pythonwin-occasionally-gives-an-error-on-opening
41 return 0, 0
42 vk = win32api.LOBYTE(info)
43 state = win32api.HIBYTE(info)
44 modifiers = 0
45 if state & 0x1:
46 modifiers |= win32con.SHIFT_PRESSED
47 if state & 0x2:
48 modifiers |= win32con.LEFT_CTRL_PRESSED | win32con.RIGHT_CTRL_PRESSED
49 if state & 0x4:
50 modifiers |= win32con.LEFT_ALT_PRESSED | win32con.RIGHT_ALT_PRESSED
51 return vk, modifiers
52 # must be a 'key name'
53 return key_name_to_vk.get(chardesc.lower()), 0

72 def parse_key_name(name):
73 name += "-" # Add a sentinal
74 start = pos = 0
75 max = len(name)
76 toks = []
77 while pos < max:
78 if name[pos] in "+-":
79 tok = name[start:pos]
80 # use the ascii lower() version of tok, so ascii chars require
81 # an explicit shift modifier - ie 'Ctrl+G' should be treated as
82 # 'ctrl+g' - 'ctrl+shift+g' would be needed if desired.
83 # This is mainly to avoid changing all the old keystroke defs
84 toks.append(tok.lower())
85 pos += 1 # skip the sep
86 start = pos
87 pos += 1
88 flags = 0
89 # do the modifiers
90 for tok in toks[:-1]:
91 mod = modifiers.get(tok.lower())
92 if mod is not None:
93 flags |= mod
94 # the key name
95 vk, this_flags = get_vk(toks[-1])
96 return vk, flags | this_flags

116 def make_key_name(vk, flags):
117 # Check alt keys.
118 flags_done = 0
119 parts = []
120 for moddata in _checks:
121 for name, checkflag in moddata:
122 if flags & checkflag:
123 parts.append(name)
124 flags_done &= checkflag
125 break
126 if flags_done & flags:
127 parts.append(hex(flags & ~flags_done))
128 # Now the key name.
129 if vk is None:
130 parts.append("<Unknown scan code>")
131 else:
132 try:
133 parts.append(key_code_to_name[vk])
134 except KeyError:
135 # Not in our virtual key map - ask Windows what character this
136 # key corresponds to.
137 scancode = win32api.MapVirtualKey(vk, MAPVK_VK_TO_CHAR)
138 parts.append(chr(scancode))
139 sep = "+"
140 if sep in parts:
141 sep = "-"
142 return sep.join([p.capitalize() for p in parts])
