53 # Untraceable locks were part of the solution to https://bugs.python.org/issue1733757
54 # which details a deadlock that could happen if a trace function invoked
55 # threading.currentThread at shutdown time---the cleanup lock would be held
56 # by the VM, and calling currentThread would try to acquire it again. The interpreter
57 # changed in 2.6 to use the `with` statement (https://hg.python.org/cpython/rev/76f577a9ec03/),
58 # which apparently doesn't trace in quite the same way.
59 if hasattr(sys, 'gettrace'):
60 old = sys.gettrace()
61 else:
62 old = None
63  
64 lst = []
65 try:
66 def trace(frame, ev, _arg):
67 lst.append((frame.f_code.co_filename, frame.f_lineno, ev))
68 print("TRACE: %s:%s %s" % lst[-1])
69 return trace
70  
71 with allocate_lock():
72 sys.settrace(trace)
73 finally:
74 sys.settrace(old)
75  
76 self.assertEqual(lst, [], "trace not empty")
