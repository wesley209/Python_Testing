96 def QueryStatus(self, pguidCmdGroup, prgCmds, cmdtextf):
97 # 'cmdtextf' is the 'cmdtextf' element from the OLECMDTEXT structure,
98 # or None if a NULL pointer was passed.
99 result = []
100 for id, flags in prgCmds:
101 flags |= axcontrol.OLECMDF_SUPPORTED | axcontrol.OLECMDF_ENABLED
102 result.append((id, flags))
103 if cmdtextf is None:
104 cmdtext = None # must return None if nothing requested.
105 # IE never seems to want any text - this code is here for
106 # demo purposes only
107 elif cmdtextf == axcontrol.OLECMDTEXTF_NAME:
108 cmdtext = "IEButton Name"
109 else:
110 cmdtext = "IEButton State"
111 return result, cmdtext

118 def SetSite(self, unknown):
119 if unknown:
120 # first get a command target
121 cmdtarget = unknown.QueryInterface(axcontrol.IID_IOleCommandTarget)
122 # then travel over to a service provider
123 serviceprovider = cmdtarget.QueryInterface(pythoncom.IID_IServiceProvider)
124 # finally ask for the internet explorer application, returned as a dispatch object
125 self.webbrowser = win32com.client.Dispatch(
126 serviceprovider.QueryService(
127 "{0002DF05-0000-0000-C000-000000000046}", pythoncom.IID_IDispatch
128 )
129 )
130 else:
131 # lose all references
132 self.webbrowser = None
