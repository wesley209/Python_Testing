48 def GetColumnInfo(self, index):
49 # We support exactly 2 columns - 'pyc size' and 'pyo size'
50 if index in [0, 1]:
51 # As per the MSDN sample, use our CLSID as the fmtid
52 if index == 0:
53 ext = ".pyc"
54 else:
55 ext = ".pyo"
56 title = ext + " size"
57 description = "Size of compiled %s file" % ext
58 col_id = (self._reg_clsid_, index) # fmtid # pid
59 col_info = (
60 col_id, # scid
61 pythoncom.VT_I4, # vt
62 commctrl.LVCFMT_RIGHT, # fmt
63 20, # cChars
64 shellcon.SHCOLSTATE_TYPE_INT
65 | shellcon.SHCOLSTATE_SECONDARYUI, # csFlags
66 title,
67 description,
68 )
69 return col_info
70 return None # Indicate no more columns.
