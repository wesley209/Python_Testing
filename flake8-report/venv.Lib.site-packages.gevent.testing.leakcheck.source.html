37 def ignores_leakcheck(func):
38 """
39 Ignore the given object during leakchecks.
40  
41 Can be applied to a method, in which case the method will run, but
42 will not be subject to leak checks.
43  
44 If applied to a class, the entire class will be skipped during leakchecks. This
45 is intended to be used for classes that are very slow and cause problems such as
46 test timeouts; typically it will be used for classes that are subclasses of a base
47 class and specify variants of behaviour (such as pool sizes).
48 """
49 func.ignore_leakcheck = True
50 return func
