142 def get_pin_and_cookie_name(
143 app: WSGIApplication,
144 ) -> tuple[str, str] | tuple[None, None]:
145 """Given an application object this returns a semi-stable 9 digit pin
146 code and a random key. The hope is that this is stable between
147 restarts to not make debugging particularly frustrating. If the pin
148 was forcefully disabled this returns `None`.
149  
150 Second item in the resulting tuple is the cookie name for remembering.
151 """
152 pin = os.environ.get("WERKZEUG_DEBUG_PIN")
153 rv = None
154 num = None
155  
156 # Pin was explicitly disabled
157 if pin == "off":
158 return None, None
159  
160 # Pin was provided explicitly
161 if pin is not None and pin.replace("-", "").isdecimal():
162 # If there are separators in the pin, return it directly
163 if "-" in pin:
164 rv = pin
165 else:
166 num = pin
167  
168 modname = getattr(app, "__module__", t.cast(object, app).__class__.__module__)
169 username: str | None
170  
171 try:
172 # getuser imports the pwd module, which does not exist in Google
173 # App Engine. It may also raise a KeyError if the UID does not
174 # have a username, such as in Docker.
175 username = getpass.getuser()
176 # Python >= 3.13 only raises OSError
177 except (ImportError, KeyError, OSError):
178 username = None
179  
180 mod = sys.modules.get(modname)
181  
182 # This information only exists to make the cookie unique on the
183 # computer, not as a security feature.
184 probably_public_bits = [
185 username,
186 modname,
187 getattr(app, "__name__", type(app).__name__),
188 getattr(mod, "__file__", None),
189 ]
190  
191 # This information is here to make it harder for an attacker to
192 # guess the cookie name. They are unlikely to be contained anywhere
193 # within the unauthenticated debug page.
194 private_bits = [str(uuid.getnode()), get_machine_id()]
195  
196 h = hashlib.sha1()
197 for bit in chain(probably_public_bits, private_bits):
198 if not bit:
199 continue
200 if isinstance(bit, str):
201 bit = bit.encode()
202 h.update(bit)
203 h.update(b"cookiesalt")
204  
205 cookie_name = f"__wzd{h.hexdigest()[:20]}"
206  
207 # If we need to generate a pin we salt it a bit more so that we don't
208 # end up with the same value and generate out 9 digits
209 if num is None:
210 h.update(b"pinsalt")
211 num = f"{int(h.hexdigest(), 16):09d}"[:9]
212  
213 # Format the pincode in groups of digits for easier remembering if
214 # we don't have a result yet.
215 if rv is None:
216 for group_size in 5, 4, 3:
217 if len(num) % group_size == 0:
218 rv = "-".join(
  • E203 Whitespace before ':'
219 num[x : x + group_size].rjust(group_size, "0")
220 for x in range(0, len(num), group_size)
221 )
222 break
223 else:
224 rv = num
225  
226 return rv, cookie_name

229 class DebuggedApplication:
230 """Enables debugging support for a given application::
231  
232 from werkzeug.debug import DebuggedApplication
233 from myapp import app
234 app = DebuggedApplication(app, evalex=True)
235  
236 The ``evalex`` argument allows evaluating expressions in any frame
237 of a traceback. This works by preserving each frame with its local
238 state. Some state, such as context globals, cannot be restored with
239 the frame by default. When ``evalex`` is enabled,
240 ``environ["werkzeug.debug.preserve_context"]`` will be a callable
241 that takes a context manager, and can be called multiple times.
242 Each context manager will be entered before evaluating code in the
243 frame, then exited again, so they can perform setup and cleanup for
244 each call.
245  
246 :param app: the WSGI application to run debugged.
247 :param evalex: enable exception evaluation feature (interactive
248 debugging). This requires a non-forking server.
249 :param request_key: The key that points to the request object in this
250 environment. This parameter is ignored in current
251 versions.
252 :param console_path: the URL for a general purpose console.
253 :param console_init_func: the function that is executed before starting
254 the general purpose console. The return value
255 is used as initial namespace.
256 :param show_hidden_frames: by default hidden traceback frames are skipped.
257 You can show them by setting this parameter
258 to `True`.
259 :param pin_security: can be used to disable the pin based security system.
260 :param pin_logging: enables the logging of the pin system.
261  
262 .. versionchanged:: 2.2
263 Added the ``werkzeug.debug.preserve_context`` environ key.
264 """

433 def check_pin_trust(self, environ: WSGIEnvironment) -> bool | None:
434 """Checks if the request passed the pin test. This returns `True` if the
435 request is trusted on a pin/cookie basis and returns `False` if not.
436 Additionally if the cookie's stored pin hash is wrong it will return
437 `None` so that appropriate action can be taken.
438 """
439 if self.pin is None:
440 return True
441 val = parse_cookie(environ).get(self.pin_cookie_name)
442 if not val or "|" not in val:
443 return False
444 ts_str, pin_hash = val.split("|", 1)
445  
446 try:
447 ts = int(ts_str)
448 except ValueError:
449 return False
450  
451 if pin_hash != hash_pin(self.pin):
452 return None
453 return (time.time() - PIN_TIME) < ts

465 def pin_auth(self, request: Request) -> Response:
466 """Authenticates with the pin."""
467 if not self.check_host_trust(request.environ):
468 return SecurityError() # type: ignore[return-value]
469  
470 exhausted = False
471 auth = False
472 trust = self.check_pin_trust(request.environ)
473 pin = t.cast(str, self.pin)
474  
475 # If the trust return value is `None` it means that the cookie is
476 # set but the stored pin hash value is bad. This means that the
477 # pin was changed. In this case we count a bad auth and unset the
478 # cookie. This way it becomes harder to guess the cookie name
479 # instead of the pin as we still count up failures.
480 bad_cookie = False
481 if trust is None:
482 self._fail_pin_auth()
483 bad_cookie = True
484  
485 # If we're trusted, we're authenticated.
486 elif trust:
487 auth = True
488  
489 # If we failed too many times, then we're locked out.
490 elif self._failed_pin_auth.value > 10:
491 exhausted = True
492  
493 # Otherwise go through pin based authentication
494 else:
495 entered_pin = request.args["pin"]
496  
497 if entered_pin.strip().replace("-", "") == pin.replace("-", ""):
498 self._failed_pin_auth.value = 0
499 auth = True
500 else:
501 self._fail_pin_auth()
502  
503 rv = Response(
504 json.dumps({"auth": auth, "exhausted": exhausted}),
505 mimetype="application/json",
506 )
507 if auth:
508 rv.set_cookie(
509 self.pin_cookie_name,
510 f"{int(time.time())}|{hash_pin(pin)}",
511 httponly=True,
512 samesite="Strict",
513 secure=request.is_secure,
514 )
515 elif bad_cookie:
516 rv.delete_cookie(self.pin_cookie_name)
517 return rv

531 def __call__(
532 self, environ: WSGIEnvironment, start_response: StartResponse
533 ) -> t.Iterable[bytes]:
534 """Dispatch the requests."""
535 # important: don't ever access a function here that reads the incoming
536 # form data! Otherwise the application won't have access to that data
537 # any more!
538 request = Request(environ)
539 response = self.debug_application
540 if request.args.get("__debugger__") == "yes":
541 cmd = request.args.get("cmd")
542 arg = request.args.get("f")
543 secret = request.args.get("s")
544 frame = self.frames.get(request.args.get("frm", type=int)) # type: ignore
545 if cmd == "resource" and arg:
546 response = self.get_resource(request, arg) # type: ignore
547 elif cmd == "pinauth" and secret == self.secret:
548 response = self.pin_auth(request) # type: ignore
549 elif cmd == "printpin" and secret == self.secret:
550 response = self.log_pin_request(request) # type: ignore
551 elif (
552 self.evalex
553 and cmd is not None
554 and frame is not None
555 and self.secret == secret
556 and self.check_pin_trust(environ)
557 ):
558 response = self.execute_command(request, cmd, frame) # type: ignore
559 elif (
560 self.evalex
561 and self.console_path is not None
562 and request.path == self.console_path
563 ):
564 response = self.display_console(request) # type: ignore
565 return response(environ, start_response)
