58 class AbstractLinkable(object):
59 # Encapsulates the standard parts of the linking and notifying
60 # protocol common to both repeatable events (Event, Semaphore) and
61 # one-time events (AsyncResult).
62 #
63 # With a few careful exceptions, instances of this object can only
64 # be used from a single thread. The exception is that certain methods
65 # may be used from multiple threads IFF:
66 #
67 # 1. They are documented as safe for that purpose; AND
68 # 2a. This object is compiled with Cython and thus is holding the GIL
69 # for the entire duration of the method; OR
70 # 2b. A subclass ensures that a Python-level native thread lock is held
71 # for the duration of the method; this is necessary in pure-Python mode.
72 # The only known implementation of such
73 # a subclass is for Semaphore. AND
74 # 3. The subclass that calls ``capture_hub`` catches
75 # and handles ``InvalidThreadUseError``
76 #
77 # TODO: As of gevent 1.5, we use the same datastructures and almost
78 # the same algorithm as Greenlet. See about unifying them more.

88 def __init__(self, hub=None):
89 # Before this implementation, AsyncResult and Semaphore
90 # maintained the order of notifications, but Event did not.
91  
92 # In gevent 1.3, before Semaphore extended this class, that
93 # was changed to not maintain the order. It was done because
94 # Event guaranteed to only call callbacks once (a set) but
95 # AsyncResult had no such guarantees. When Semaphore was
96 # changed to extend this class, it lost its ordering
97 # guarantees. Unfortunately, that made it unfair. There are
98 # rare cases that this can starve a greenlet
99 # (https://github.com/gevent/gevent/issues/1487) and maybe
100 # even lead to deadlock (not tested).
101  
102 # So in gevent 1.5 we go back to maintaining order. But it's
103 # still important not to make duplicate calls, and it's also
104 # important to avoid O(n^2) behaviour that can result from
105 # naive use of a simple list due to the need to handle removed
106 # links in the _notify_links loop. Cython has special support for
107 # built-in sets, lists, and dicts, but not ordereddict. Rather than
108 # use two data structures, or a dict({link: order}), we simply use a
109 # list and remove objects as we go, keeping track of them so as not to
110 # have duplicates called. This makes `unlink` O(n), but we can avoid
111 # calling it in the common case in _wait_core (even so, the number of
112 # waiters should usually be pretty small)
113 self._links = []
114 self._notifier = None
115 # This is conceptually a class attribute, defined here for ease of access in
116 # cython. If it's true, when notifiers fire, all existing callbacks are called.
117 # If its false, we only call callbacks as long as ready() returns true.
118 self._notify_all = True
119 # we don't want to do get_hub() here to allow defining module-level objects
120 # without initializing the hub. However, for multiple-thread safety, as soon
121 # as a waiting method is entered, even if it won't have to wait, we
122 # need to grab the hub and assign ownership. But we don't want to grab one prematurely.
123 # The example is three threads, the main thread and two worker threads; if we create
124 # a Semaphore in the main thread but only use it in the two threads, if we had grabbed
125 # the main thread's hub, the two worker threads would have a dependency on it, meaning that
126 # if the main event loop is blocked, the worker threads might get blocked too.
127 self.hub = hub
