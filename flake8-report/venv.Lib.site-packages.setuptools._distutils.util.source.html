87 def get_macosx_target_ver():
88 """Return the version of macOS for which we are building.
89  
90 The target version defaults to the version in sysconfig latched at time
91 the Python interpreter was built, unless overridden by an environment
92 variable. If neither source has a value, then None is returned"""
93  
94 syscfg_ver = get_macosx_target_ver_from_syscfg()
95 env_ver = os.environ.get(MACOSX_VERSION_VAR)
96  
97 if env_ver:
98 # Validate overridden version against sysconfig version, if have both.
99 # Ensure that the deployment target of the build process is not less
100 # than 10.3 if the interpreter was built for 10.3 or later. This
101 # ensures extension modules are built with correct compatibility
102 # values, specifically LDSHARED which can use
103 # '-undefined dynamic_lookup' which only works on >= 10.3.

143 def change_root(new_root, pathname):
144 """Return 'pathname' with 'new_root' prepended. If 'pathname' is
145 relative, this is equivalent to "os.path.join(new_root,pathname)".
146 Otherwise, it requires making 'pathname' relative and then joining the
147 two, which is tricky on DOS/Windows and Mac OS.
148 """
149 if os.name == 'posix':
150 if not os.path.isabs(pathname):
151 return os.path.join(new_root, pathname)
152 else:
153 return os.path.join(new_root, pathname[1:])
154  
155 elif os.name == 'nt':
156 (drive, path) = os.path.splitdrive(pathname)
157 if path[0] == os.sep:
158 path = path[1:]
159 return os.path.join(new_root, path)

161 raise DistutilsPlatformError(f"nothing known about platform '{os.name}'")

244 def split_quoted(s):
245 """Split a string up according to Unix shell-like rules for quotes and
246 backslashes. In short: words are delimited by spaces, as long as those
247 spaces are not escaped by a backslash, or inside a quoted string.
248 Single and double quotes are equivalent, and the quote characters can
249 be backslash-escaped. The backslash is stripped from any two-character
250 escape sequence, leaving only the escaped character. The quote
251 characters are stripped from any quoted string. Returns a list of
252 words.
253 """
254  
255 # This is a nice algorithm for splitting up a single string, since it
256 # doesn't require character-by-character examination. It was a little
257 # bit of a brain-bender to get it working right, though...
258 if _wordchars_re is None:
259 _init_regex()
260  
261 s = s.strip()
262 words = []
263 pos = 0
264  
265 while s:
266 m = _wordchars_re.match(s, pos)
267 end = m.end()
268 if end == len(s):
269 words.append(s[:end])
270 break
271  
272 if s[end] in string.whitespace:
273 # unescaped, unquoted whitespace: now
274 # we definitely have a word delimiter
275 words.append(s[:end])
276 s = s[end:].lstrip()
277 pos = 0
278  
279 elif s[end] == '\\':
280 # preserve whatever is being escaped;
281 # will become part of the current word
  • E203 Whitespace before ':'
282 s = s[:end] + s[end + 1 :]
283 pos = end + 1
284  
285 else:
286 if s[end] == "'": # slurp singly-quoted string
287 m = _squote_re.match(s, end)
288 elif s[end] == '"': # slurp doubly-quoted string
289 m = _dquote_re.match(s, end)
290 else:
291 raise RuntimeError("this can't happen (bad char '%c')" % s[end])
292  
293 if m is None:
294 raise ValueError(f"bad string (mismatched {s[end]} quotes?)")
295  
296 (beg, end) = m.span()
  • E203 Whitespace before ':'
297 s = s[:beg] + s[beg + 1 : end - 1] + s[end:]
298 pos = m.end() - 2
299  
300 if pos >= len(s):
301 words.append(s)
302 break
303  
304 return words

329 def strtobool(val):
330 """Convert a string representation of truth to true (1) or false (0).
331  
332 True values are 'y', 'yes', 't', 'true', 'on', and '1'; false values
333 are 'n', 'no', 'f', 'false', 'off', and '0'. Raises ValueError if
334 'val' is anything else.
335 """
336 val = val.lower()
337 if val in ('y', 'yes', 't', 'true', 'on', '1'):
338 return 1
339 elif val in ('n', 'no', 'f', 'false', 'off', '0'):
340 return 0
341 else:
342 raise ValueError(f"invalid truth value {val!r}")

345 def byte_compile( # noqa: C901
346 py_files,
347 optimize=0,
348 force=False,
349 prefix=None,
350 base_dir=None,
351 verbose=True,
352 dry_run=False,
353 direct=None,
354 ):
355 """Byte-compile a collection of Python source files to .pyc
356 files in a __pycache__ subdirectory. 'py_files' is a list
357 of files to compile; any files that don't end in ".py" are silently
358 skipped. 'optimize' must be one of the following:
359 0 - don't optimize
360 1 - normal optimization (like "python -O")
361 2 - extra optimization (like "python -OO")
362 If 'force' is true, all files are recompiled regardless of
363 timestamps.
364  
365 The source filename encoded in each bytecode file defaults to the
366 filenames listed in 'py_files'; you can modify these with 'prefix' and
367 'basedir'. 'prefix' is a string that will be stripped off of each
368 source filename, and 'base_dir' is a directory name that will be
369 prepended (after 'prefix' is stripped). You can supply either or both
370 (or neither) of 'prefix' and 'base_dir', as you wish.
371  
372 If 'dry_run' is true, doesn't actually do anything that would
373 affect the filesystem.
374  
375 Byte-compilation is either done directly in this interpreter process
376 with the standard py_compile module, or indirectly by writing a
377 temporary script and executing it. Normally, you should let
378 'byte_compile()' figure out to use direct compilation or not (see
379 the source for details). The 'direct' flag is used by the script
380 generated in indirect mode; unless you know what you're doing, leave
381 it set to None.
382 """
383  
384 # nothing is done if sys.dont_write_bytecode is True
385 if sys.dont_write_bytecode:
386 raise DistutilsByteCompileError('byte-compiling is disabled.')
387  
388 # First, if the caller didn't force us into direct or indirect mode,
389 # figure out which mode we should be in. We take a conservative
390 # approach: choose direct mode *only* if the current interpreter is
391 # in debug mode and optimize is 0. If we're not in debug mode (-O
392 # or -OO), we don't know which level of optimization this
393 # interpreter is running with, so we can't do direct
394 # byte-compilation and be certain that it's the right thing. Thus,
395 # always compile indirectly if the current interpreter is in either
396 # optimize mode, or if either optimization level was requested by
397 # the caller.
398 if direct is None:
399 direct = __debug__ and optimize == 0
400  
401 # "Indirect" byte-compilation: write a temporary script and then
402 # run it with the appropriate flags.
403 if not direct:
404 (script_fd, script_name) = tempfile.mkstemp(".py")
405 log.info("writing byte-compilation script '%s'", script_name)
406 if not dry_run:
407 script = os.fdopen(script_fd, "w", encoding='utf-8')
408  
409 with script:
410 script.write(
411 """\
412 from distutils.util import byte_compile
413 files = [
414 """
415 )
416  
417 # XXX would be nice to write absolute filenames, just for
418 # safety's sake (script should be more robust in the face of
419 # chdir'ing before running it). But this requires abspath'ing
420 # 'prefix' as well, and that breaks the hack in build_lib's
421 # 'byte_compile()' method that carefully tacks on a trailing
422 # slash (os.sep really) to make sure the prefix here is "just
423 # right". This whole prefix business is rather delicate -- the
424 # problem is that it's really a directory, but I'm treating it
425 # as a dumb string, so trailing slashes and so forth matter.
426  
427 script.write(",\n".join(map(repr, py_files)) + "]\n")
428 script.write(
429 f"""
430 byte_compile(files, optimize={optimize!r}, force={force!r},
431 prefix={prefix!r}, base_dir={base_dir!r},
432 verbose={verbose!r}, dry_run=False,
433 direct=True)
434 """
435 )
436  
437 cmd = [sys.executable]
438 cmd.extend(subprocess._optim_args_from_interpreter_flags())
439 cmd.append(script_name)
440 spawn(cmd, dry_run=dry_run)
441 execute(os.remove, (script_name,), f"removing {script_name}", dry_run=dry_run)
442  
443 # "Direct" byte-compilation: use the py_compile module to compile
444 # right here, right now. Note that the script generated in indirect
445 # mode simply calls 'byte_compile()' in direct mode, a weird sort of
446 # cross-process recursion. Hey, it works!
447 else:
448 from py_compile import compile
449  
450 for file in py_files:
451 if file[-3:] != ".py":
452 # This lets us be lazy and not filter filenames in
453 # the "install_lib" command.
454 continue
455  
456 # Terminology from the py_compile module:
457 # cfile - byte-compiled file
458 # dfile - purported source filename (same as 'file' by default)
459 if optimize >= 0:
460 opt = '' if optimize == 0 else optimize
461 cfile = importlib.util.cache_from_source(file, optimization=opt)
462 else:
463 cfile = importlib.util.cache_from_source(file)
464 dfile = file
465 if prefix:
466 if file[: len(prefix)] != prefix:
467 raise ValueError(
468 f"invalid prefix: filename {file!r} doesn't start with {prefix!r}"
469 )
  • E203 Whitespace before ':'
470 dfile = dfile[len(prefix) :]
471 if base_dir:
472 dfile = os.path.join(base_dir, dfile)
473  
474 cfile_base = os.path.basename(cfile)
475 if direct:
476 if force or newer(file, cfile):
477 log.info("byte-compiling %s to %s", file, cfile_base)
478 if not dry_run:
479 compile(file, cfile, dfile)
480 else:
481 log.debug("skipping byte-compilation of %s to %s", file, cfile_base)

484 def rfc822_escape(header):
485 """Return a version of the string escaped for inclusion in an
486 RFC-822 header, by ensuring there are 8 spaces space after each newline.
487 """
488 indent = 8 * " "
489 lines = header.splitlines(keepends=True)
490  
491 # Emulate the behaviour of `str.split`
492 # (the terminal line break in `splitlines` does not result in an extra line):
493 ends_in_newline = lines and lines[-1].splitlines()[0] != lines[-1]
494 suffix = indent if ends_in_newline else ""
495  
496 return indent.join(lines) + suffix
