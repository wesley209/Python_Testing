191 def testLongLongPathNames(self):
192 # We need filename where the FQN is > 256 - simplest way is to create a
193 # 250 character directory in the cwd (except - cwd may be on a drive
194 # not supporting \\\\?\\ (eg, network share) - so use temp.
195 import win32file
196  
197 basename = "a" * 250
198 # but we need to ensure we use the 'long' version of the
199 # temp dir for later comparison.
200 long_temp_dir = win32api.GetLongPathNameW(tempfile.gettempdir())
201 fname = "\\\\?\\" + os.path.join(long_temp_dir, basename)
202 try:
203 win32file.CreateDirectoryW(fname, None)
204 except win32api.error as details:
205 if details.winerror != winerror.ERROR_ALREADY_EXISTS:
206 raise
207 try:
208 # GetFileAttributes automatically calls GetFileAttributesW when
209 # passed unicode
210 try:
211 attr = win32api.GetFileAttributes(fname)
212 except win32api.error as details:
213 if details.winerror != winerror.ERROR_FILENAME_EXCED_RANGE:
214 raise
215  
216 attr = win32api.GetFileAttributes(str(fname))
217 self.assertTrue(attr & win32con.FILE_ATTRIBUTE_DIRECTORY, attr)
218  
219 long_name = win32api.GetLongPathNameW(fname)
220 self.assertEqual(long_name.lower(), fname.lower())
