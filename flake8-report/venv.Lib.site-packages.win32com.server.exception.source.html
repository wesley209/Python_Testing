23 """An Exception object that is understood by the framework.
24  
25 If the framework is presented with an exception of type class,

28

59 # convert a WIN32 error into an HRESULT
60 scode = scode or hresult
61 if scode and scode != 1: # We don't want S_FALSE mapped!
62 if scode >= -32768 and scode < 32768:
63 # this is HRESULT_FROM_WIN32()
64 scode = -2147024896 | (scode & 0x0000FFFF)
65 self.scode = scode
66  
67 self.description = description or desc
68 if scode == 1 and not self.description:
69 self.description = "S_FALSE"
