135 def calculate_response_time_percentile(response_times: dict[int, int], num_requests: int, percent: float) -> int:
136 """
137 Get the response time that a certain number of percent of the requests
138 finished within. Arguments:
139  
140 response_times: A StatsEntry.response_times dict
141 num_requests: Number of request made (could be derived from response_times,
142 but we save some CPU cycles by using the value which we already store)
143 percent: The percentile we want to calculate. Specified in range: 0.0 - 1.0
144 """
145 num_of_request = int(num_requests * percent)
146  
147 processed_count = 0
148 for response_time in sorted(response_times.keys(), reverse=True):
149 processed_count += response_times[response_time]
150 if num_requests - processed_count <= num_of_request:
151 return response_time
152 # if all response times were None
153 return 0

156 def diff_response_time_dicts(latest: dict[int, int], old: dict[int, int]) -> dict[int, int]:
157 """
158 Returns the delta between two {response_times:request_count} dicts.
159  
160 Used together with the response_times cache to get the response times for the
161 last X seconds, which in turn is used to calculate the current response time
162 percentiles.
163 """
164 new = {}
165 for t in latest:
166 if diff := latest[t] - old.get(t, 0):
167 new[t] = diff
168 return new

479 def extend(self, other: StatsEntry) -> None:
480 """
481 Extend the data from the current StatsEntry with the stats from another
482 StatsEntry instance.
483 """
484 # save the old last_request_timestamp, to see if we should store a new copy
485 # of the response times in the response times cache
486 old_last_request_timestamp = self.last_request_timestamp
487  
488 if self.last_request_timestamp is not None and other.last_request_timestamp is not None:
489 self.last_request_timestamp = max(self.last_request_timestamp, other.last_request_timestamp)
490 elif other.last_request_timestamp is not None:
491 self.last_request_timestamp = other.last_request_timestamp
492 self.start_time = min(self.start_time, other.start_time)
493 self.num_requests += other.num_requests
494 self.num_none_requests += other.num_none_requests
495 self.num_failures += other.num_failures
496 self.total_response_time += other.total_response_time
497 self.max_response_time = max(self.max_response_time, other.max_response_time)
498 if self.min_response_time is not None and other.min_response_time is not None:
499 self.min_response_time = min(self.min_response_time, other.min_response_time)
500 elif other.min_response_time is not None:
501 # this means self.min_response_time is None, so we can safely replace it
502 self.min_response_time = other.min_response_time
503 self.total_content_length += other.total_content_length
504  
505 for key in other.response_times:
506 self.response_times[key] = self.response_times.get(key, 0) + other.response_times[key]
507 for key in other.num_reqs_per_sec:
508 self.num_reqs_per_sec[key] = self.num_reqs_per_sec.get(key, 0) + other.num_reqs_per_sec[key]
509 for key in other.num_fail_per_sec:
510 self.num_fail_per_sec[key] = self.num_fail_per_sec.get(key, 0) + other.num_fail_per_sec[key]
511  
512 if self.use_response_times_cache:
513 # If we've entered a new second, we'll cache the response times. Note that there
514 # might still be reports from other worker nodes - that contains requests for the same
515 # time periods - that hasn't been received/accounted for yet. This will cause the cache to
516 # lag behind a second or two, but since StatsEntry.current_response_time_percentile()
517 # (which is what the response times cache is used for) uses an approximation of the
518 # last 10 seconds anyway, it should be fine to ignore this.
519 last_time = int(self.last_request_timestamp) if self.last_request_timestamp else None
520 if last_time and last_time > (old_last_request_timestamp and int(old_last_request_timestamp) or 0):
521 self._cache_response_times(last_time)

590 def get_current_response_time_percentile(self, percent: float) -> int | None:
591 """
592 Calculate the *current* response time for a certain percentile. We use a sliding
593 window of (approximately) the last 10 seconds (specified by CURRENT_RESPONSE_TIME_PERCENTILE_WINDOW)
594 when calculating this.
595 """
596 if not self.use_response_times_cache:
597 raise ValueError(
598 "StatsEntry.use_response_times_cache must be set to True if we should be able to calculate the _current_ response time percentile"
599 )
600 # First, we want to determine which of the cached response_times dicts we should
601 # use to get response_times for approximately 10 seconds ago.
602 t = int(time.time())
603 # Since we can't be sure that the cache contains an entry for every second.
604 # We'll construct a list of timestamps which we consider acceptable keys to be used
605 # when trying to fetch the cached response_times. We construct this list in such a way
606 # that it's ordered by preference by starting to add t-10, then t-11, t-9, t-12, t-8,
607 # and so on
608 acceptable_timestamps: list[int] = []
609 acceptable_timestamps.append(t - CURRENT_RESPONSE_TIME_PERCENTILE_WINDOW)
610 for i in range(1, 9):
611 acceptable_timestamps.append(t - CURRENT_RESPONSE_TIME_PERCENTILE_WINDOW - i)
612 acceptable_timestamps.append(t - CURRENT_RESPONSE_TIME_PERCENTILE_WINDOW + i)
613  
614 cached: CachedResponseTimes | None = None
615 if self.response_times_cache is not None:
616 for ts in acceptable_timestamps:
617 if ts in self.response_times_cache:
618 cached = self.response_times_cache[ts]
619 break
620  
621 if cached:
622 # If we found an acceptable cached response times, we'll calculate a new response
623 # times dict of the last 10 seconds (approximately) by diffing it with the current
624 # total response times. Then we'll use that to calculate a response time percentile
625 # for that timeframe
626 return calculate_response_time_percentile(
627 diff_response_time_dicts(self.response_times, cached.response_times),
628 self.num_requests - cached.num_requests,
629 percent,
630 )
631 # if time was not in response times cache window
632 return None

650 self.response_times_cache[t] = CachedResponseTimes(
651 response_times=copy(self.response_times),
652 num_requests=self.num_requests,
653 )
654  
655 # We'll use a cache size of CURRENT_RESPONSE_TIME_PERCENTILE_WINDOW + 10 since - in the extreme case -
656 # we might still use response times (from the cache) for t-CURRENT_RESPONSE_TIME_PERCENTILE_WINDOW-10
657 # to calculate the current response time percentile, if we're missing cached values for the subsequent
658 # 20 seconds
659 cache_size = CURRENT_RESPONSE_TIME_PERCENTILE_WINDOW + 10
660  
661 if len(self.response_times_cache) > cache_size:
662 # only keep the latest 20 response_times dicts
663 for _ in range(len(self.response_times_cache) - cache_size):
664 self.response_times_cache.popitem(last=False)

698 @classmethod
699 def parse_error(cls, error: Exception | str | None) -> str:
700 if isinstance(error, str):
701 string_error = error
702 else:
703 string_error = repr(error)
704 target = "object at 0x"
705 target_index = string_error.find(target)
706 if target_index < 0:
707 return string_error
708 start = target_index + len(target) - 2
709 end = string_error.find(">", start)
710 if end < 0:
711 return string_error
712 hex_address = string_error[start:end]
713 return string_error.replace(hex_address, "0x....")

715 @classmethod
716 def create_key(cls, method: str, name: str, error: Exception | str | None) -> str:
717 key = f"{method}.{name}.{StatsError.parse_error(error)!r}"
718 return hashlib.sha256(key.encode("utf-8")).hexdigest()

723 def to_name(self) -> str:
724 error = self.error
725 if isinstance(error, str): # in distributed mode, all errors have been converted to strings
726 if error.startswith("CatchResponseError("):
727 # unwrap CatchResponseErrors
728 length = len("CatchResponseError(")
729 unwrapped_error = error[length:-1]
730 else:
731 unwrapped_error = error
732 else: # in standalone mode, errors are still objects
733 if isinstance(error, CatchResponseError):
734 # unwrap CatchResponseErrors
735 unwrapped_error = error.args[0]
736 else:
737 unwrapped_error = repr(error)

739 return f"{self.method} {self.name}: {unwrapped_error}"

770 """
771 total is the number of requests made
