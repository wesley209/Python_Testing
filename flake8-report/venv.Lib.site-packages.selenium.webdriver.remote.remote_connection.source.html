190 @classmethod
191 def get_remote_connection_headers(cls, parsed_url, keep_alive=False):
192 """Get headers for remote request.
193  
194 :Args:
195 - parsed_url - The parsed url
196 - keep_alive (Boolean) - Is this a keep-alive connection (default: False)
197 """
198  
199 system = platform.system().lower()
200 if system == "darwin":
201 system = "mac"
202  
203 headers = {
204 "Accept": "application/json",
205 "Content-Type": "application/json;charset=UTF-8",
206 "User-Agent": f"selenium/{__version__} (python {system})",
207 }
208  
209 if parsed_url.username:
210 base64string = b64encode(f"{parsed_url.username}:{parsed_url.password}".encode())
211 headers.update({"Authorization": f"Basic {base64string.decode()}"})
212  
213 if keep_alive:
214 headers.update({"Connection": "keep-alive"})
215  
216 return headers

219 if self._url.startswith("https://"):
220 return os.environ.get("https_proxy", os.environ.get("HTTPS_PROXY"))

224 def _identify_http_proxy_auth(self):
225 url = self._proxy_url
  • E203 Whitespace before ':'
226 url = url[url.find(":") + 3 :]
227 return "@" in url and len(url[: url.find("@")]) > 0

229 def _separate_http_proxy_auth(self):
230 url = self._proxy_url
231 protocol = url[: url.find(":") + 3]
  • E203 Whitespace before ':'
232 no_protocol = url[len(protocol) :]
233 auth = no_protocol[: no_protocol.find("@")]
  • E203 Whitespace before ':'
234 proxy_without_auth = protocol + no_protocol[len(auth) + 1 :]
235 return proxy_without_auth, auth

237 def _get_connection_manager(self):
238 pool_manager_init_args = {"timeout": self.get_timeout()}
239 if self._ca_certs:
240 pool_manager_init_args["cert_reqs"] = "CERT_REQUIRED"
241 pool_manager_init_args["ca_certs"] = self._ca_certs
242  
243 if self._proxy_url:
244 if self._proxy_url.lower().startswith("sock"):
245 from urllib3.contrib.socks import SOCKSProxyManager
246  
247 return SOCKSProxyManager(self._proxy_url, **pool_manager_init_args)
248 if self._identify_http_proxy_auth():
249 self._proxy_url, self._basic_proxy_auth = self._separate_http_proxy_auth()
250 pool_manager_init_args["proxy_headers"] = urllib3.make_headers(proxy_basic_auth=self._basic_proxy_auth)
251 return urllib3.ProxyManager(self._proxy_url, **pool_manager_init_args)
252  
253 return urllib3.PoolManager(**pool_manager_init_args)

255 def __init__(self, remote_server_addr: str, keep_alive: bool = False, ignore_proxy: bool = False):
256 self.keep_alive = keep_alive
257 self._url = remote_server_addr
258  
259 # Env var NO_PROXY will override this part of the code
260 _no_proxy = os.environ.get("no_proxy", os.environ.get("NO_PROXY"))
261 if _no_proxy:
262 for npu in _no_proxy.split(","):
263 npu = npu.strip()
264 if npu == "*":
265 ignore_proxy = True
266 break
267 n_url = parse.urlparse(npu)
268 remote_add = parse.urlparse(self._url)
269 if n_url.netloc:
270 if remote_add.netloc == n_url.netloc:
271 ignore_proxy = True
272 break
273 else:
274 if n_url.path in remote_add.netloc:
275 ignore_proxy = True
276 break
277  
278 self._proxy_url = self._get_proxy_url() if not ignore_proxy else None
279 if keep_alive:
280 self._conn = self._get_connection_manager()
281 self._commands = remote_commands

283 def execute(self, command, params):
284 """Send a command to the remote server.
285  
286 Any path substitutions required for the URL mapped to the command should be
287 included in the command parameters.
288  
289 :Args:
290 - command - A string specifying the command to execute.
291 - params - A dictionary of named parameters to send with the command as
292 its JSON payload.
293 """
294 command_info = self._commands[command]
295 assert command_info is not None, f"Unrecognised command {command}"
296 path_string = command_info[1]
297 path = string.Template(path_string).substitute(params)
298 substitute_params = {word[1:] for word in path_string.split("/") if word.startswith("$")} # remove dollar sign
299 if isinstance(params, dict) and substitute_params:
300 for word in substitute_params:
301 del params[word]
302 data = utils.dump_json(params)
303 url = f"{self._url}{path}"
304 trimmed = self._trim_large_entries(params)
305 LOGGER.debug("%s %s %s", command_info[0], url, str(trimmed))
306 return self._request(command_info[0], url, body=data)

308 def _request(self, method, url, body=None):
309 """Send an HTTP request to the remote server.
310  
311 :Args:
312 - method - A string for the HTTP method to send the request with.
313 - url - A string for the URL to send the request to.
314 - body - A string for request body. Ignored unless method is POST or PUT.
315  
316 :Returns:
317 A dictionary with the server's parsed JSON response.
318 """
319 parsed_url = parse.urlparse(url)
320 headers = self.get_remote_connection_headers(parsed_url, self.keep_alive)
321 response = None
322 if body and method not in ("POST", "PUT"):
323 body = None
324  
325 if self.keep_alive:
326 response = self._conn.request(method, url, body=body, headers=headers)
327 statuscode = response.status
328 else:
329 conn = self._get_connection_manager()
330 with conn as http:
331 response = http.request(method, url, body=body, headers=headers)
332 statuscode = response.status
333 data = response.data.decode("UTF-8")
334 LOGGER.debug("Remote response: status=%s | data=%s | headers=%s", response.status, data, response.headers)
335 try:
336 if 300 <= statuscode < 304:
337 return self._request("GET", response.headers.get("location", None))
338 if 399 < statuscode <= 500:
339 return {"status": statuscode, "value": data}
340 content_type = []
341 if response.headers.get("Content-Type", None):
342 content_type = response.headers.get("Content-Type", None).split(";")
343 if not any([x.startswith("image/png") for x in content_type]):
344 try:
345 data = utils.load_json(data.strip())
346 except ValueError:
347 if 199 < statuscode < 300:
348 status = ErrorCode.SUCCESS
349 else:
350 status = ErrorCode.UNKNOWN_ERROR
351 return {"status": status, "value": data.strip()}
352  
353 # Some drivers incorrectly return a response
354 # with no 'value' field when they should return null.
355 if "value" not in data:
356 data["value"] = None
357 return data
358 data = {"status": 0, "value": data}
359 return data
360 finally:
361 LOGGER.debug("Finished Request")
362 response.close()

369 def _trim_large_entries(self, input_dict, max_length=100):
370 """Truncate string values in a dictionary if they exceed max_length.
371  
372 :param dict: Dictionary with potentially large values
373 :param max_length: Maximum allowed length of string values
374 :return: Dictionary with truncated string values
375 """
376 output_dictionary = {}
377 for key, value in input_dict.items():
378 if isinstance(value, dict):
379 output_dictionary[key] = self._trim_large_entries(value, max_length)
380 elif isinstance(value, str) and len(value) > max_length:
381 output_dictionary[key] = value[:max_length] + "..."
382 else:
383 output_dictionary[key] = value
384  
385 return output_dictionary
