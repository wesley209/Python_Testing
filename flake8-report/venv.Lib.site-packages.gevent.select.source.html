111 def select(self, rlist, wlist, timeout):
112 watchers = []
113 # read and write are the collected ready objects, accumulated
114 # by the callback. Note that we could get spurious callbacks
115 # if the socket is closed while we're blocked. We can't easily
116 # detect that (libev filters the events passed so we can't
117 # pass arbitrary events). After an iteration of polling for
118 # IO, libev will invoke all the pending IO watchers, and then
119 # any newly added (fed) events, and then we will invoke added
120 # callbacks. With libev 4.27+ and EV_VERIFY, it's critical to
121 # close our watcher immediately once we get an event. That
122 # could be the close event (coming just before the actual
123 # close happens), and once the FD is closed, libev will abort
124 # the process if we stop the watcher.
125 read = []
126 write = []
127 event = Event()
128 add_read = self._make_callback(read, event, _EV_READ)
129 add_write = self._make_callback(write, event, _EV_WRITE)
130  
131 try:
132 self._make_watchers(watchers,
133 (rlist, add_read),
134 (wlist, add_write))
135 event.wait(timeout=timeout)
136 return read, write, []
137 finally:
138 self._closeall(watchers)

141 def select(rlist, wlist, xlist, timeout=None): # pylint:disable=unused-argument
142 """An implementation of :obj:`select.select` that blocks only the current greenlet.
143  
144 .. caution:: *xlist* is ignored.
145  
146 .. versionchanged:: 1.2a1
147 Raise a :exc:`ValueError` if timeout is negative. This matches Python 3's
148 behaviour (Python 2 would raise a ``select.error``). Previously gevent had
149 undefined behaviour.
150 .. versionchanged:: 1.2a1
151 Raise an exception if any of the file descriptors are invalid.
152 """
153 if timeout is not None and timeout < 0:
154 # Raise an error like the real implementation; which error
155 # depends on the version. Python 3, where select.error is OSError,
156 # raises a ValueError (which makes sense). Older pythons raise
157 # the error from the select syscall...but we don't actually get there.
158 # We choose to just raise the ValueError as it makes more sense and is
159 # forward compatible
160 raise ValueError("timeout must be non-negative")
161  
162 # since rlist and wlist can be any iterable we will have to first
163 # copy them into a list, so we can use them in both _original_select
164 # and in SelectResult.select. We don't need to do it for xlist, since
165 # that one will only be passed into _original_select
166 rlist = rlist if isinstance(rlist, (list, tuple)) else list(rlist)
167 wlist = wlist if isinstance(wlist, (list, tuple)) else list(wlist)
168  
169 # First, do a poll with the original select system call. This is
170 # the most efficient way to check to see if any of the file
171 # descriptors have previously been closed and raise the correct
172 # corresponding exception. (Because libev tends to just return
173 # them as ready, or, if built with EV_VERIFY >= 2 and libev >=
174 # 4.27, crash the process. And libuv also tends to crash the
175 # process.)
176 #
177 # We accept the *xlist* here even though we can't
178 # below because this is all about error handling.
179 sel_results = ((), (), ())
180 try:
181 sel_results = _original_select(rlist, wlist, xlist, 0)
182 except error as e:
183 enumber = getattr(e, 'errno', None) or e.args[0]
184 if enumber != EINTR:
185 # Ignore interrupted syscalls
186 raise
187  
188 if sel_results[0] or sel_results[1] or sel_results[2] or (timeout is not None and timeout == 0):
189 # If we actually had stuff ready, go ahead and return it. No need
190 # to go through the trouble of doing our own stuff.
191  
192 # Likewise, if the timeout is 0, we already did a 0 timeout
193 # select and we don't need to do it again. Note that in libuv,
194 # zero duration timers may be called immediately, without
195 # cycling the event loop at all. 2.7/test_telnetlib.py "hangs"
196 # calling zero-duration timers if we go to the loop here.
197  
198 # However, because this is typically a place where scheduling switches
199 # can occur, we need to make sure that's still the case; otherwise a single
200 # consumer could monopolize the thread. (shows up in test_ftplib.)
201 _g_sleep()
202 return sel_results
203  
204 result = SelectResult()
205 return result.select(rlist, wlist, timeout)

254

305 def poll(self, timeout=None):
306 """
307 poll the registered fds.
308  
309 .. versionchanged:: 1.2a1
310 File descriptors that are closed are reported with POLLNVAL.
311  
312 .. versionchanged:: 1.3a2
313 Under libuv, interpret *timeout* values less than 0 the same as *None*,
314 i.e., block. This was always the case with libev.
315 """
316 result = PollResult()
317 watchers = self._get_started_watchers(result.add_event)
318 try:
319 if timeout is not None:
320 if timeout < 0:
321 # The docs for python say that an omitted timeout,
322 # a negative timeout and a timeout of None are all
323 # supposed to block forever. Many, but not all
324 # OS's accept any negative number to mean that. Some
325 # OS's raise errors for anything negative but not -1.
326 # Python 3.7 changes to always pass exactly -1 in that
327 # case from selectors.
328  
329 # Our Timeout class currently does not have a defined behaviour
330 # for negative values. On libuv, it uses a check watcher and effectively
331 # doesn't block. On libev, it seems to block. In either case, we
332 # *want* to block, so turn this into the sure fire block request.
333 timeout = None
334 elif timeout:
335 # The docs for poll.poll say timeout is in
336 # milliseconds. Our result objects work in
337 # seconds, so this should be *=, shouldn't it?
338 timeout /= 1000.0
339 result.event.wait(timeout=timeout)
340 return list(result.events)
341 finally:
342 for awatcher in watchers:
343 awatcher.stop()
344 awatcher.close()
