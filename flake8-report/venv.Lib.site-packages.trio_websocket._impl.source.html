281 def _url_to_host(url, ssl_context):
282 '''
283 Convert a WebSocket URL to a (host,port,resource) tuple.
284  
285 The returned ``ssl_context`` is either the same object that was passed in,
286 or if ``ssl_context`` is None, then a bool indicating if a default SSL
287 context needs to be created.
288  
289 :param str url: A WebSocket URL.
290 :type ssl_context: ssl.SSLContext or None
291 :returns: A tuple of ``(host, port, resource, ssl_context)``.
292 '''
293 url = str(url) # For backward compat with isinstance(url, yarl.URL).
294 parts = urllib.parse.urlsplit(url)
295 if parts.scheme not in ('ws', 'wss'):
296 raise ValueError('WebSocket URL scheme must be "ws:" or "wss:"')
297 if ssl_context is None:
298 ssl_context = parts.scheme == 'wss'
299 elif parts.scheme == 'ws':
300 raise ValueError('SSL context must be None for ws: URL scheme')
301 host = parts.hostname
302 if parts.port is not None:
303 port = parts.port
304 else:
305 port = 443 if ssl_context else 80
306 path_qs = parts.path
307 # RFC 7230, Section 5.3.1:
308 # If the target URI's path component is empty, the client MUST
309 # send "/" as the path within the origin-form of request-target.
310 if not path_qs:
311 path_qs = '/'
312 if '?' in url:
313 path_qs += '?' + parts.query
314 return host, port, path_qs, ssl_context
