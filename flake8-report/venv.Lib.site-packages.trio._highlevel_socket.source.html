204 # Accept error handling
205 # =====================
206 #
207 # Literature review
208 # -----------------
209 #
210 # Here's a list of all the possible errors that accept() can return, according
211 # to the POSIX spec or the Linux, FreeBSD, macOS, and Windows docs:
212 #
213 # Can't happen with a Trio socket:
214 # - EAGAIN/(WSA)EWOULDBLOCK
215 # - EINTR
216 # - WSANOTINITIALISED
217 # - WSAEINPROGRESS: a blocking call is already in progress
218 # - WSAEINTR: someone called WSACancelBlockingCall, but we don't make blocking
219 # calls in the first place
220 #
221 # Something is wrong with our call:
222 # - EBADF: not a file descriptor
223 # - (WSA)EINVAL: socket isn't listening, or (Linux, BSD) bad flags
224 # - (WSA)ENOTSOCK: not a socket
225 # - (WSA)EOPNOTSUPP: this kind of socket doesn't support accept
226 # - (Linux, FreeBSD, Windows) EFAULT: the sockaddr pointer points to readonly
227 # memory
228 #
229 # Something is wrong with the environment:
230 # - (WSA)EMFILE: this process hit its fd limit
231 # - ENFILE: the system hit its fd limit
232 # - (WSA)ENOBUFS, ENOMEM: unspecified memory problems
233 #
234 # Something is wrong with the connection we were going to accept. There's a
235 # ton of variability between systems here:
236 # - ECONNABORTED: documented everywhere, but apparently only the BSDs do this
237 # (signals a connection was closed/reset before being accepted)
238 # - EPROTO: unspecified protocol error
239 # - (Linux) EPERM: firewall rule prevented connection
240 # - (Linux) ENETDOWN, EPROTO, ENOPROTOOPT, EHOSTDOWN, ENONET, EHOSTUNREACH,
241 # EOPNOTSUPP, ENETUNREACH, ENOSR, ESOCKTNOSUPPORT, EPROTONOSUPPORT,
242 # ETIMEDOUT, ... or any other error that the socket could give, because
243 # apparently if an error happens on a connection before it's accept()ed,
244 # Linux will report that error from accept().
245 # - (Windows) WSAECONNRESET, WSAENETDOWN
246 #
247 #
248 # Code review
249 # -----------
250 #
251 # What do other libraries do?
252 #
253 # Twisted on Unix or when using nonblocking I/O on Windows:
254 # - ignores EPERM, with comment about Linux firewalls
255 # - logs and ignores EMFILE, ENOBUFS, ENFILE, ENOMEM, ECONNABORTED
256 # Comment notes that ECONNABORTED is a BSDism and that Linux returns the
257 # socket before having it fail, and macOS just silently discards it.
258 # - other errors are raised, which is logged + kills the socket
259 # ref: src/twisted/internet/tcp.py, Port.doRead
260 #
261 # Twisted using IOCP on Windows:
262 # - logs and ignores all errors
263 # ref: src/twisted/internet/iocpreactor/tcp.py, Port.handleAccept
264 #
265 # Tornado:
266 # - ignore ECONNABORTED (comments notes that it was observed on FreeBSD)
267 # - everything else raised, but all this does (by default) is cause it to be
268 # logged and then ignored
269 # (ref: tornado/netutil.py, tornado/ioloop.py)
270 #
271 # libuv on Unix:
272 # - ignores ECONNABORTED
273 # - does a "trick" for EMFILE or ENFILE
274 # - all other errors passed to the connection_cb to be handled
275 # (ref: src/unix/stream.c:uv__server_io, uv__emfile_trick)
276 #
277 # libuv on Windows:
278 # src/win/tcp.c:uv_tcp_queue_accept
279 # this calls AcceptEx, and then arranges to call:
280 # src/win/tcp.c:uv_process_tcp_accept_req
281 # this gets the result from AcceptEx. If the original AcceptEx call failed,
282 # then "we stop accepting connections and report this error to the
283 # connection callback". I think this is for things like ENOTSOCK. If
284 # AcceptEx successfully queues an overlapped operation, and then that
285 # reports an error, it's just discarded.
286 #
287 # asyncio, selector mode:
288 # - ignores EWOULDBLOCK, EINTR, ECONNABORTED
289 # - on EMFILE, ENFILE, ENOBUFS, ENOMEM, logs an error and then disables the
290 # listening loop for 1 second
291 # - everything else raises, but then the event loop just logs and ignores it
292 # (selector_events.py: BaseSelectorEventLoop._accept_connection)
293 #
294 #
295 # What should we do?
296 # ------------------
297 #
298 # When accept() returns an error, we can either ignore it or raise it.
299 #
300 # We have a long list of errors that should be ignored, and a long list of
301 # errors that should be raised. The big question is what to do with an error
302 # that isn't on either list. On Linux apparently you can get nearly arbitrary
303 # errors from accept() and they should be ignored, because it just indicates a
304 # socket that crashed before it began, and there isn't really anything to be
305 # done about this, plus on other platforms you may not get any indication at
306 # all, so programs have to tolerate not getting any indication too. OTOH if we
307 # get an unexpected error then it could indicate something arbitrarily bad --
308 # after all, it's unexpected.
309 #
310 # Given that we know that other libraries seem to be getting along fine with a
311 # fairly minimal list of errors to ignore, I think we'll be OK if we write
312 # down that list and then raise on everything else.
313 #
314 # The other question is what to do about the capacity problem errors: EMFILE,
315 # ENFILE, ENOBUFS, ENOMEM. Just flat out ignoring these is clearly not optimal
316 # -- at the very least you want to log them, and probably you want to take
317 # some remedial action. And if we ignore them then it prevents higher levels
318 # from doing anything clever with them. So we raise them.
319  
320 _ignorable_accept_errno_names = [
321 # Linux can do this when the a connection is denied by the firewall
322 "EPERM",
323 # BSDs with an early close/reset
324 "ECONNABORTED",
325 # All the other miscellany noted above -- may not happen in practice, but
326 # whatever.
327 "EPROTO",
328 "ENETDOWN",
329 "ENOPROTOOPT",
330 "EHOSTDOWN",
331 "ENONET",
332 "EHOSTUNREACH",
333 "EOPNOTSUPP",
334 "ENETUNREACH",
335 "ENOSR",
336 "ESOCKTNOSUPPORT",
337 "EPROTONOSUPPORT",
338 "ETIMEDOUT",
339 "ECONNRESET",
340 ]

342 # Not all errnos are defined on all platforms

350 class SocketListener(Listener[SocketStream]):
351 """A :class:`~trio.abc.Listener` that uses a listening socket to accept
352 incoming connections as :class:`SocketStream` objects.
353  
354 Args:
355 socket: The Trio socket object to wrap. Must have type ``SOCK_STREAM``,

357

359 socket; closing the :class:`SocketListener` will also close the socket.
360  
361 .. attribute:: socket
362  
363 The Trio socket object that this stream wraps.
364

383 async def accept(self) -> SocketStream:
384 """Accept an incoming connection.
385  
386 Returns:
387 :class:`SocketStream`
388  
389 Raises:
390 OSError: if the underlying call to ``accept`` raises an unexpected
391 error.
392 ClosedResourceError: if you already closed the socket.
393  
394 This method handles routine errors like ``ECONNABORTED``, but passes
395 other errors on to its caller. In particular, it does *not* make any
396 special effort to handle resource exhaustion errors like ``EMFILE``,
397 ``ENFILE``, ``ENOBUFS``, ``ENOMEM``.
398  
399 """
400 while True:
401 try:
402 sock, _ = await self.socket.accept()
403 except OSError as exc:
404 if exc.errno in _closed_stream_errnos:
405 raise trio.ClosedResourceError from None
406 if exc.errno not in _ignorable_accept_errnos:
407 raise
408 else:
409 return SocketStream(sock)
