30 def update_wrapper(wrapper,
31 wrapped,
32 assigned=WRAPPER_ASSIGNMENTS,
33 updated=WRAPPER_UPDATES):
34 """
35 Based on code from the standard library ``functools``, but
36 doesn't perform any of the troublesome imports.
37  
38 functools imports RLock from _thread for purposes of the
39 ``lru_cache``, making it problematic to use from gevent.
40  
41 The other imports are somewhat heavy: abc, collections, types.
42 """
43 for attr in assigned:
44 try:
45 value = getattr(wrapped, attr)
46 except AttributeError:
47 pass
48 else:
49 setattr(wrapper, attr, value)
50 for attr in updated:
51 getattr(wrapper, attr).update(getattr(wrapped, attr, {}))
52 # Issue #17482: set __wrapped__ last so we don't inadvertently copy it
53 # from the wrapped function when updating __dict__
54 wrapper.__wrapped__ = wrapped
55 # Return the wrapper so this can be used as a decorator via partial()
56 return wrapper

66 """
67 Copy attributes defined in ``source.__dict__`` to the dictionary
68 in globs (which should be the caller's :func:`globals`).
69  
70 Names that start with ``__`` are ignored (unless they are in
71 *dunder_names_to_keep*). Anything found in *names_to_ignore* is
72 also ignored.
73  
74 If *only_names* is given, only those attributes will be
75 considered. In this case, *ignore_missing_names* says whether or
76 not to raise an :exc:`AttributeError` if one of those names can't
77 be found.
78  
79 If *cleanup_globs* has a true value, then common things imported but
80 not used at runtime are removed, including this function.
81  
82 Returns a list of the names copied; this should be assigned to ``__imports__``.
83 """
84 if only_names:
85 if ignore_missing_names:
86 items = ((k, getattr(source, k, _NONE)) for k in only_names)
87 else:
88 items = ((k, getattr(source, k)) for k in only_names)
89 else:
90 items = iteritems(source.__dict__)
91  
92 copied = []
93 for key, value in items:
94 if value is _NONE:
95 continue
96 if key in names_to_ignore:
97 continue
98 if key.startswith("__") and key not in dunder_names_to_keep:
99 continue
100 globs[key] = value
101 copied.append(key)
102  
103 if cleanup_globs:
104 if 'copy_globals' in globs:
105 del globs['copy_globals']
106  
107 return copied
