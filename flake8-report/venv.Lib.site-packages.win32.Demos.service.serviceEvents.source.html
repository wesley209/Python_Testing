1 # A Demo of a service that takes advantage of the additional notifications
2 # available in later Windows versions.

4 # Note that all output is written as event log entries - so you must install

43 # Override the base class so we can accept additional events.
44 def GetAcceptedControls(self):
45 # say we accept them all.
46 rc = win32serviceutil.ServiceFramework.GetAcceptedControls(self)
47 rc |= (
48 win32service.SERVICE_ACCEPT_PARAMCHANGE
49 | win32service.SERVICE_ACCEPT_NETBINDCHANGE
50 | win32service.SERVICE_CONTROL_DEVICEEVENT
51 | win32service.SERVICE_ACCEPT_HARDWAREPROFILECHANGE
52 | win32service.SERVICE_ACCEPT_POWEREVENT
53 | win32service.SERVICE_ACCEPT_SESSIONCHANGE
54 )
55 return rc

57 # All extra events are sent via SvcOtherEx (SvcOther remains as a
58 # function taking only the first args for backwards compat)
59 def SvcOtherEx(self, control, event_type, data):
60 # This is only showing a few of the extra events - see the MSDN
61 # docs for "HandlerEx callback" for more info.
62 if control == win32service.SERVICE_CONTROL_DEVICEEVENT:
63 info = win32gui_struct.UnpackDEV_BROADCAST(data)
64 msg = f"A device event occurred: {event_type:x} - {info}"
65 elif control == win32service.SERVICE_CONTROL_HARDWAREPROFILECHANGE:
66 msg = f"A hardware profile changed: type={event_type}, data={data}"
67 elif control == win32service.SERVICE_CONTROL_POWEREVENT:
68 msg = "A power event: setting %s" % data
69 elif control == win32service.SERVICE_CONTROL_SESSIONCHANGE:
70 # data is a single elt tuple, but this could potentially grow
71 # in the future if the win32 struct does
72 msg = f"Session event: type={event_type}, data={data}"
73 else:
74 msg = "Other event: code=%d, type=%s, data=%s" % (control, event_type, data)
