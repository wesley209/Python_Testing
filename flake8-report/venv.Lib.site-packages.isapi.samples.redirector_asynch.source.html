26 # The callback made when IIS completes the asynch write.
27 def io_callback(ecb, fp, cbIO, errcode):
28 print("IO callback", ecb, fp, cbIO, errcode)
29 chunk = fp.read(CHUNK_SIZE)
30 if chunk:
31 ecb.WriteClient(chunk, isapicon.HSE_IO_ASYNC)
32 # and wait for the next callback to say this chunk is done.
33 else:
34 # eof - say we are complete.
35 fp.close()
36 ecb.DoneWithSession()

43 def Dispatch(self, ecb):
44 print('IIS dispatching "{}"'.format(ecb.GetServerVariable("URL")))
45 url = ecb.GetServerVariable("URL")

47 new_url = proxy + url
48 print("Opening %s" % new_url)
49 fp = urllib.request.urlopen(new_url)
50 headers = fp.info()
51 ecb.SendResponseHeaders("200 OK", str(headers) + "\r\n", False)
52 # now send the first chunk asynchronously
53 ecb.ReqIOCompletion(io_callback, fp)
54 chunk = fp.read(CHUNK_SIZE)
55 if chunk:
56 ecb.WriteClient(chunk, isapicon.HSE_IO_ASYNC)
57 return isapicon.HSE_STATUS_PENDING
58 # no data - just close things now.
59 ecb.DoneWithSession()
60 return isapicon.HSE_STATUS_SUCCESS
