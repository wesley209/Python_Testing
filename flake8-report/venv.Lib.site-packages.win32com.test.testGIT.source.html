1 """Testing pasing object between multiple COM threads
2  
3 Uses standard COM marshalling to pass objects between threads. Even
4 though Python generally seems to work when you just pass COM objects
5 between threads, it shouldn't.
6  
7 This shows the "correct" way to do it.
8  
9 It shows that although we create new threads to use the Python.Interpreter,
10 COM marshalls back all calls to that object to the main Python thread,
11 which must be running a message loop (as this sample does).
12  
13 When this test is run in "free threaded" mode (at this stage, you must
14 manually mark the COM objects as "ThreadingModel=Free", or run from a
15 service which has marked itself as free-threaded), then no marshalling
16 is done, and the Python.Interpreter object start doing the "expected" thing
17 - ie, it reports being on the same thread as its caller!
18  
19 Python.exe needs a good way to mark itself as FreeThreaded - at the moment
20 this is a pain in the but!
21 """

82 def BeginThreadsSimpleMarshal(numThreads, cookie):
83 """Creates multiple threads using simple (but slower) marshalling.
84  
85 Single interpreter object, but a new stream is created per thread.
86  
87 Returns the handles the threads will set when complete.
88 """
89 ret = []
90 for i in range(numThreads):
91 hEvent = win32event.CreateEvent(None, 0, 0, None)
92 _thread.start_new(TestInterpInThread, (hEvent, cookie))
93 ret.append(hEvent)
94 return ret

98 print("The main thread is %d" % (win32api.GetCurrentThreadId()))
99 GIT = CreateGIT()
100 interp = win32com.client.Dispatch("Python.Interpreter")
101 cookie = GIT.RegisterInterfaceInGlobal(interp._oleobj_, pythoncom.IID_IDispatch)

103 events = fn(4, cookie)
104 numFinished = 0
105 while 1:
106 try:
107 rc = win32event.MsgWaitForMultipleObjects(
108 events, 0, 2000, win32event.QS_ALLINPUT

110 if rc >= win32event.WAIT_OBJECT_0 and rc < win32event.WAIT_OBJECT_0 + len(
111 events

113 numFinished += 1
114 if numFinished >= len(events):
115 break
116 elif rc == win32event.WAIT_OBJECT_0 + len(events): # a message
