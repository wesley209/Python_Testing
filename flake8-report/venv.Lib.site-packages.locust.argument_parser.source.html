54 def add_argument(self, *args, **kwargs) -> configargparse.Action:
55 """
56 This method supports the same args as ArgumentParser.add_argument(..)
57 as well as the additional args below.
58  
59 Arguments:
60 include_in_web_ui: If True (default), the argument will show in the UI.
61 is_secret: If True (default is False) and include_in_web_ui is True, the argument will show in the UI with a password masked text input.
62  
63 Returns:
64 argparse.Action: the new argparse action
65 """
66 include_in_web_ui = kwargs.pop("include_in_web_ui", True)
67 is_secret = kwargs.pop("is_secret", False)
68 action = super().add_argument(*args, **kwargs)
69 action.include_in_web_ui = include_in_web_ui
70 action.is_secret = is_secret
71 return action
72  
73 @property
74 def args_included_in_web_ui(self) -> dict[str, configargparse.Action]:
75 return {a.dest: a for a in self._actions if hasattr(a, "include_in_web_ui") and a.include_in_web_ui}

77 @property
78 def secret_args_included_in_web_ui(self) -> dict[str, configargparse.Action]:
79 return {
80 a.dest: a
81 for a in self._actions
82 if a.dest in self.args_included_in_web_ui and hasattr(a, "is_secret") and a.is_secret

86 class LocustTomlConfigParser(configargparse.TomlConfigParser):
87 def parse(self, stream):
88 try:
89 config = tomllib.loads(stream.read())
90 except Exception as e:
91 raise configargparse.ConfigFileParserException(f"Couldn't parse TOML file: {e}")
92  
93 # convert to dict and filter based on section names
94 result = OrderedDict()
95  
96 for section in self.sections:
97 if data := configargparse.get_toml_section(config, section):
98 for key, value in data.items():
99 if isinstance(value, list):
100 result[key] = value
101 elif value is None:
102 pass
103 else:
104 result[key] = str(value)
105 break
106  
107 return result

110 def parse_locustfile_paths(paths: list[str]) -> list[str]:
111 """
112 Returns a list of relative file paths.
113  
114 Args:
115 paths (list[str]): paths taken from the -f command
116  
117 Returns:
118 list[str]: Parsed locust file paths
119 """
120 # Parse each path and unpack the returned lists as a single list
121 return [parsed for path in paths for parsed in _parse_locustfile_path(path)]

124 def _parse_locustfile_path(path: str) -> list[str]:
125 parsed_paths = []
126 if is_url(path):
127 # Download the file and use the new path as locustfile
128 parsed_paths.append(download_locustfile_from_url(path))
129 elif os.path.isdir(path):
130 # Find all .py files in directory tree
131 parsed_paths.extend(get_abspaths_in(path, extension=".py"))
132 if not parsed_paths:
133 sys.stderr.write(f"Could not find any locustfiles in directory '{path}'")
134 sys.exit(1)
135 else:
136 # If file exists add the abspath
137 if os.path.exists(path) and path.endswith(".py"):
138 parsed_paths.append(os.path.abspath(path))
139 else:
140 note_about_file_endings = "Ensure your locustfile ends with '.py' or is a directory with locustfiles. "
141 sys.stderr.write(f"Could not find '{path}'. {note_about_file_endings}See --help for available options.\n")
142 sys.exit(1)
143  
144 return parsed_paths

147 def download_locustfile_from_url(url: str) -> str:
148 """
149 Attempt to download and save locustfile from url.
150 Returns path to downloaded file.
151 """
152 try:
153 response = requests.get(url)
154 # Check if response is valid python code
155 ast.parse(response.text)
156 except requests.exceptions.RequestException as e:
157 sys.stderr.write(f"Failed to get locustfile from: {url}. Exception: {e}")
158 sys.exit(1)
159 except SyntaxError:
160 sys.stderr.write(f"Failed to get locustfile from: {url}. Response is not valid python code.")
161 sys.exit(1)
162  
163 with open(os.path.join(tempfile.gettempdir(), urlparse(url).path.split("/")[-1]), "w") as locustfile:
164 locustfile.write(response.text)
165  
166 atexit.register(exit_handler, locustfile.name)
167 return locustfile.name

216 def download_locustfile_from_master(master_host: str, master_port: int) -> str:
217 client_id = socket.gethostname() + "_download_locustfile_" + uuid4().hex
218 tempclient = zmqrpc.Client(master_host, master_port, client_id)
219 got_reply = False
220  
221 def ask_for_locustfile():
222 while not got_reply:
223 tempclient.send(Message("locustfile", {"version": version}, client_id))
224 gevent.sleep(1)
225  
226 def log_warning():
227 gevent.sleep(10)
228 while not got_reply:
229 sys.stderr.write("Waiting to connect to master to receive locustfile...\n")
230 gevent.sleep(60)
231  
232 def wait_for_reply():
233 return tempclient.recv()
234  
235 gevent.spawn(ask_for_locustfile)
236 gevent.spawn(log_warning)
237  
238 try:
239 # wait same time as for client_ready ack. not that it is really relevant...
240 msg = gevent.spawn(wait_for_reply).get(timeout=runners.CONNECT_TIMEOUT * runners.CONNECT_RETRY_COUNT)
241 got_reply = True
242 except gevent.Timeout:
243 sys.stderr.write(
244 f"Got no locustfile response from master, gave up after {runners.CONNECT_TIMEOUT * runners.CONNECT_RETRY_COUNT}s\n"
245 )
246 sys.exit(1)
247  
248 if msg.type != "locustfile":
249 sys.stderr.write(f"Got wrong message type from master {msg.type}\n")
250 sys.exit(1)
251  
252 if "error" in msg.data:
253 sys.stderr.write(f"Got error from master: {msg.data['error']}\n")
254 sys.exit(1)
255  
256 tempclient.close()
257 return msg.data.get("locustfiles", [])

278 def parse_locustfile_option(args=None) -> list[str]:
279 """
280 Construct a command line parser that is only used to parse the -f argument so that we can
281 import the test scripts in case any of them adds additional command line arguments to the
282 parser
283  
284 Returns:
285 parsed_paths (List): List of locustfile paths
286 """
287 parser = get_empty_argument_parser(add_help=False)
288 parser.add_argument(
289 "-h",
290 "--help",
291 action="store_true",
292 default=False,
293 )
294 parser.add_argument(
295 "--version",
296 "-V",
297 action="store_true",
298 default=False,
299 )
300 # the following arguments are only used for downloading the locustfile from master
301 parser.add_argument(
302 "--worker",
303 action="store_true",
304 env_var="LOCUST_MODE_WORKER",
305 )
306 parser.add_argument(
307 "--master", # this is just here to prevent argparse from giving the dreaded "ambiguous option: --master could match --master-host, --master-port"
308 action="store_true",
309 env_var="LOCUST_MODE_MASTER",
310 )
311 parser.add_argument(
312 "--master-host",
313 default="127.0.0.1",
314 env_var="LOCUST_MASTER_NODE_HOST",
315 )
316 parser.add_argument(
317 "--master-port",
318 type=int,
319 default=5557,
320 env_var="LOCUST_MASTER_NODE_PORT",
321 )
322  
323 options, _ = parser.parse_known_args(args=args)
324  
325 if options.help or options.version:
326 # if --help or --version is specified we'll call parse_options which will print the help/version message
327 parse_options(args=args)
328  
329 if options.locustfile == "-":
330 if not options.worker:
331 sys.stderr.write(
332 "locustfile was set to '-' (meaning to download from master) but --worker was not specified.\n"
333 )
334 sys.exit(1)
335 # having this in argument_parser module is a bit weird, but it needs to be done early
336 locustfile_sources = download_locustfile_from_master(options.master_host, options.master_port)
337 locustfile_list = parse_locustfiles_from_master(locustfile_sources)
338 else:
339 locustfile_list = [f.strip() for f in options.locustfile.split(",")]
340  
341 parsed_paths = parse_locustfile_paths(locustfile_list)
342  
343 if not parsed_paths:
344 note_about_file_endings = ""
345 user_friendly_locustfile_name = options.locustfile
346  
347 if not options.locustfile.endswith(".py"):
348 note_about_file_endings = "Ensure your locustfile ends with '.py' or is a directory with parsed_paths. "
349 sys.stderr.write(
350 f"Could not find '{user_friendly_locustfile_name}'. {note_about_file_endings}See --help for available options.\n"
351 )
352 sys.exit(1)
353  
354 return parsed_paths

763 def get_parser(default_config_files=DEFAULT_CONFIG_FILES) -> LocustArgumentParser:
764 # get a parser that is only able to parse the -f argument
765 parser = get_empty_argument_parser(add_help=True, default_config_files=default_config_files)
766 # add all the other supported arguments
767 setup_parser_arguments(parser)
768 # fire event to provide a hook for locustscripts and plugins to add command line arguments
769 locust.events.init_command_line_parser.fire(parser=parser)
770 return parser

773 def parse_options(args=None) -> configargparse.Namespace:
774 parser = get_parser()
775 parsed_opts = parser.parse_args(args=args)
776 if parsed_opts.stats_history_enabled and (parsed_opts.csv_prefix is None):
777 parser.error("'--csv-full-history' requires '--csv'.")
778 return parsed_opts

781 def default_args_dict() -> dict:
782 # returns a dict containing the default arguments (before any custom arguments are added)
783 default_parser = get_empty_argument_parser()
784 setup_parser_arguments(default_parser)
785 # Dont read config files because they may contain custom arguments, which would fail parsing in the next step
786 default_parser._default_config_files = {}
787 return vars(default_parser.parse([]))
