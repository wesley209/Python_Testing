460 def __handle_death_before_start(self, args):
461 # args is (t, v, tb) or simply t or v.
462 # The last two cases are transformed into (t, v, None);
463 # if the single argument is an exception type, a new instance
464 # is created; if the single argument is not an exception type and also
465 # not an exception, it is wrapped in a BaseException (this is not
466 # documented, but should result in better behaviour in the event of a
467 # user error---instead of silently printing something to stderr, we still
468 # kill the greenlet).
469 if self._exc_info is None and self.dead:
470 # the greenlet was never switched to before and it will
471 # never be; _report_error was not called, the result was
472 # not set, and the links weren't notified. Let's do it
473 # here.
474 #
475 # checking that self.dead is true is essential, because
476 # throw() does not necessarily kill the greenlet (if the
477 # exception raised by throw() is caught somewhere inside
478 # the greenlet).
479 if len(args) == 1:
480 arg = args[0]
481 if isinstance(arg, type) and issubclass(arg, BaseException):
482 args = (arg, arg(), None)
483 else:
484 args = (type(arg), arg, None)
485 elif not args:
486 args = (GreenletExit, GreenletExit(), None)
487 if not issubclass(args[0], BaseException):
488 # Random non-type, non-exception arguments.
489 args = (BaseException, BaseException(args), None)
490 assert issubclass(args[0], BaseException)
491 self.__report_error(args)
