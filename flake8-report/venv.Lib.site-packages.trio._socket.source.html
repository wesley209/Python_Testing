41 # _stdlib_socket.socket supports 13 different socket families, see
42 # https://docs.python.org/3/library/socket.html#socket-families
43 # and the return type of several methods in SocketType will depend on those. Typeshed
44 # has ended up typing those return types as `Any` in most cases, but for users that
45 # know which family/families they're working in we could make SocketType a generic type,
46 # where you specify the return values you expect from those methods depending on the
47 # protocol the socket will be handling.
48 # But without the ability to default the value to `Any` it will be overly cumbersome for
49 # most users, so currently we just specify it as `Any`. Otherwise we would write:
50 # `AddressFormat = TypeVar("AddressFormat")`
51 # but instead we simply do:
52 AddressFormat: TypeAlias = Any

103 def set_custom_hostname_resolver(
104 hostname_resolver: HostnameResolver | None,
105 ) -> HostnameResolver | None:
106 """Set a custom hostname resolver.
107  
108 By default, Trio's :func:`getaddrinfo` and :func:`getnameinfo` functions
109 use the standard system resolver functions. This function allows you to
110 customize that behavior. The main intended use case is for testing, but it
111 might also be useful for using third-party resolvers like `c-ares
112 <https://c-ares.haxx.se/>`__ (though be warned that these rarely make
113 perfect drop-in replacements for the system resolver). See
114 :class:`trio.abc.HostnameResolver` for more details.
115  
116 Setting a custom hostname resolver affects all future calls to
117 :func:`getaddrinfo` and :func:`getnameinfo` within the enclosing call to
118 :func:`trio.run`. All other hostname resolution in Trio is implemented in
119 terms of these functions.
120  
121 Generally you should call this function just once, right at the beginning
122 of your program.
123  
124 Args:
125 hostname_resolver (trio.abc.HostnameResolver or None): The new custom
126 hostname resolver, or None to restore the default behavior.
127  
128 Returns:
129 The previous hostname resolver (which may be None).
130  
131 """
132 old = _resolver.get(None)
133 _resolver.set(hostname_resolver)
134 return old

190 """Look up a numeric address given a name.
191  
192 Arguments and return values are identical to :func:`socket.getaddrinfo`,
193 except that this version is async.
194  
195 Also, :func:`trio.socket.getaddrinfo` correctly uses IDNA 2008 to process
196 non-ASCII domain names. (:func:`socket.getaddrinfo` uses IDNA 2003, which
197 can give the wrong result in some cases and cause you to connect to a
198 different host than the one you intended; see `bpo-17305
199 <https://bugs.python.org/issue17305>`__.)
200  
201 This function's behavior can be customized using
202 :func:`set_custom_hostname_resolver`.
203  
204 """
205  
206 # If host and port are numeric, then getaddrinfo doesn't block and we can
207 # skip the whole thread thing, which seems worthwhile. So we try first
208 # with the _NUMERIC_ONLY flags set, and then only spawn a thread if that
209 # fails with EAI_NONAME:
210 def numeric_only_failure(exc: BaseException) -> bool:
211 return (
212 isinstance(exc, _stdlib_socket.gaierror)
213 and exc.errno == _stdlib_socket.EAI_NONAME
214 )
215  
216 async with _try_sync(numeric_only_failure):
217 return _stdlib_socket.getaddrinfo(
218 host,
219 port,
220 family,
221 type,
222 proto,
223 flags | _NUMERIC_ONLY,
224 )
225 # That failed; it's a real hostname. We better use a thread.
226 #
227 # Also, it might be a unicode hostname, in which case we want to do our
228 # own encoding using the idna module, rather than letting Python do
229 # it. (Python will use the old IDNA 2003 standard, and possibly get the
230 # wrong answer - see bpo-17305). However, the idna module is picky, and
231 # will refuse to process some valid hostname strings, like "::1". So if
232 # it's already ascii, we pass it through; otherwise, we encode it to.
233 if isinstance(host, str):
234 try:
235 host = host.encode("ascii")
236 except UnicodeEncodeError:
237 # UTS-46 defines various normalizations; in particular, by default
238 # idna.encode will error out if the hostname has Capital Letters
239 # in it; with uts46=True it will lowercase them instead.
240 host = _idna.encode(host, uts46=True)
241 hr = _resolver.get(None)
242 if hr is not None:
243 return await hr.getaddrinfo(host, port, family, type, proto, flags)
244 else:
245 return await trio.to_thread.run_sync(
246 _stdlib_socket.getaddrinfo,
247 host,
248 port,
249 family,
250 type,
251 proto,
252 flags,
253 abandon_on_cancel=True,
254 )

385 def _sniff_sockopts_for_fileno(
386 family: AddressFamily | int,
387 type_: SocketKind | int,
388 proto: int,
389 fileno: int | None,
390 ) -> tuple[AddressFamily | int, SocketKind | int, int]:
391 """Correct SOCKOPTS for given fileno, falling back to provided values."""
392 # Wrap the raw fileno into a Python socket object
393 # This object might have the wrong metadata, but it lets us easily call getsockopt
394 # and then we'll throw it away and construct a new one with the correct metadata.
395 if sys.platform != "linux":
396 return family, type_, proto
397 from socket import ( # type: ignore[attr-defined,unused-ignore]
398 SO_DOMAIN,
399 SO_PROTOCOL,
400 SO_TYPE,
401 SOL_SOCKET,
402 )
403  
404 sockobj = _stdlib_socket.socket(family, type_, proto, fileno=fileno)
405 try:
406 family = sockobj.getsockopt(SOL_SOCKET, SO_DOMAIN)
407 proto = sockobj.getsockopt(SOL_SOCKET, SO_PROTOCOL)
408 type_ = sockobj.getsockopt(SOL_SOCKET, SO_TYPE)
409 finally:
410 # Unwrap it again, so that sockobj.__del__ doesn't try to close our socket
411 sockobj.detach()
412 return family, type_, proto

457 # Take an address in Python's representation, and returns a new address in
458 # the same representation, but with names resolved to numbers,
459 # etc.
460 #
461 # local=True means that the address is being used with bind() or similar
462 # local=False means that the address is being used with connect() or sendto() or
463 # similar.
464 #
465  
466  
467 # Using a TypeVar to indicate we return the same type of address appears to give errors
468 # when passed a union of address types.
469 # @overload likely works, but is extremely verbose.
470 # NOTE: this function does not always checkpoint
471 async def _resolve_address_nocp(
472 type_: int,
473 family: AddressFamily,
474 proto: int,
475 *,
476 ipv6_v6only: bool | int,
477 address: AddressFormat,
478 local: bool,
479 ) -> Any:
480 # Do some pre-checking (or exit early for non-IP sockets)
481 if family == _stdlib_socket.AF_INET:
482 if not isinstance(address, tuple) or not len(address) == 2:
483 raise ValueError("address should be a (host, port) tuple")
484 elif family == _stdlib_socket.AF_INET6:
485 if not isinstance(address, tuple) or not 2 <= len(address) <= 4:
486 raise ValueError(
487 "address should be a (host, port, [flowinfo, [scopeid]]) tuple",
488 )
489 elif hasattr(_stdlib_socket, "AF_UNIX") and family == _stdlib_socket.AF_UNIX:
490 # unwrap path-likes
491 assert isinstance(address, (str, bytes, os.PathLike))
492 return os.fspath(address)
493 else:
494 return address
495  
496 # -- From here on we know we have IPv4 or IPV6 --
497 host: str | None
498 host, port, *_ = address
499 # Fast path for the simple case: already-resolved IP address,
500 # already-resolved port. This is particularly important for UDP, since
501 # every sendto call goes through here.
502 if isinstance(port, int) and host is not None:
503 try:
504 _stdlib_socket.inet_pton(family, host)
505 except (OSError, TypeError):
506 pass
507 else:
508 return address
509 # Special cases to match the stdlib, see gh-277
510 if host == "":
511 host = None
512 if host == "<broadcast>":
513 host = "255.255.255.255"
514 flags = 0
515 if local:
516 flags |= _stdlib_socket.AI_PASSIVE
517 # Since we always pass in an explicit family here, AI_ADDRCONFIG
518 # doesn't add any value -- if we have no ipv6 connectivity and are
519 # working with an ipv6 socket, then things will break soon enough! And
520 # if we do enable it, then it makes it impossible to even run tests
521 # for ipv6 address resolution on travis-ci, which as of 2017-03-07 has
522 # no ipv6.
523 # flags |= AI_ADDRCONFIG
524 if family == _stdlib_socket.AF_INET6 and not ipv6_v6only:
525 flags |= _stdlib_socket.AI_V4MAPPED
526 gai_res = await getaddrinfo(host, port, family, type_, proto, flags)
527 # AFAICT from the spec it's not possible for getaddrinfo to return an
528 # empty list.
529 assert len(gai_res) >= 1
530 # Address is the last item in the first entry
531 (*_, normed), *_ = gai_res
532 # The above ignored any flowid and scopeid in the passed-in address,
533 # so restore them if present:
534 if family == _stdlib_socket.AF_INET6:
535 list_normed = list(normed)
536 assert len(normed) == 4
537 if len(address) >= 3:
538 list_normed[2] = address[2]
539 if len(address) >= 4:
540 list_normed[3] = address[3]
541 return tuple(list_normed)
542 return normed

985 async def _nonblocking_helper(
986 self,
987 wait_fn: Callable[[_stdlib_socket.socket], Awaitable[None]],
988 fn: Callable[Concatenate[_stdlib_socket.socket, P], T],
989 *args: P.args,
990 **kwargs: P.kwargs,
991 ) -> T:
992 # We have to reconcile two conflicting goals:
993 # - We want to make it look like we always blocked in doing these
994 # operations. The obvious way is to always do an IO wait before
995 # calling the function.
996 # - But, we also want to provide the correct semantics, and part
997 # of that means giving correct errors. So, for example, if you
998 # haven't called .listen(), then .accept() raises an error
999 # immediately. But in this same circumstance, then on macOS, the
1000 # socket does not register as readable. So if we block waiting
1001 # for read *before* we call accept, then we'll be waiting
1002 # forever instead of properly raising an error. (On Linux,
1003 # interestingly, AFAICT a socket that can't possible read/write
1004 # *does* count as readable/writable for select() purposes. But
1005 # not on macOS.)
1006 #
1007 # So, we have to call the function once, with the appropriate
1008 # cancellation/yielding sandwich if it succeeds, and if it gives
1009 # BlockingIOError *then* we fall back to IO wait.
1010 #
1011 # XX think if this can be combined with the similar logic for IOCP
1012 # submission...
1013 async with _try_sync():
1014 return fn(self._sock, *args, **kwargs)
1015 # First attempt raised BlockingIOError:
1016 while True:
1017 await wait_fn(self._sock)
1018 try:
1019 return fn(self._sock, *args, **kwargs)
1020 except BlockingIOError:
1021 pass

1041 async def connect(self, address: AddressFormat) -> None:
1042 # nonblocking connect is weird -- you call it to start things
1043 # off, then the socket becomes writable as a completion
1044 # notification. This means it isn't really cancellable... we close the
1045 # socket if cancelled, to avoid confusion.
1046 try:
1047 address = await self._resolve_address_nocp(address, local=False)
1048 async with _try_sync():
1049 # An interesting puzzle: can a non-blocking connect() return EINTR
1050 # (= raise InterruptedError)? PEP 475 specifically left this as
1051 # the one place where it lets an InterruptedError escape instead
1052 # of automatically retrying. This is based on the idea that EINTR
1053 # from connect means that the connection was already started, and
1054 # will continue in the background. For a blocking connect, this
1055 # sort of makes sense: if it returns EINTR then the connection
1056 # attempt is continuing in the background, and on many system you
1057 # can't then call connect() again because there is already a
1058 # connect happening. See:
1059 #
1060 # http://www.madore.org/~david/computers/connect-intr.html
1061 #
1062 # For a non-blocking connect, it doesn't make as much sense --
1063 # surely the interrupt didn't happen after we successfully
1064 # initiated the connect and are just waiting for it to complete,
1065 # because a non-blocking connect does not wait! And the spec
1066 # describes the interaction between EINTR/blocking connect, but
1067 # doesn't have anything useful to say about non-blocking connect:
1068 #
1069 # http://pubs.opengroup.org/onlinepubs/007904975/functions/connect.html
1070 #
1071 # So we have a conundrum: if EINTR means that the connect() hasn't
1072 # happened (like it does for essentially every other syscall),
1073 # then InterruptedError should be caught and retried. If EINTR
1074 # means that the connect() has successfully started, then
1075 # InterruptedError should be caught and ignored. Which should we
1076 # do?
1077 #
1078 # In practice, the resolution is probably that non-blocking
1079 # connect simply never returns EINTR, so the question of how to
1080 # handle it is moot. Someone spelunked macOS/FreeBSD and
1081 # confirmed this is true there:
1082 #
1083 # https://stackoverflow.com/questions/14134440/eintr-and-non-blocking-calls
1084 #
1085 # and exarkun seems to think it's true in general of non-blocking
1086 # calls:
1087 #
1088 # https://twistedmatrix.com/pipermail/twisted-python/2010-September/022864.html
1089 # (and indeed, AFAICT twisted doesn't try to handle
1090 # InterruptedError).
1091 #
1092 # So we don't try to catch InterruptedError. This way if it
1093 # happens, someone will hopefully tell us, and then hopefully we
1094 # can investigate their system to figure out what its semantics
1095 # are.
1096 return self._sock.connect(address)
1097 # It raised BlockingIOError, meaning that it's started the
1098 # connection attempt. We wait for it to complete:
1099 await _core.wait_writable(self._sock)
1100 except trio.Cancelled:
1101 # We can't really cancel a connect, and the socket is in an
1102 # indeterminate state. Better to close it so we don't get
1103 # confused.
1104 self._sock.close()
1105 raise
1106 # Okay, the connect finished, but it might have failed:
1107 err = self._sock.getsockopt(_stdlib_socket.SOL_SOCKET, _stdlib_socket.SO_ERROR)
1108 if err != 0:
1109 raise OSError(err, f"Error connecting to {address!r}: {os.strerror(err)}")
