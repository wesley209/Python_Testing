89 def _construct_basic_auth_str(username, password):
90 """Construct Authorization header value to be used in HTTP Basic Auth"""
91 if isinstance(username, str):
92 username = username.encode("latin1")
93 if isinstance(password, str):
94 password = password.encode("latin1")
95 return "Basic " + b64encode(b":".join((username, password))).strip().decode("ascii")
