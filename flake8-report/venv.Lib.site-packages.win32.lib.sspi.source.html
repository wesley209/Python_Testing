1 """
2 Helper classes for SSPI authentication via the win32security module.
3  
4 SSPI authentication involves a token-exchange "dance", the exact details
5 of which depends on the authentication provider used. There are also
6 a number of complex flags and constants that need to be used - in most
7 cases, there are reasonable defaults.
8  
9 These classes attempt to hide these details from you until you really need
10 to know. They are not designed to handle all cases, just the common ones.
11 If you need finer control than offered here, just use the win32security
12 functions directly.
13 """

45 def encrypt(self, data):
46 """Encrypt a string, returning a tuple of (encrypted_data, trailer).
47 These can be passed to decrypt to get back the original string.
48 """
49 pkg_size_info = self.ctxt.QueryContextAttributes(sspicon.SECPKG_ATTR_SIZES)
50 trailersize = pkg_size_info["SecurityTrailer"]
51  
52 encbuf = win32security.PySecBufferDescType()
53 encbuf.append(win32security.PySecBufferType(len(data), sspicon.SECBUFFER_DATA))
54 encbuf.append(
55 win32security.PySecBufferType(trailersize, sspicon.SECBUFFER_TOKEN)
56 )
57 encbuf[0].Buffer = data
58 self.ctxt.EncryptMessage(0, encbuf, self._get_next_seq_num())
59 return encbuf[0].Buffer, encbuf[1].Buffer

69 encbuf[1].Buffer = trailer

73 def sign(self, data):
74 """sign a string suitable for transmission, returning the signature.
75 Passing the data and signature to verify will determine if the data
76 is unchanged.
77 """
78 pkg_size_info = self.ctxt.QueryContextAttributes(sspicon.SECPKG_ATTR_SIZES)
79 sigsize = pkg_size_info["MaxSignature"]

81 sigbuf.append(win32security.PySecBufferType(len(data), sspicon.SECBUFFER_DATA))

100 def unwrap(self, token):
101 """
102 GSSAPI's unwrap with SSPI.
103 https://docs.microsoft.com/en-us/windows/win32/secauthn/sspi-kerberos-interoperability-with-gssapi
104  
105 Usable mainly with Kerberos SSPI package, but this is not enforced.
106  
107 Return the clear text, and a boolean that is True if the token was encrypted.
108 """
109 buffer = win32security.PySecBufferDescType()
110 # This buffer will contain a "stream", which is the token coming from the other side
111 buffer.append(
112 win32security.PySecBufferType(len(token), sspicon.SECBUFFER_STREAM)
113 )
114 buffer[0].Buffer = token
115  
116 # This buffer will receive the clear, or just unwrapped text if no encryption was used.
117 # Will be resized by the lib.
118 buffer.append(win32security.PySecBufferType(0, sspicon.SECBUFFER_DATA))
119  
120 pfQOP = self.ctxt.DecryptMessage(buffer, self._get_next_seq_num())
121  
122 r = buffer[1].Buffer
123 return r, not (pfQOP == sspicon.SECQOP_WRAP_NO_ENCRYPT)

125 def wrap(self, msg, encrypt=False):
126 """
127 GSSAPI's wrap with SSPI.
128 https://docs.microsoft.com/en-us/windows/win32/secauthn/sspi-kerberos-interoperability-with-gssapi
129  
130 Usable mainly with Kerberos SSPI package, but this is not enforced.
131  
132 Wrap a message to be sent to the other side. Encrypted if encrypt is True.
133 """
134  
135 size_info = self.ctxt.QueryContextAttributes(sspicon.SECPKG_ATTR_SIZES)
136 trailer_size = size_info["SecurityTrailer"]
137 block_size = size_info["BlockSize"]
138  
139 buffer = win32security.PySecBufferDescType()
140  
141 # This buffer will contain unencrypted data to wrap, and maybe encrypt.
142 buffer.append(win32security.PySecBufferType(len(msg), sspicon.SECBUFFER_DATA))
143 buffer[0].Buffer = msg
144  
145 # Will receive the token that forms the beginning of the msg
146 buffer.append(
147 win32security.PySecBufferType(trailer_size, sspicon.SECBUFFER_TOKEN)
148 )
149  
150 # The trailer is needed in case of block encryption
151 buffer.append(
152 win32security.PySecBufferType(block_size, sspicon.SECBUFFER_PADDING)
153 )
154  
155 fQOP = 0 if encrypt else sspicon.SECQOP_WRAP_NO_ENCRYPT
156 self.ctxt.EncryptMessage(fQOP, buffer, self._get_next_seq_num())
157  
158 # Sec token, then data, then padding
159 r = buffer[1].Buffer + buffer[0].Buffer + buffer[2].Buffer
160 return r
