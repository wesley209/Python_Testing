94 # See: #461 as to why this is needed.
95 # The gist is that threading.main_thread() has the capability to lie to us
96 # if somebody else edits the threading ident cache to replace the main
97 # thread; causing threading.current_thread() to return a _DummyThread,
98 # causing the C-c check to fail, and so on.
99 # Trying to use signal out of the main thread will fail, so we can then
100 # reliably check if this is the main thread without relying on a
101 # potentially modified threading.
102 def is_main_thread() -> bool:
103 """Attempt to reliably check if we are in the main thread."""
104 try:
105 signal.signal(signal.SIGINT, signal.getsignal(signal.SIGINT))
106 return True
107 except (TypeError, ValueError):
108 return False
