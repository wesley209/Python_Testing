6 """Provides a class Stdin which can be used to emulate the regular old
7 sys.stdin for the PythonWin interactive window. Right now it just pops
8 up a input() dialog. With luck, someone will integrate it into the
9 actual PythonWin interactive window someday.
10  
11 WARNING: Importing this file automatically replaces sys.stdin with an
12 instance of Stdin (below). This is useful because you can just open
13 Stdin.py in PythonWin and hit the import button to get it set up right
14 if you don't feel like changing PythonWin's source. To put things back
15 the way they were, simply use this magic incantation:
16 import sys
17 sys.stdin = sys.stdin.real_file
18 """

24 class Stdin:
25 def __init__(self):
26 self.real_file = sys.stdin # NOTE: Likely to be None
27 self.buffer = ""
28 self.closed = False
29  
30 def __getattr__(self, name):
31 """Forward most functions to the real sys.stdin for absolute realism."""
32 if self.real_file is None:
33 raise AttributeError(name)
34 return getattr(self.real_file, name)
35  
36 def isatty(self):
37 """Return 1 if the file is connected to a tty(-like) device, else 0."""
38 return 1

40 def read(self, size=-1):
41 """Read at most size bytes from the file (less if the read
42 hits EOF or no more data is immediately available on a pipe,
43 tty or similar device). If the size argument is negative or
44 omitted, read all data until EOF is reached. The bytes are
45 returned as a string object. An empty string is returned when
46 EOF is encountered immediately. (For certain files, like ttys,
47 it makes sense to continue reading after an EOF is hit.)"""
48 result_size = self.__get_lines(size)
49 return self.__extract_from_buffer(result_size)
50  
51 def readline(self, size=-1):
52 """Read one entire line from the file. A trailing newline
53 character is kept in the string2.6 (but may be absent when a file ends
54 with an incomplete line). If the size argument is present and
55 non-negative, it is a maximum byte count (including the trailing
56 newline) and an incomplete line may be returned. An empty string is
57 returned when EOF is hit immediately. Note: unlike stdio's fgets(),
58 the returned string contains null characters ('\0') if they occurred
59 in the input.
60 """
61 maximum_result_size = self.__get_lines(size, lambda buffer: "\n" in buffer)
62  
63 if "\n" in self.buffer[:maximum_result_size]:
64 result_size = self.buffer.find("\n", 0, maximum_result_size) + 1
65 assert result_size > 0
66 else:
67 result_size = maximum_result_size
68  
69 return self.__extract_from_buffer(result_size)
70  
71 def __extract_from_buffer(self, character_count):
72 """Remove the first character_count characters from the internal buffer and
73 return them.
74 """
75 result = self.buffer[:character_count]
76 self.buffer = self.buffer[character_count:]
77 return result
78  
79 def __get_lines(self, desired_size, done_reading=lambda buffer: False):
80 """Keep adding lines to our internal buffer until done_reading(self.buffer)
81 is true or EOF has been reached or we have desired_size bytes in the buffer.
82 If desired_size < 0, we are never satisfied until we reach EOF. If done_reading
83 is not supplied, it is not consulted.
84  
85 If desired_size < 0, returns the length of the internal buffer. Otherwise,
86 returns desired_size.
87 """
88 while not done_reading(self.buffer) and (
89 desired_size < 0 or len(self.buffer) < desired_size
90 ):
91 try:
92 self.__get_line()
93 except (
94 EOFError,
95 KeyboardInterrupt,
96 ): # deal with cancellation of get_input_line dialog
97 desired_size = len(self.buffer) # Be satisfied!
98  
99 if desired_size < 0:
100 return len(self.buffer)
101 else:
102 return desired_size
103  
104 def __get_line(self):
105 """Grab one line from get_input_line() and append it to the buffer."""
106 line = get_input_line()
107 print(">>>", line) # echo input to console
108 self.buffer = self.buffer + line + "\n"
109  
110 def readlines(self, *sizehint):
111 """Read until EOF using readline() and return a list containing the lines
112 thus read. If the optional sizehint argument is present, instead of
113 reading up to EOF, whole lines totalling approximately sizehint bytes
114 (possibly after rounding up to an internal buffer size) are read.
115 """
116 result = []
117 total_read = 0
118 while sizehint == () or total_read < sizehint[0]:
119 line = self.readline()
120 if line == "":
121 break
122 total_read += len(line)
123 result.append(line)
124 return result
