34 class StreamServer(BaseServer):
35 """
36 A generic TCP server.
37  
38 Accepts connections on a listening socket and spawns user-provided
39 *handle* function for each connection with 2 arguments: the client
40 socket and the client address.
41  
42 Note that although the errors in a successfully spawned handler
43 will not affect the server or other connections, the errors raised
44 by :func:`accept` and *spawn* cause the server to stop accepting
45 for a short amount of time. The exact period depends on the values
46 of :attr:`min_delay` and :attr:`max_delay` attributes.
47  
48 The delay starts with :attr:`min_delay` and doubles with each
49 successive error until it reaches :attr:`max_delay`. A successful
50 :func:`accept` resets the delay to :attr:`min_delay` again.
51  
52 See :class:`~gevent.baseserver.BaseServer` for information on defining the *handle*
53 function and important restrictions on it.
54  
55 **SSL Support**
56  
57 The server can optionally work in SSL mode when given the correct
58 keyword arguments. (That is, the presence of any keyword arguments
59 will trigger SSL mode.) On Python 2.7.9 and later (any Python
60 version that supports the :class:`ssl.SSLContext`), this can be
61 done with a configured ``SSLContext``. On any Python version, it
62 can be done by passing the appropriate arguments for
63 :func:`ssl.wrap_socket`.
64  
65 The incoming socket will be wrapped into an SSL socket before
66 being passed to the *handle* function.
67  
68 If the *ssl_context* keyword argument is present, it should
69 contain an :class:`ssl.SSLContext`. The remaining keyword
70 arguments are passed to the :meth:`ssl.SSLContext.wrap_socket`
71 method of that object. Depending on the Python version, supported arguments
72 may include:
73  
74 - server_hostname
75 - suppress_ragged_eofs
76 - do_handshake_on_connect
77  
78 .. caution:: When using an SSLContext, it should either be
79 imported from :mod:`gevent.ssl`, or the process needs to be monkey-patched.
80 If the process is not monkey-patched and you pass the standard library
81 SSLContext, the resulting client sockets will not cooperate with gevent.
82  
83 Otherwise, keyword arguments are assumed to apply to :func:`ssl.wrap_socket`.
84 These keyword arguments may include:
85  
86 - keyfile
87 - certfile
88 - cert_reqs
89 - ssl_version
90 - ca_certs
91 - suppress_ragged_eofs
92 - do_handshake_on_connect
93 - ciphers
94  
95 .. versionchanged:: 1.2a2
96 Add support for the *ssl_context* keyword argument.
97

258 def _tcp_listener(address, backlog=50, reuse_addr=None, family=AF_INET):
259 """A shortcut to create a TCP socket, bind it and put it into listening state."""
260 sock = GeventSocket(family=family)
261 if reuse_addr is not None:
262 sock.setsockopt(SOL_SOCKET, SO_REUSEADDR, reuse_addr)
263 try:
264 sock.bind(address)
265 except SocketError as ex:
266 strerror = getattr(ex, 'strerror', None)
267 if strerror is not None:
268 ex.strerror = strerror + ': ' + repr(address)
269 raise
270 sock.listen(backlog)
271 sock.setblocking(0)
272 return sock
