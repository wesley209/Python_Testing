37 def VS_FIXEDFILEINFO(maj, min, sub, build, debug=0, is_dll=1):
38 return struct.pack(
39 "lllllllllllll",
40 VS_FFI_SIGNATURE, # dwSignature
41 VS_FFI_STRUCVERSION, # dwStrucVersion
42 (maj << 16) | min, # dwFileVersionMS
43 (sub << 16) | build, # dwFileVersionLS
44 (maj << 16) | min, # dwProductVersionMS
45 (sub << 16) | build, # dwProductVersionLS
46 VS_FFI_FILEFLAGSMASK, # dwFileFlagsMask
47 file_flags(debug), # dwFileFlags
48 VOS_NT_WINDOWS32, # dwFileOS
49 file_type(is_dll), # dwFileType
50 0x00000000, # dwFileSubtype
51 0x00000000, # dwFileDateMS
52 0x00000000, # dwFileDateLS
53 )

56 def nullterm(s):
57 # get raw bytes for a NULL terminated unicode string.
58 return (str(s) + "\0").encode("utf-16le")

61 def pad32(s, extra=2):
62 # extra is normally 2 to deal with wLength
  • E741 Ambiguous variable name 'l'
63 l = 4 - ((len(s) + extra) & 3)
64 if l < 4:
65 return s + (null_byte * l)
66 return s

73 def String(key, value):
74 key = nullterm(key)
75 value = nullterm(value)
76 result = struct.pack("hh", len(value) // 2, 1) # wValueLength, wType
77 result += key
78 result = pad32(result) + value
79 return addlen(result)

82 def StringTable(key, data):
83 key = nullterm(key)
84 result = struct.pack("hh", 0, 1) # wValueLength, wType
85 result += key
86 for k, v in data.items():
87 result += String(k, v)
88 result = pad32(result)
89 return addlen(result)

101 result = struct.pack("hh", len(value), 0) # wValueLength, wType

116 def VS_VERSION_INFO(maj, min, sub, build, sdata, vdata, debug=0, is_dll=1):
117 ffi = VS_FIXEDFILEINFO(maj, min, sub, build, debug, is_dll)
118 result = struct.pack("hh", len(ffi), 0) # wValueLength, wType
119 result += nullterm("VS_VERSION_INFO")
120 result = pad32(result) + ffi
121 result = pad32(result) + StringFileInfo(sdata) + VarFileInfo(vdata)
122 return addlen(result)

125 def stamp(pathname, options):
126 # For some reason, the API functions report success if the file is open
127 # but doesn't work! Try and open the file for writing, just to see if it is
128 # likely the stamp will work!
129 try:
130 f = open(pathname, "a+b")
131 f.close()
132 except OSError as why:
133 print(f"WARNING: File {pathname} could not be opened - {why}")
134  
135 ver = options.version
136 try:
137 bits = [int(i) for i in ver.split(".")]
138 vmaj, vmin, vsub, vbuild = bits
139 except (IndexError, TypeError, ValueError):
140 raise ValueError("--version must be a.b.c.d (all integers) - got %r" % ver)
141  
142 ifn = options.internal_name
143 if not ifn:
144 ifn = os.path.basename(pathname)
145 ofn = options.original_filename
146 if ofn is None:
147 ofn = os.path.basename(pathname)
148  
149 sdata = {
150 "Comments": options.comments,
151 "CompanyName": options.company,
152 "FileDescription": options.description,
153 "FileVersion": ver,
154 "InternalName": ifn,
155 "LegalCopyright": options.copyright,
156 "LegalTrademarks": options.trademarks,
157 "OriginalFilename": ofn,
158 "ProductName": options.product,
159 "ProductVersion": ver,
160 }
161 vdata = {
162 "Translation": struct.pack("hh", 0x409, 1252),
163 }
164 is_dll = options.dll
165 if is_dll is None:
166 is_dll = os.path.splitext(pathname)[1].lower() in ".dll .pyd".split()
167 is_debug = options.debug
168 if is_debug is None:
169 is_debug = os.path.splitext(pathname)[0].lower().endswith("_d")
170 # convert None to blank strings
171 for k, v in list(sdata.items()):
172 if v is None:
173 sdata[k] = ""
174 vs = VS_VERSION_INFO(vmaj, vmin, vsub, vbuild, sdata, vdata, is_debug, is_dll)
175  
176 h = BeginUpdateResource(pathname, 0)
177 UpdateResource(h, 16, 1, vs)
178 EndUpdateResource(h, 0)
179  
180 if options.verbose:
181 print("Stamped:", pathname)
