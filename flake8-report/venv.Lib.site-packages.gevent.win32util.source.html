33 class _ErrorFormatter(object):
34 """
35 Formatter for Windows error messages.
36  
37 @ivar winError: A callable which takes one integer error number argument
38 and returns an L{exceptions.WindowsError} instance for that error (like
39 L{ctypes.WinError}).
40  
41 @ivar formatMessage: A callable which takes one integer error number
42 argument and returns a C{str} giving the message for that error (like
43 L{win32api.FormatMessage}).
44  
45 @ivar errorTab: A mapping from integer error numbers to C{str} messages
46 which correspond to those errors (like L{socket.errorTab}).
47 """
48 def __init__(self, WinError, FormatMessage, errorTab):
49 self.winError = WinError
50 self.formatMessage = FormatMessage
51 self.errorTab = errorTab

53 @classmethod
54 def fromEnvironment(cls):
55 """
56 Get as many of the platform-specific error translation objects as
57 possible and return an instance of C{cls} created with them.
58 """
59 try:
60 from ctypes import WinError
61 except ImportError:
62 WinError = None
63 try:
64 from win32api import FormatMessage
65 except ImportError:
66 FormatMessage = None
67 try:
68 from socket import errorTab
69 except ImportError:
70 errorTab = None
71 return cls(WinError, FormatMessage, errorTab)

73 def formatError(self, errorcode):
74 """
75 Returns the string associated with a Windows error message, such as the
76 ones found in socket.error.
77  
78 Attempts direct lookup against the win32 API via ctypes and then
79 pywin32 if available), then in the error table in the socket module,
80 then finally defaulting to C{os.strerror}.
81  
82 @param errorcode: the Windows error code
83 @type errorcode: C{int}
84  
85 @return: The error message string
86 @rtype: C{str}
87 """
88 if self.winError is not None:
89 return str(self.winError(errorcode))
90 if self.formatMessage is not None:
91 return self.formatMessage(errorcode)
92 if self.errorTab is not None:
93 result = self.errorTab.get(errorcode)
94 if result is not None:
95 return result
96 return os.strerror(errorcode)
