1 # Extension to format a paragraph
2  
3 # Does basic, standard text formatting, and also understands Python
4 # comment blocks. Thus, for editing Python source code, this
5 # extension is really only suitable for reformatting these comment
6 # blocks or triple-quoted strings.
7  
8 # Known problems with comment reformatting:
9 # * If there is a selection marked, and the first line of the
10 # selection is not complete, the block will probably not be detected
11 # as comments, and will have the normal "text formatting" rules
12 # applied.
13 # * If a comment block has leading whitespace that mixes tabs and
14 # spaces, they will not be considered part of the same block.
15 # * Fancy comments, like this bulleted list, aren't handled :-)

90 def find_paragraph(text, mark):
91 lineno, col = list(map(int, mark.split(".")))
92 line = text.get("%d.0" % lineno, "%d.0 lineend" % lineno)
93 while text.compare("%d.0" % lineno, "<", "end") and is_all_white(line):
94 lineno += 1
95 line = text.get("%d.0" % lineno, "%d.0 lineend" % lineno)
96 first_lineno = lineno
97 comment_header = get_comment_header(line)
98 comment_header_len = len(comment_header)
99 while get_comment_header(line) == comment_header and not is_all_white(
100 line[comment_header_len:]
101 ):
102 lineno += 1
103 line = text.get("%d.0" % lineno, "%d.0 lineend" % lineno)
104 last = "%d.0" % lineno
105 # Search back to beginning of paragraph
106 lineno = first_lineno - 1
107 line = text.get("%d.0" % lineno, "%d.0 lineend" % lineno)
108 while (
109 lineno > 0
110 and get_comment_header(line) == comment_header
111 and not is_all_white(line[comment_header_len:])
112 ):
113 lineno -= 1
114 line = text.get("%d.0" % lineno, "%d.0 lineend" % lineno)
115 first = "%d.0" % (lineno + 1)
116 return first, last, comment_header, text.get(first, last)

119 def reformat_paragraph(data, limit=70):
120 lines = data.split("\n")
121 i = 0
122 n = len(lines)
123 while i < n and is_all_white(lines[i]):
124 i += 1
125 if i >= n:
126 return data
127 indent1 = get_indent(lines[i])
128 if i + 1 < n and not is_all_white(lines[i + 1]):
129 indent2 = get_indent(lines[i + 1])
130 else:
131 indent2 = indent1
132 new = lines[:i]
133 partial = indent1
134 while i < n and not is_all_white(lines[i]):
135 # XXX Should take double space after period (etc.) into account
136 words = re.split(r"(\s+)", lines[i])
137 for j in range(0, len(words), 2):
138 word = words[j]
139 if not word:
140 continue # Can happen when line ends in whitespace
141 if len((partial + word).expandtabs()) > limit and partial != indent1:
142 new.append(partial.rstrip())
143 partial = indent2
144 partial += word + " "
145 if j + 1 < len(words) and words[j + 1] != " ":
146 partial += " "
147 i += 1
148 new.append(partial.rstrip())
149 # XXX Should reformat remaining paragraphs as well
150 new.extend(lines[i:])
151 return "\n".join(new)

154 def is_all_white(line):
155 return re.match(r"^\s*$", line) is not None

158 def get_indent(line):
159 return re.match(r"^(\s*)", line).group()

162 def get_comment_header(line):
163 m = re.match(r"^(\s*#*)", line)
164 if m is None:
165 return ""
166 return m.group(1)
