37 def _collects(func):
38 # We've seen OSError: No such file or directory /proc/PID/fd/NUM.
39 # This occurs in the loop that checks open files. It first does
40 # listdir() and then tries readlink() on each file. But the file
41 # went away. This must be because of async GC in PyPy running
42 # destructors at arbitrary times. This became an issue in PyPy 7.2
43 # but could theoretically be an issue with any objects caught in a
44 # cycle. This is one reason we GC before we begin. (The other is
45 # to clean up outstanding objects that will close files in
46 # __del__.)
47 #
48 # Note that this can hide errors, though, by causing greenlets to get
49 # collected and drop references and thus close files. We should be deterministic
50 # and careful about closing things.
51 @functools.wraps(func)
52 def f(**kw):
53 gc.collect()
54 gc.collect()
55 enabled = gc.isenabled()
56 gc.disable()
57  
58 try:
59 return func(**kw)
60 finally:
61 if enabled:
62 gc.enable()
63 return f
