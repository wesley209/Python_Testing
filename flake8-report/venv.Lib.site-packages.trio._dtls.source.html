147 # DTLS records are:
148 # - 1 byte content type
149 # - 2 bytes version
150 # - 8 bytes epoch+seqno
151 # Technically this is 2 bytes epoch then 6 bytes seqno, but we treat it as
152 # a single 8-byte integer, where epoch changes are represented as jumping
153 # forward by 2**(6*8).
154 # - 2 bytes payload length (unsigned big-endian)
155 # - payload
156 RECORD_HEADER = struct.Struct("!B2sQH")

259 def decode_client_hello_untrusted(packet: bytes) -> tuple[int, bytes, bytes]:
260 # Raises BadPacket if parsing fails
261 # Returns (record epoch_seqno, cookie from the packet, data that should be
262 # hashed into cookie)
263 try:
264 # ClientHello has to be the first record in the packet
265 record = next(records_untrusted(packet))
266 # no-cover because at time of writing, this is unreachable:
267 # decode_client_hello_untrusted is only called on packets that have passed
268 # is_client_hello_untrusted, which confirms the content type.
269 if record.content_type != ContentType.handshake: # pragma: no cover
270 raise BadPacket("not a handshake record")
271 fragment = decode_handshake_fragment_untrusted(record.payload)
272 if fragment.msg_type != HandshakeType.client_hello:
273 raise BadPacket("not a ClientHello")
274 # ClientHello can't be fragmented, because reassembly requires holding
275 # per-connection state, and we refuse to allocate per-connection state
276 # until after we get a valid ClientHello.
277 if fragment.frag_offset != 0:
278 raise BadPacket("fragmented ClientHello")
279 if fragment.frag_len != fragment.msg_len:
280 raise BadPacket("fragmented ClientHello")
281  
282 # As per RFC 6347:
283 #
284 # When responding to a HelloVerifyRequest, the client MUST use the
285 # same parameter values (version, random, session_id, cipher_suites,
286 # compression_method) as it did in the original ClientHello. The
287 # server SHOULD use those values to generate its cookie and verify that
288 # they are correct upon cookie receipt.
289 #
290 # However, the record-layer framing can and will change (e.g. the
291 # second ClientHello will have a new record-layer sequence number). So
292 # we need to pull out the handshake message alone, discarding the
293 # record-layer stuff, and then we're going to hash all of it *except*
294 # the cookie.
295  
296 body = fragment.frag
297 # ClientHello is:
298 #
299 # - 2 bytes client_version
300 # - 32 bytes random
301 # - 1 byte session_id length
302 # - session_id
303 # - 1 byte cookie length
304 # - cookie
305 # - everything else
306 #
307 # So to find the cookie, so we need to figure out how long the
308 # session_id is and skip past it.
309 session_id_len = body[2 + 32]
310 cookie_len_offset = 2 + 32 + 1 + session_id_len
311 cookie_len = body[cookie_len_offset]
312  
313 cookie_start = cookie_len_offset + 1
314 cookie_end = cookie_start + cookie_len
315  
316 before_cookie = body[:cookie_len_offset]
317 cookie = body[cookie_start:cookie_end]
318 after_cookie = body[cookie_end:]
319  
320 if len(cookie) != cookie_len:
321 raise BadPacket("short cookie")
322 return (record.epoch_seqno, cookie, before_cookie + after_cookie)
323  
324 except (struct.error, IndexError) as exc:
325 raise BadPacket("bad ClientHello") from exc

362 # This takes a raw outgoing handshake volley that openssl generated, and
363 # reconstructs the handshake messages inside it, so that we can repack them
364 # into records while retransmitting. So the data ought to be well-behaved --
365 # it's not coming from the network.
366 def decode_volley_trusted(
367 volley: bytes,
368 ) -> list[_AnyHandshakeMessage]:
369 messages: list[_AnyHandshakeMessage] = []
370 messages_by_seq = {}
371 for record in records_untrusted(volley):
372 # ChangeCipherSpec isn't a handshake message, so it can't be fragmented.
373 # Handshake messages with epoch > 0 are encrypted, so we can't fragment them
374 # either. Fortunately, ChangeCipherSpec has a 1 byte payload, and the only
375 # encrypted handshake message is Finished, whose payload is a single hash value
376 # -- so 32 bytes for SHA-256, 64 for SHA-512, etc. Neither is going to be so
377 # large that it has to be fragmented to fit into a single packet.
378 if record.epoch_seqno & EPOCH_MASK:
379 messages.append(OpaqueHandshakeMessage(record))
380 elif record.content_type in (ContentType.change_cipher_spec, ContentType.alert):
381 messages.append(
382 PseudoHandshakeMessage(
383 record.version,
384 record.content_type,
385 record.payload,
386 ),
387 )
388 else:
389 assert record.content_type == ContentType.handshake
390 fragment = decode_handshake_fragment_untrusted(record.payload)
391 msg_type = HandshakeType(fragment.msg_type)
392 if fragment.msg_seq not in messages_by_seq:
393 msg = HandshakeMessage(
394 record.version,
395 msg_type,
396 fragment.msg_seq,
397 bytearray(fragment.msg_len),
398 )
399 messages.append(msg)
400 messages_by_seq[fragment.msg_seq] = msg
401 else:
402 msg = messages_by_seq[fragment.msg_seq]
403 assert msg.msg_type == fragment.msg_type
404 assert msg.msg_seq == fragment.msg_seq
405 assert len(msg.body) == fragment.msg_len
406  
407 msg.body[
  • E203 Whitespace before ':'
408 fragment.frag_offset : fragment.frag_offset + fragment.frag_len
409 ] = fragment.frag
410  
411 return messages

497 # This bit requires implementing a bona fide cryptographic protocol, so even though it's
498 # a simple one let's take a moment to discuss the design.
499 #
500 # Our goal is to force new incoming handshakes that claim to be coming from a
501 # given ip:port to prove that they can also receive packets sent to that
502 # ip:port. (There's nothing in UDP to stop someone from forging the return
503 # address, and it's often used for stuff like DoS reflection attacks, where
504 # an attacker tries to trick us into sending data at some innocent victim.)
505 # For more details, see:
506 #
507 # https://datatracker.ietf.org/doc/html/rfc6347#section-4.2.1
508 #
509 # To do this, when we receive an initial ClientHello, we calculate a magic
510 # cookie, and send it back as a HelloVerifyRequest. Then the client sends us a
511 # second ClientHello, this time with the magic cookie included, and after we
512 # check that this cookie is valid we go ahead and start the handshake proper.
513 #
514 # So the magic cookie needs the following properties:
515 # - No-one can forge it without knowing our secret key
516 # - It ensures that the ip, port, and ClientHello contents from the response
517 # match those in the challenge
518 # - It expires after a short-ish period (so that if an attacker manages to steal one, it
519 # won't be useful for long)
520 # - It doesn't require storing any peer-specific state on our side
521 #
522 # To do that, we take the ip/port/ClientHello data and compute an HMAC of them, using a
523 # secret key we generate on startup. We also include:
524 #
525 # - The current time (using Trio's clock), rounded to the nearest 30 seconds
526 # - A random salt
527 #
528 # Then the cookie is the salt and the HMAC digest concatenated together.
529 #
530 # When verifying a cookie, we use the salt + new ip/port/ClientHello data to recompute
531 # the HMAC digest, for both the current time and the current time minus 30 seconds, and
532 # if either of them match, we consider the cookie good.
533 #
534 # Including the rounded-off time like this means that each cookie is good for at least
535 # 30 seconds, and possibly as much as 60 seconds.
536 #
537 # The salt is probably not necessary -- I'm pretty sure that all it does is make it hard
538 # for an attacker to figure out when our clock ticks over a 30 second boundary. Which is
539 # probably pretty harmless? But it's easier to add the salt than to convince myself that
540 # it's *completely* harmless, so, salt it is.
541  
542 COOKIE_REFRESH_INTERVAL = 30 # seconds
543 KEY_BYTES = 32
544 COOKIE_HASH = "sha256"
545 SALT_BYTES = 8
546 # 32 bytes was the maximum cookie length in DTLS 1.0. DTLS 1.2 raised it to 255. I doubt
547 # there are any DTLS 1.0 implementations still in the wild, but really 32 bytes is
548 # plenty, and it also gets rid of a confusing warning in Wireshark output.
549 #
550 # We truncate the cookie to 32 bytes, of which 8 bytes is salt, so that leaves 24 bytes
551 # of truncated HMAC = 192 bit security, which is still massive overkill. (TCP uses 32
552 # *bits* for this.) HMAC truncation is explicitly noted as safe in RFC 2104:
553 # https://datatracker.ietf.org/doc/html/rfc2104#section-5
554 COOKIE_LENGTH = 32

558 return int(trio.current_time() / COOKIE_REFRESH_INTERVAL)

561 # Simple deterministic and invertible serializer -- i.e., a useful tool for converting

564 out = []

567 out.append(field)

571 def _make_cookie(
572 key: bytes,
573 salt: bytes,
574 tick: int,
575 address: Any,
576 client_hello_bits: bytes,
577 ) -> bytes:
578 assert len(salt) == SALT_BYTES
579 assert len(key) == KEY_BYTES
580  
581 signable_data = _signable(
582 salt,
583 struct.pack("!Q", tick),
584 # address is a mix of strings and ints, and variable length, so pack
585 # it into a single nested field
586 _signable(*(str(part).encode() for part in address)),
587 client_hello_bits,
588 )
589  
590 return (salt + hmac.digest(key, signable_data, COOKIE_HASH))[:COOKIE_LENGTH]

593 def valid_cookie(
594 key: bytes,
595 cookie: bytes,
596 address: Any,
597 client_hello_bits: bytes,
598 ) -> bool:
599 if len(cookie) > SALT_BYTES:
600 salt = cookie[:SALT_BYTES]
601  
602 tick = _current_cookie_tick()
603  
604 cur_cookie = _make_cookie(key, salt, tick, address, client_hello_bits)
605 old_cookie = _make_cookie(
606 key,
607 salt,
608 max(tick - 1, 0),
609 address,
610 client_hello_bits,
611 )
612  
613 # I doubt using a short-circuiting 'or' here would leak any meaningful
614 # information, but why risk it when '|' is just as easy.
615 return hmac.compare_digest(cookie, cur_cookie) | hmac.compare_digest(
616 cookie,
617 old_cookie,
618 )
619 else:
620 return False

623 def challenge_for(
624 key: bytes,
625 address: Any,
626 epoch_seqno: int,
627 client_hello_bits: bytes,
628 ) -> bytes:
629 salt = os.urandom(SALT_BYTES)
630 tick = _current_cookie_tick()
631 cookie = _make_cookie(key, salt, tick, address, client_hello_bits)
632  
633 # HelloVerifyRequest body is:
634 # - 2 bytes version
635 # - length-prefixed cookie
636 #
637 # The DTLS 1.2 spec says that for this message specifically we should use
638 # the DTLS 1.0 version.
639 #
640 # (It also says the opposite of that, but that part is a mistake:
641 # https://www.rfc-editor.org/errata/eid4103
642 # ).
643 #
644 # And I guess we use this for both the message-level and record-level
645 # ProtocolVersions, since we haven't negotiated anything else yet?
646 body = ProtocolVersion.DTLS10 + bytes([len(cookie)]) + cookie
647  
648 # RFC says have to copy the client's record number
649 # Errata says it should be handshake message number
650 # Openssl copies back record sequence number, and always sets message seq
651 # number 0. So I guess we'll follow openssl.
652 hs = HandshakeFragment(
653 msg_type=HandshakeType.hello_verify_request,
654 msg_len=len(body),
655 msg_seq=0,
656 frag_offset=0,
657 frag_len=len(body),
658 frag=body,
659 )
660 payload = encode_handshake_fragment(hs)
661  
662 packet = encode_record(
663 Record(ContentType.handshake, ProtocolVersion.DTLS10, epoch_seqno, payload),
664 )
665 return packet

687 async def handle_client_hello_untrusted(
688 endpoint: DTLSEndpoint,
689 address: Any,
690 packet: bytes,
691 ) -> None:
692 # it's trivial to write a simple function that directly calls this to
693 # get code coverage, but it should maybe:
694 # 1. be removed
695 # 2. be asserted
696 # 3. Write a complicated test case where this happens "organically"
697 if endpoint._listening_context is None: # pragma: no cover
698 return
699  
700 try:
701 epoch_seqno, cookie, bits = decode_client_hello_untrusted(packet)
702 except BadPacket:
703 return
704  
705 if endpoint._listening_key is None:
706 endpoint._listening_key = os.urandom(KEY_BYTES)
707  
708 if not valid_cookie(endpoint._listening_key, cookie, address, bits):
709 challenge_packet = challenge_for(
710 endpoint._listening_key,
711 address,
712 epoch_seqno,
713 bits,
714 )
715 try:
716 async with endpoint._send_lock:
717 await endpoint.socket.sendto(challenge_packet, address)
718 except (OSError, trio.ClosedResourceError):
719 pass
720 else:
721 # We got a real, valid ClientHello!
722 stream = DTLSChannel._create(endpoint, address, endpoint._listening_context)
723 # Our HelloRetryRequest had some sequence number. We need our future sequence
724 # numbers to be larger than it, so our peer knows that our future records aren't
725 # stale/duplicates. But, we don't know what this sequence number was. What we do
726 # know is:
727 # - the HelloRetryRequest seqno was copied it from the initial ClientHello
728 # - the new ClientHello has a higher seqno than the initial ClientHello
729 # So, if we copy the new ClientHello's seqno into our first real handshake
730 # record and increment from there, that should work.
731 stream._record_encoder.set_first_record_number(epoch_seqno)
732 # Process the ClientHello
733 try:
734 stream._ssl.bio_write(packet)
735 stream._ssl.DTLSv1_listen()
736 except SSL.Error: # pragma: no cover
737 # ...OpenSSL didn't like it, so I guess we didn't have a valid ClientHello
738 # after all.
739 return
740  
741 # Some old versions of OpenSSL have a bug with memory BIOs, where DTLSv1_listen
742 # consumes the ClientHello out of the BIO, but then do_handshake expects the
743 # ClientHello to still be in there (but not the one that ships with Ubuntu
744 # 20.04). In particular, this is known to affect the OpenSSL v1.1.1 that ships
745 # with Ubuntu 18.04. To work around this, we deliver a second copy of the
746 # ClientHello after DTLSv1_listen has completed. This is safe to do
747 # unconditionally, because on newer versions of OpenSSL, the second ClientHello
748 # is treated as a duplicate packet, which is a normal thing that can happen over
749 # UDP. For more details, see:
750 #
751 # https://github.com/pyca/pyopenssl/blob/e84e7b57d1838de70ab7a27089fbee78ce0d2106/tests/test_ssl.py#L4226-L4293
752 #
753 # This was fixed in v1.1.1a, and all later versions. So maybe in 2024 or so we
754 # can delete this. The fix landed in OpenSSL master as 079ef6bd534d2, and then
755 # was backported to the 1.1.1 branch as d1bfd8076e28.
756 stream._ssl.bio_write(packet)
757  
758 # Check if we have an existing association
759 old_stream = endpoint._streams.get(address)
760 if old_stream is not None:
761 if old_stream._client_hello == (cookie, bits):
762 # ...This was just a duplicate of the last ClientHello, so never mind.
763 return
764 else:
765 # Ok, this *really is* a new handshake; the old stream should go away.
766 old_stream._set_replaced()
767 stream._client_hello = (cookie, bits)
768 endpoint._streams[address] = stream
769 endpoint._incoming_connections_q.s.send_nowait(stream)

772 async def dtls_receive_loop(
773 endpoint_ref: ReferenceType[DTLSEndpoint],
774 sock: SocketType,
775 ) -> None:
776 try:
777 while True:
778 try:
779 packet, address = await sock.recvfrom(MAX_UDP_PACKET_SIZE)
780 except OSError as exc:
781 if exc.errno == errno.ECONNRESET:
782 # Windows only: "On a UDP-datagram socket [ECONNRESET]
783 # indicates a previous send operation resulted in an ICMP Port
784 # Unreachable message" -- https://docs.microsoft.com/en-us/windows/win32/api/winsock/nf-winsock-recvfrom
785 #
786 # This is totally useless -- there's nothing we can do with this
787 # information. So we just ignore it and retry the recv.
788 continue
789 else:
790 raise
791 endpoint = endpoint_ref()
792 try:
793 if endpoint is None:
794 return
795 if is_client_hello_untrusted(packet):
796 await handle_client_hello_untrusted(endpoint, address, packet)
797 elif address in endpoint._streams:
798 stream = endpoint._streams[address]
799 if stream._did_handshake and part_of_handshake_untrusted(packet):
800 # The peer just sent us more handshake messages, that aren't a
801 # ClientHello, and we thought the handshake was done. Some of
802 # the packets that we sent to finish the handshake must have
803 # gotten lost. So re-send them. We do this directly here instead
804 # of just putting it into the queue and letting the receiver do
805 # it, because there's no guarantee that anyone is reading from
806 # the queue, because we think the handshake is done!
807 await stream._resend_final_volley()
808 else:
809 try:
810 # mypy for some reason cannot determine type of _q
811 stream._q.s.send_nowait(packet) # type:ignore[has-type]
812 except trio.WouldBlock:
813 stream._packets_dropped_in_trio += 1
814 else:
815 # Drop packet
816 pass
817 finally:
818 del endpoint
819 except trio.ClosedResourceError:
820 # socket was closed
821 return
822 except OSError as exc:
823 if exc.errno in (errno.EBADF, errno.ENOTSOCK):
824 # socket was closed
825 return
826 else: # pragma: no cover
827 # ??? shouldn't happen
828 raise

969 async def do_handshake(self, *, initial_retransmit_timeout: float = 1.0) -> None:
970 """Perform the handshake.

976 specifically.

979 tasks the first call will perform the handshake, and the rest will be no-ops.

986 handshake packets that don't receive a response. This lets you set the

988 times the round-trip time to your peer, but 1 second is a reasonable
989 default. There's `some useful guidance here
990 <https://tlswg.org/dtls13-spec/draft-ietf-tls-dtls13.html#name-timer-values>`__.

1142 def set_ciphertext_mtu(self, new_mtu: int) -> None:
1143 """Tells Trio the `largest amount of data that can be sent in a single packet to
1144 this peer <https://en.wikipedia.org/wiki/Maximum_transmission_unit>`__.
1145  
1146 Trio doesn't actually enforce this limit if you pass a huge packet to `send`,
1147 then we'll dutifully encrypt it and attempt to send it. But calling this method
1148 does have two useful effects:
1149  
1150 - If called before the handshake is performed, then Trio will automatically
1151 fragment handshake messages to fit within the given MTU. It also might
1152 fragment them even smaller, if it detects signs of packet loss, so setting
1153 this should never be necessary to make a successful connection. But, the
1154 packet loss detection only happens after multiple timeouts have expired, so if
1155 you have reason to believe that a smaller MTU is required, then you can set
1156 this to skip those timeouts and establish the connection more quickly.
1157  
1158 - It changes the value returned from `get_cleartext_mtu`. So if you have some
1159 kind of estimate of the network-level MTU, then you can use this to figure out
1160 how much overhead DTLS will need for hashes/padding/etc., and how much space
1161 you have left for your application data.
1162  
1163 The MTU here is measuring the largest UDP *payload* you think can be sent, the
1164 amount of encrypted data that can be handed to the operating system in a single
1165 call to `send`. It should *not* include IP/UDP headers. Note that OS estimates
1166 of the MTU often are link-layer MTUs, so you have to subtract off 28 bytes on
1167 IPv4 and 48 bytes on IPv6 to get the ciphertext MTU.
1168  
1169 By default, Trio assumes an MTU of 1472 bytes on IPv4, and 1452 bytes on IPv6,
1170 which correspond to the common Ethernet MTU of 1500 bytes after accounting for
1171 IP/UDP overhead.
1172  
1173 """
1174 self._handshake_mtu = new_mtu
1175 self._ssl.set_ciphertext_mtu(new_mtu)

1371 def connect(
1372 self,
1373 address: tuple[str, int],
1374 ssl_context: SSL.Context,
1375 ) -> DTLSChannel:
1376 """Initiate an outgoing DTLS connection.
1377  
1378 Notice that this is a synchronous method. That's because it doesn't actually
1379 initiate any I/O it just sets up a `DTLSChannel` object. The actual handshake
1380 doesn't occur until you start using the `DTLSChannel`. This gives you a chance
1381 to do further configuration first, like setting MTU etc.
1382  
1383 Args:
1384 address: The address to connect to. Usually a (host, port) tuple, like
1385 ``("127.0.0.1", 12345)``.
1386 ssl_context (OpenSSL.SSL.Context): The PyOpenSSL context object to use for
1387 this connection.
1388  
1389 Returns:
1390 DTLSChannel
1391  
1392 """
1393 # it would be nice if we could detect when 'address' is our own endpoint (a
1394 # loopback connection), because that can't work
1395 # but I don't see how to do it reliably
1396 self._check_closed()
1397 channel = DTLSChannel._create(self, address, ssl_context)
1398 channel._ssl.set_connect_state()
1399 old_channel = self._streams.get(address)
1400 if old_channel is not None:
1401 old_channel._set_replaced()
1402 self._streams[address] = channel
1403 return channel
