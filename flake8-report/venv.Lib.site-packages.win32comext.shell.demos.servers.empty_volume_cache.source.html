118 def GetSpaceUsed(self, callback):
119 total = [0] # See _WalkCallback above
120 try:
121 for d in self._GetDirectories():
122 os.path.walk(d, self._WalkCallback, (callback, total))
123 print("After looking in", d, "we have", total[0], "bytes")
124 except pythoncom.error as exc:
125 # This will be raised by the callback when the user selects 'cancel'.
126 if exc.hresult != winerror.E_ABORT:
127 raise # that's the documented error code!
128 print("User cancelled the operation")
129 return total[0]
