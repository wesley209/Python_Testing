66 def wait_for_pending_cleanups(self,
67 initial_active_threads=None,
68 initial_main_greenlets=None):
69 initial_active_threads = initial_active_threads or self.threads_before_test
70 initial_main_greenlets = initial_main_greenlets or self.main_greenlets_before_test
71 sleep_time = self.cleanup_attempt_sleep_duration
72 # NOTE: This is racy! A Python-level thread object may be dead
73 # and gone, but the C thread may not yet have fired its
74 # destructors and added to the queue. There's no particular
75 # way to know that's about to happen. We try to watch the
76 # Python threads to make sure they, at least, have gone away.
77 # Counting the main greenlets, which we can easily do deterministically,
78 # also helps.
79  
80 # Always sleep at least once to let other threads run
81 sleep(sleep_time)
82 quit_after = time() + self.cleanup_max_sleep_seconds
83 # TODO: We could add an API that calls us back when a particular main greenlet is deleted?
84 # It would have to drop the GIL
85 while (
86 get_pending_cleanup_count()
87 or active_thread_count() > initial_active_threads
88 or (not self.expect_greenlet_leak
89 and get_total_main_greenlets() > initial_main_greenlets)):
90 sleep(sleep_time)
91 if time() > quit_after:
92 print("Time limit exceeded.")
93 print("Threads: Waiting for only", initial_active_threads,
94 "-->", active_thread_count())
95 print("MGlets : Waiting for only", initial_main_greenlets,
96 "-->", get_total_main_greenlets())

146 def get_expected_returncodes_for_aborted_process(self):
147 import signal
148 # The child should be aborted in an unusual way. On POSIX
149 # platforms, this is done with abort() and signal.SIGABRT,
150 # which is reflected in a negative return value; however, on
151 # Windows, even though we observe the child print "Fatal
152 # Python error: Aborted" and in older versions of the C
153 # runtime "This application has requested the Runtime to
154 # terminate it in an unusual way," it always has an exit code
155 # of 3. This is interesting because 3 is the error code for
156 # ERROR_PATH_NOT_FOUND; BUT: the C runtime abort() function
157 # also uses this code.
158 #
159 # If we link to the static C library on Windows, the error
160 # code changes to '0xc0000409' (hex(3221226505)), which
161 # apparently is STATUS_STACK_BUFFER_OVERRUN; but "What this
162 # means is that nowadays when you get a
163 # STATUS_STACK_BUFFER_OVERRUN, it doesnâ€™t actually mean that
164 # there is a stack buffer overrun. It just means that the
165 # application decided to terminate itself with great haste."
166 #
167 #
168 # On windows, we've also seen '0xc0000005' (hex(3221225477)).
169 # That's "Access Violation"
170 #
171 # See
172 # https://devblogs.microsoft.com/oldnewthing/20110519-00/?p=10623
173 # and
174 # https://docs.microsoft.com/en-us/previous-versions/k089yyh0(v=vs.140)?redirectedfrom=MSDN
175 # and
176 # https://devblogs.microsoft.com/oldnewthing/20190108-00/?p=100655
177 expected_exit = (
178 -signal.SIGABRT,
179 # But beginning on Python 3.11, the faulthandler
180 # that prints the C backtraces sometimes segfaults after
181 # reporting the exception but before printing the stack.
182 # This has only been seen on linux/gcc.
183 -signal.SIGSEGV,
184 ) if not WIN else (
185 3,
186 0xc0000409,
187 0xc0000005,
188 )
189 return expected_exit
