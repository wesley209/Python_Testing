15 # This is a specialized TestCase adaptor which wraps a real test.
16 class LeakTestCase(unittest.TestCase):
17 """An 'adaptor' which takes another test. In debug builds we execute the
18 test once to remove one-off side-effects, then capture the total
19 reference count, then execute the test a few times. If the total
20 refcount at the end is greater than we first captured, we have a leak!
21  
22 In release builds the test is executed just once, as normal.
23  
24 Generally used automatically by the test runner - you can safely
25 ignore this.
26 """
27  
28 def __init__(self, real_test):
29 unittest.TestCase.__init__(self)
30 self.real_test = real_test
31 self.num_test_cases = 1
32 self.num_leak_iters = 2 # seems to be enough!
33 if hasattr(sys, "gettotalrefcount"):
34 self.num_test_cases += self.num_leak_iters
35  
36 def countTestCases(self):
37 return self.num_test_cases

39 def __call__(self, result=None):
40 # For the COM suite's sake, always ensure we don't leak
41 # gateways/interfaces
42 from pythoncom import _GetGatewayCount, _GetInterfaceCount
43  
44 gc.collect()
45 ni = _GetInterfaceCount()
46 ng = _GetGatewayCount()
47 self.real_test(result)
48 # Failed - no point checking anything else
49 if result.shouldStop or not result.wasSuccessful():
50 return
51 self._do_leak_tests(result)
52 gc.collect()
53 lost_i = _GetInterfaceCount() - ni
54 lost_g = _GetGatewayCount() - ng
55 if lost_i or lost_g:
56 msg = "%d interface objects and %d gateway objects leaked" % (
57 lost_i,
58 lost_g,
59 )
60 exc = AssertionError(msg)
61 result.addFailure(self.real_test, (exc.__class__, exc, None))

64 raise NotImplementedError("not used")

66 def _do_leak_tests(self, result=None):
67 try:
68 gtrc = sys.gettotalrefcount
69 except AttributeError:
70 return # can't do leak tests in this build
71 # Assume already called once, to prime any caches etc
72 gc.collect()
73 trc = gtrc()
74 for i in range(self.num_leak_iters):
75 self.real_test(result)
76 if result.shouldStop:
77 break
78 del i # created after we remembered the refcount!
79 # int division here means one or 2 stray references won't force
80 # failure, but one per loop
81 gc.collect()
82 lost = (gtrc() - trc) // self.num_leak_iters
83 if lost < 0:
84 msg = "LeakTest: %s appeared to gain %d references!!" % (
85 self.real_test,
86 -lost,
87 )
88 result.addFailure(self.real_test, (AssertionError, msg, None))
89 if lost > 0:
90 msg = "LeakTest: %s lost %d references" % (self.real_test, lost)
91 exc = AssertionError(msg)
92 result.addFailure(self.real_test, (exc.__class__, exc, None))

172 # Find a test "fixture" (eg, binary test file) expected to be very close to
173 # the test being run.
174 # If the tests are being run from the "installed" version, then these fixtures
175 # probably don't exist - the test is "skipped".
176 # But it's fatal if we think we might be running from a pywin32 source tree.
177 def find_test_fixture(basename, extra_dir="."):
178 # look for the test file in various places
179 candidates = [
180 os.path.dirname(sys.argv[0]),
181 extra_dir,
182 ".",
183 ]
184 for candidate in candidates:
185 fname = os.path.join(candidate, basename)
186 if os.path.isfile(fname):
187 return fname
188 else:
189 # Can't find it - see if this is expected or not.
190 # This module is typically always in the installed dir, so use argv[0]
191 this_file = os.path.normcase(os.path.abspath(sys.argv[0]))
192 dirs_to_check = site.getsitepackages()[:]
193 if site.USER_SITE:
194 dirs_to_check.append(site.USER_SITE)
195  
196 for d in dirs_to_check:
197 d = os.path.normcase(d)
198 if os.path.commonprefix([this_file, d]) == d:
199 # looks like we are in an installed Python, so skip the text.
200 raise TestSkipped(f"Can't find test fixture '{fname}'")
201 # Looks like we are running from source, so this is fatal.
202 raise RuntimeError(f"Can't find test fixture '{fname}'")

210 # The 'TestResult' subclass that records the failures and has the special
211 # handling for the TestSkipped exception.
212 class TestResult(unittest.TextTestResult):
213 def __init__(self, *args, **kw):
214 super().__init__(*args, **kw)
215 self.skips = {} # count of skips for each reason.

217 def addError(self, test, err):
218 """Called when an error has occurred. 'err' is a tuple of values as
219 returned by sys.exc_info().
220 """
221 # translate a couple of 'well-known' exceptions into 'skipped'
222 import pywintypes
223  
224 exc_val = err[1]
225 # translate ERROR_ACCESS_DENIED for non-admin users to be skipped.
226 # (access denied errors for an admin user aren't expected.)
227 if (
228 isinstance(exc_val, pywintypes.error)
229 and exc_val.winerror in non_admin_error_codes
230 and not check_is_admin()
231 ):
232 exc_val = TestSkipped(exc_val)
233 # and COM errors due to objects not being registered (the com test
234 # suite will attempt to catch this and handle it itself if the user
235 # is admin)
236 elif isinstance(exc_val, pywintypes.com_error) and exc_val.hresult in [
237 winerror.CO_E_CLASSSTRING,
238 winerror.REGDB_E_CLASSNOTREG,
239 winerror.TYPE_E_LIBNOTREGISTERED,
240 ]:
241 exc_val = TestSkipped(exc_val)
242 # NotImplemented generally means the platform doesn't support the
243 # functionality.
244 elif isinstance(exc_val, NotImplementedError):
245 exc_val = TestSkipped(NotImplementedError)
246  
247 if isinstance(exc_val, TestSkipped):
248 reason = exc_val.args[0]
249 # if the reason itself is another exception, get its args.
250 try:
251 reason = tuple(reason.args)
252 except (AttributeError, TypeError):
253 pass
254 self.skips.setdefault(reason, 0)
255 self.skips[reason] += 1
256 if self.showAll:
257 self.stream.writeln(f"SKIP ({reason})")
258 elif self.dots:
259 self.stream.write("S")
260 self.stream.flush()
261 return
262 super().addError(test, err)
263  
264 def printErrors(self):
265 super().printErrors()
266 for reason, num_skipped in self.skips.items():
267 self.stream.writeln("SKIPPED: %d tests - %s" % (num_skipped, reason))
